var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=915&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

    }

}

function doCommentSubmit()

                $("#cmtTextarea").val('');

 <h1><b>1.가자미 데리야끼 조림 </b>
 <span>by.요리쿡조리쿡</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>중급 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/02/02735ad26ea525c8649e2638f15345e81.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>반건 가자미 2마리, 양파 1개, 청 홍고추 각 2개씩(큰건 1개씩)</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>데리야끼소스 6큰술 (데리야끼소스) : 간장 8큰술, 식초 3큰술, 설탕 4큰술, 청주 4큰술, 다진마늘 1큰술, 참기름 1큰술, 올리고당 1큰술, 생강가루 0.5작은술, 물 1/4컵</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>꼬들꼬들하게 잘 마른 반건조 가자미... 생가자미보다 맛이 더 쫄깃하니 구워도 더 맛이 좋은거 같아요.</li>
 
 <li>손질이 잘 된 가자미라 양념이 잘 배이게 칼집만 3~4군데 내 주었답니다.</li>
 
 <li>양파는 반을 갈라 채를 썰어주시고, 고추는 어슷썰어주세요. 청양고추를 넣어주면 칼칼한 맛이 또 좋겠다 싶지요. 요부분은 아이들때문에 패쓰~~</li>
 
 <li>데리야끼소스 6큰술 준비했구요... 데리야끼소스를 몇배수로 만들어서 냄비에 넣고 부글부글 한번 끓여서 식혀준 다음, 냉장고에 넣고 사용하시면 만능양념장처럼 요리에 요긴하게 사용할 수 있어 편하답니다.</li>
 
 <li>달군팬에 가자미를 올려서 앞뒤로 노릇노릇하게 구워주세요. 생선을 구우실때는 빨리 뒤집으려고 하면 눌러붙기 쉬우니 중약불에서 기름을 최소화 하고, 서서히 구워서 시간을 두고 한두번만 뒤집어야 타지않고 담백하니 잘 구워진답니다.</li>
 
 <li>노릇노릇 구워진 가자미 위에 채썬 양파와 청홍고추를 올리고 데리야끼소스를 6큰술을 골고루 뿌려줍니다. 기호에 따라 소스양은 가감하셔요.</li>
 
 <li>중간에 한두번 이렇게 소스가 생선살에 스며들게 국물을 위에 뿌려 주면서 졸여줍니다. 다 익은 생선이니 잠깐동안만 중약불에서 졸여주시면 되요.</li>
 
 <li>국물이 걸쭉해지게 졸여지면 접시에 담아내시면 된답니다.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>2.가자미 무조림</b>
 <span>by.과일짱</span>
 <div class="recipe_info">
 <b>분량 :</b>4인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>중급 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/05/28/bde6f60fbe4d6290e1ccf6b3c17c01b51.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>가자미 4마리, 양파 1/2개, 청홍고추 약간</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>조림 양념장 - 무우 1/4, 간장 4, 고추장 1, 고추가루 2, 물엿 1, 생강즙, 마늘, 후추, 참기름, 생수1컵 (뒤포리육수면 더 좋아요)</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>머리와 꼬리를 떼어내고 먹기좋은 크기로 잘라주세요.</li>
 
 <li>각종 채소를 썰어 준비합니다.</li>
 
 <li>조림 양념장을 준비합니다.</li>
 
 <li>냄비에 무와 양파를 깔고 가자미를 올려주어요.</li>
 
 <li>양념장을 부어주고 센불로 끓이다가 중불로 줄여 무까지 익을수 있도록 은근하게 졸여주세요.</li>
 
 <li>중간에 뚜껑을 열고 국물을 끼얹어줘 가면서 졸여주시면 좋아요. 간을 보고 싱겁다 싶으면 간장이나 소금을 추가로 넣으시면 되되고, 짜다 싶으면 물을 조금 더 추가로 넣으시면 돼요.</li>
 
 <li>마지막으로 청홍고추 넣어주고 한번 우르르 끓으면 됩니다.</li>
 </ol>
 </dd>

 <h1><b>3.가자미 튀김 만들기</b>
 <span>by.봉자바리</span>
 <div class="recipe_info">
 <b>분량 :</b>4인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>초급 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/08/99abe501215cb30ed1ca4cba15bf9463.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>가자미 1마리, 밀가루 약간</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>간장 2큰술, 설탕1½큰술, 식초1½큰술, 물1큰술, 마늘, 대파</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>마늘은 편썰어서 물에 담궈놓습니다. 그럼 마늘의 전분도 제거되고 매운맛도 제거됩니다. 대파는 총총 썰어서 준비해주시고. 가자미는 큰놈으로 한마리 준비했답니다.</li>
 
 <li>손질되어있는 가자미라서 딱히 따로 손질할건 없었구요.. 흐르는물에 깨끗히 씻어서 먹기좋게 잘라 키친타올에서 물기를 톡톡 털어줍니다.</li>
 
 <li>물기 제거한 가자미에 밀가루를 톡톡 옷입혀 주세요.</li>
 
 <li>오일을 팔팔 온도를 높여서 아주 뜨거울때 가자미를 넣고 튀겨주셔야 합니다. 안그러면 가자미가 흐물흐물 다 부서져 버려요. 고온의 기름에서 튀겨주세요 꼭</li>
 
 <li>&lt;소스만들기&gt; 1. 팬에 기름을 살짤두르고 편썰어 놓은 마늘을 볶아 줍니다. 바싹 볶아 주세요~</li>
 
 <li>바싹 볶아진 마늘에 소스의 재료를 몽땅 넣고 끓여줍니다. 간장 2큰술, 설탕1½큰술, 식초1½큰술, 물1큰술 대파만 빼고요. 소스가 어우러지면 대파를 넣고 농도있게 끓여내 줍니다. 간장에 따라서 짠기가 다르니 간장양은 조절해주세요</li>
 
 <li>가지마 올려놓고 소스를 위에서 주루룩~뿌려내 줍니다.</li>
 </ol>
 </dd>
