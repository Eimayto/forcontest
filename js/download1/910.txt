var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=910&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

    }

}

function doCommentSubmit()

                $("#cmtTextarea").val('');

 <h1><b>1.간단한 한끼 고소한 들깨수제비</b>
 <span>by.아콰마린</span>
 <div class="recipe_info">
 <b>분량 :</b>1인분 <b>난이도 :</b>아무나 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/03/30/bd407d369c8d0789522c5a7d5dd9ecdf1.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>밀가루 2컵, 물 2/3컵, 소금 1꼬집, 식용유 약간, 감자 1개, 표고버섯 2개, 깻잎 5장, 당근 약간, 대파 약간</dd>
 </dl>

 <li>간단하지만 든든히 한끼 해결할수 있는 들깨수제비 완성!</li>

 <h1><b>2.별미 바지락수제비</b>
 <span>by.예예</span>
 <div class="recipe_info">

 <li>완성된 반죽은 비닐봉지에 담아 육수를 만들동안 냉장실에서 숙성시켜줄거에요.</li>
 
 <li>먼저 냄비에 분량의 육수 재료를 모두 넣고 팔팔 끓여 육수를 만들어 주세요.</li>
 
 <li>끓어오르면 건더기를 건져내고 바지락을 넣어 바지락이 입벌어질때까지만 살짝 끓입니다.</li>
 
 <li>바지락은 금새 익어 입을 벌릴거에요. 그러면 불을 끄고 바지락을 모두 건져냅니다. 바지락을 건져낸 육수는 면보에 걸러 깨끗하고 맑은 육수만 받아내세요~</li>
 
 <li>칼국수에 넣을 양파와 당근은 채썰어 준비하시구여. 대파는 어슷썰어 준비합니다.</li>
 
 <li>걸러낸 깨끗한 육수는 냄비에 담고 채썬 양파와 당근을 먼저 넣어줍니다.</li>
 
 <li>수제비 반죽을 꺼내어 얇게 펼처 떼어넣어 주세요</li>
 
 <li>먹을만큼 반죽을 떼어 넣은후 반죽이 익었으면 마지막으로 익혀낸 바지락과 대파를 넣어주세요. 국간장과 소금으로 입맛에 맞게 간을 해줍니다. 한소끔만 더 끓여내면 바지락수제비가 완성이에요.</li>
 </ol>

 <h1><b>3.바지락 얼큰 수제비 만들기</b>
 <span>by.예쁜포비</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>초급 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/07/09/f159bd950a59f7095fce170f834cd3c5.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>밀가루(수제비 반죽 밀가루) 250g, 달걀 1개, 물 6큰술, 애호박 1/4개, 바지락 1봉지, 양파 1/2개</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>다진 마늘 1작은술, 고추장 1큰술, 고춧가루 1큰술, 국간장 1+1/2큰술, 국시장국 2큰술, 멸치 다시마물 6컵, 소금, 후춧가루 약간씩</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>볼에 밀가루와 물, 달걀을 넣고 치대가면서 반죽을 한 덩어리로 만들어 주세요.</li>
 
 <li>한덩어리가 된 반죽을 비닐 봉지에 넣고 냉장고에서 약 30분 정도 숙성시켜 주세요.</li>
 
 <li>바지락은 소금을 약간 넣고 해캄을 시켜 주세요. 요즘은 해캄이 잘 되어 나온 바지락도 많이 있더라구요.</li>
 
 <li>애호박과 감자는 두툼하게 반달썰기 해 주시고 양파는 채썰어 주세요.</li>
 
 <li>멸치다시마 육수에 고추장과 고추가루를 풀어 한소금 끓여 주세요.</li>
 
 <li>바지락과 야채를 끓는 국물에 넣어 주세요.</li>
 
 <li>말랑말랑한 수제비 반죽을 얇게 손으로 밀어 떼어 넣어 주세요.</li>
 
 <li>수제비 반죽이 동동 떠오르면 다진마늘, 국간장, 국시장국, 소금, 후춧가루로 간을 해 주세요. 부족한 간은 소금으로 맞추세요.</li>
 </ol>
 </dd>
