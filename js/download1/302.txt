var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {
    if (typeof q_link == 'undefined' || q_link == '') {
        q_link = $("#listBlog [id^=q_link_]:checked").val();

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=302&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

    }

}

function doCommentSubmit()

                $("#cmtTextarea").val('');

 <h1><b>1.등갈비김치찜 황금레시피</b>
 <span>by.마리안느40</span>
 <div class="recipe_info">
 <b>난이도 :</b>초급 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/04/17/15037afe99233f7bf2ec684d09b5aca71.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>등갈비 10쪽, 신김치큰거 반쪽, 육수 - 국대접 3개 생수, 멸치 한줌, 다시마 한장</dd>

 <dt>[양념 및 소스재료]</dt>

 <dd>
 <ol>
 
 <li>일단은 등갈비 물에 담궈서 핏물 빼주세요 한...3~4시간정도 뺐나봐요... 
조리하기 전에 미리 국대접 3개 생수, 멸치 한줌, 다시마 한장 넣고 육수내서 준비해두시구요</li>
 
 <li>등갈비 한번 삶아 주시는데 잡내 제거를 위해 월계수잎 3~4장, 통후추 1수저, 물 잠길 정도로 넣고 한번 부르르~ 끓여주실게요~</li>
 
 <li>끓으면 불순물 제거 해주시고 등갈비 건져 팬에 넣어주세요</li>
 
 <li>한번 삶아낸 등갈비에 밑간해볼게요. 고추가루2, 설탕1, 마늘1 &lt;밥숟가락계량&gt; 넣으시고</li>
 
 <li>일회용 장갑끼고 골고루 버무려 주시고요</li>
 
 <li>신김치 한쪽이 너무 큰거라 칼로 대가리 반으로 잘라서 반쪽만 넣었어요</li>
 
 <li>이제 준비한 멸치육수 부어서 푹 끓여주기만 하면되요.. ㅎㅎ 너무 쉽죠?</li>
 
 <li>센불에 한번 푹 끓여 주시고 불끄고 한김 식혔다가 파 송송 썰어 넣고 다시 한번 자박자박 끓여서 내시면 되요. 그럼 좀 더 푹 맛있게 익은듯한...등갈비 김치찜이 완성된답니다.</li>
 </ol>

 <span>by.메기맹이</span>
 <div class="recipe_info">

<br>

다진마늘 1큰술, 다진파 1, 1/2큰술, 참기름 1큰술, 매실액 1큰술</dd>

 <li>먹기 좋게 뼈 한대씩 썰어서 준비하세요. 자르실 때 한쪽은 뼈에 바싹 자르시고 다른 쪽은 살을 듬뿍 해서 자르시면 살이 골고루 분포된다고 합니다.</li>
 
 <li>끓는 물에 대파, 통후추, 된장 등을 넣고 고기를 넣은 후 다시 끓어 오르면 청주를 붓고 30분 정도 삶아주세요.</li>
 
 <li>소스 재료를 한데 넣고 갈아서 준비하세요.</li>
 
 <li>끓인 등갈비를 찬물에 담궜다가 물기를 제거하여 소스와 함께 섞어 30분~1시간 정도 뒀다가 오븐에 살짝 구워주세요. 

220도로 예열된 오븐에 10분 내외 혹은 후라이팬에 졸이 듯 볶아주셔도 무관합니다.</li>
 
 <li>팬에 치즈를 깔고 옥수수콘을 얹은 뒤 등갈비도 함께 얹어 약불에 치즈가 녹을 정도로 구워 냅니다.</li>

 <h1><b>3.[등갈비구이]오븐없어도 맛있게 만들수 있는 등갈비 구이</b>
 <span>by.논현쌤</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>2시간 이상 <b>난이도 :</b>아무나 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/08/22/f165e3b29adbd1874f9d19cc5d664fbc1.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>등갈비 600g, 통후추 20-30알, 마늘 5알, 소주 2T, 파뿌리 2개</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>A1 바베큐 소스 4T, 발사믹식초 2T, 매실액 1T, 올리고당 1T</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>물은 등갈비가 잠길만큼 붓고 등갈비 핏물을 약 3시간동안 빼주세요</li>
 
 <li>등갈비 핏물을 빼는 동안 양념장 분량대로 냄비에 넣고 약불에서 보글보글 끓을때까지 끓여주시고 식혀주세요
매콤한 소스를 원하시면 베트남 고추를 함께 넣고 끓여주시면 됩니다
소스 맛을 보시고 취향에 따라 양념을 더 추가 하시면 됩니다
매실액이 없을경우 설탕 1T와 올리고당 1T를 넣어주시면 됩니다</li>
 
 <li>핏물을 뺀 등갈비는 찬물에 3-4번 헹군 후 마늘 5알 파뿌리 2개, 소주 2T와 물을 넣고 삶아주세요. 물양은 등갈비가 충분히 잠길만큼 넣어주세요. 고기를 푹 삶아줘야 하니 물 양은 적은것보다 많은게 더 좋아요
고기와 뼈가 분리될정도로 삶아주세요. 저는 약 1시간 정도 삶았어요</li>
 
 <li>고기가 다 삶아졌으면 고기만 건져주시고 팬에 고기와 양념장을 넣고 30분 정도 재워둔 후 약불에서 고기가 타지 않게 구워주시면 됩니다
다 익은 고기라서 오랫동안 굽지 않아도 되요. 양념이 노릇노릇해질때까지만 구워주세요</li>
 
 <li>접시에 올리고 그 위에 파슬리 가루를 솔솔 뿌려주면 끝</li>
 </ol>
