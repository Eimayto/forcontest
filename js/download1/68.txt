(쑥갓 대신에 미나리를 사용 하셔도 좋아요. 대신에 줄기 부분은 굵기 때문에 미열로는 익지 않을 수 있어, 한번 끓이시는 게 좋답니다) 레시피 상세보기 2.마음과 몸을 따숩게 녹여주는 비프스튜,, by.서현마미 조리시간 :30분 이내 난이도 :초급 [주재료] 소고기 등심 1kg, 밑간(소금, 후춧가루, 레드와인 2Ts), 올리브오일 2Ts, 양파 1개, 감자 2개, 당근 1개, 샐러리스틱 2개, 다진마늘 1Ts, 홀토마토 1캔(411g), 토마토페이스트 2Ts, 레드와인 150cc, 물 1500cc, 치킨스톡 1개, 월계수잎 1장, 바질가루 약간, 파슬리가루 [요리순서] 소고기는 큼직하게 썰어서 소금과 후춧가루, 레드와인을 넣어 밑간을 해두세요. 재료 준비~당근, 감자, 양파, 샐러리는 큼직큼직하게 썰어서 준비해요.야채를 작게 썰면 오랜시간 끓이면서 다 뭉개져서 찾아볼수가 없답니다. 홀토마토는 숟가락으로 대충 으깨어서 준비하고~ 큼직한 냄비를 준비해서 달구어주고,올리브오일을 둘러준 다음 밑간해둔 소고기를 넣어서 겉면이 익도록 볶아주세요. 센불에서~겉면이 익은 색이 나면 다른 그릇에 덜어두고, 그 냄비에 다진마늘과 야채를 넣어서 볶아줍니다. 어느정도 볶다가 홀토마토, 토마토페이스트, 레드와인, 물, 치킨스톡, 월계수잎, 바질가루를 넣어서 끓여줘요. 바글바글 끓으면 불을 약하게 줄여서 2시간 정도 두어요.중간중간 거품과 기름을 걷어내 주어야 맑은 국물의 스튜를 맛볼수 있어요. 두어시간 뭉근하게 끓이다가뚜껑을 열고 중강불에서 10분정도 끓여서 국물이 좀 졸아들게 해줍니다. 레시피 상세보기 3.영양가득 전복밥 by.예쁜포비 분량 :3인분 조리시간 :90분 이내 난이도 :중급 [주재료] 쌀 2컵, 전복 4개, 표고버섯 3개, 은행 10알, 참기름 1큰술, 밥물 1+1/2컵 [양념 및 소스재료] 영양부추 1/4줌, 청양고추 1개, 간장 3큰술, 참기름 1큰술 [요리순서] 쌀은 씻어서 건져 주세요. 불리셔도 되지만.. 전 압력밥솥에서 취사할 것이기 때문에 그냥 사용했습니다. 냄비밥을 하실 분들은 꼭 30분 불리세요. 오늘 가락시장에서 사온 신선한 전복과 표고버섯입니다. 전복 손질하는 법 알고 계시나요? 전복은 안쪽에서 수저를 이용해서 껍질과 살을 분리해 주세요. 이 때 내장이 부서지지 않도록 조심하세요. 떼어낸 전복의 모습입니다. 표시된 부분이 전복 내장입니다. 칼로 내장이 터지지 않도록 잘 분리해 주세요. 내장을 분리하면 사진에서 표시된 입 부분을 잘라 버려 주세요. 전복 내장도 함께 사용할 것입니다. 전복과 표고버섯은 슬라이스해 주세요. 달군 팬에 참기름을 두른 후 전복 내장과 전복을 넣고 잘 볶아 주세요. 이때 내장은 터뜨려 주세요. 쌀도 함께 살짝 볶아 주세요. 밥솥 내솥에 쌀과, 전복, 표고버섯을 넣고 밥 물은 1컵 반을 넣어 주세요. 표고버섯이 있어서 물을 평소보다 적게 잡았습니다. ​압력밥솥의 백미코스를 눌러 취사를 해 주었습니다. 뭐.. 넘 간단하지요.^^ 밥이 되는 동안 은행을 달군 팬에 볶아 주었습니다. 구운 은행은 키친타월을 이용해서 껍질을 벗겨 주세요. 그리고 영양부추를 넣은 간장을 만들어 주세요. 영양부추와 청양고추는 잘게 다져서 간장, 참기름과 함께 잘 섞어 주시면 됩니다. 간장을 만드는 사이~~ 밥이 다 되었네요.^^ 먹기 좋게 영양부추를 살짝 올린 후 전복을 젓가락으로 골라 위에 담았습니다. 좀~~ 뽀대납니다.ㅋㅋ매콤한 간장도 곁들여 주세요. 레시피 상세보기 " />

var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {
    if (typeof q_link == 'undefined' || q_link == '') {
        q_link = $("#listBlog [id^=q_link_]:checked").val();

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=68&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

    } else {
        $("#top_blog_select_url").hide();
        $("#top_blog_select_id").show();
    }

}

function doCommentSubmit()
{
    var txtVal = $("#cmtTextarea").val();
    if(!txtVal)
    {
        alert('의견을 입력해주세요.');
        return ;
    }
    if(txtVal.length < 4)
    {
        alert('의견은 4자 이상으로 작성해 주세요.');
        return ;
    }
    $.ajax({
        type: "POST",
        url: "/bbs/ajax.html",
        data : $("#insCmtFrm").serialize(),
        success: function(result) {
            if(result == 'SUCCESS')
            {
                $("#cmtTextarea").val('');
                getCommentList();
                var ccnt = parseInt($("#recipeCommentListCount").html(),10);
                $("#recipeCommentListCount").html(ccnt+1);
            }
            else if(result == 'NO_PARAM')
            {
                alert('파라미터 부족. 다시 확인해 주세요.')
            }
            else
            {
                alert('등록 실패! 잠시후 다시 시도해 주세요.')
            }
        },
        error: function () {
            alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
        }
    });
}

            }
        });
    }
}

 <h1><b>1.시원한 꽃게탕 </b>
 <span>by.니코로빈</span>
 <div class="recipe_info">
 <b>분량 :</b>3인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>중급 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/08/11/8c54af5c57481b44e035ec2bb1ba3b9b1.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>무 1토막(280g), 애호박 1/2개, 대파 1대(흰부분만 사용), 두부 약1/2모, 꽃게 4마리, 쑥갓(또는 미나리) 약간, 국물용 멸치 12마리, 다시마(5x5) 2장, 청양고추 1개, 물 8컵</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>시판 재래된장 3스푼(90g), 고춧가루 2스푼(10g), 다진마늘 1큰술, 청주 2큰술, 천일염 약간</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>된장3스푼, 고춧가루2스푼, 다진마늘1큰술, 청추2큰술을 볼에 넣고 잘 섞어 양념장을 만들어둡니다.</li>
 
 <li>애호박은 적당히 도톰한 굵기로 반달썰기, 무는 나박하게 썰기, 대파는 어슷썰기, 두부는 먹기 좋은 크기로 네모지게 썰어서 준비 합니다.</li>
 
 <li>분량의 물과 다시마2장, 국물용멸치12마리, 청양고추1개를 넣고 중불에서 끓여 주세요.</li>
 
 <li>물이 끓기 직전에 다시마는 건져냅니다.</li>
 
 <li>약불로 줄이시고... 7~10분정도 더 끓인 후 멸치, 청양고추도 건져냅니다.</li>
 
 <li>나박하게 썰어 놓은 무를 넣고, 중간불에서 끓기 시작하면, 무가 익을 때까지 약불에서 끓여 줍니다.</li>
 
 <li>꽃게는 깨끗하게 손질 된 상태라서 차가운 물에서 해동한 후 한번 더 씻어서 준비 했어요.</li>
 
 <li>무가 익으면 체에 받쳐 양념장을 풀어 넣으시고,</li>
 
 <li>애호박과 손질 된 꽃게를 넣고 한소끔 끓여 주세요.</li>
 
 <li>마지막으로 두부와 대파를 넣고 잠시 더 끓이시고요... 간이 부족하다면 천일염을 넣어 맞추어 주세요.</li>
 
 <li>불에서 내린 후 쑥갓을 얹으면 완성입니다. (쑥갓 대신에 미나리를 사용 하셔도 좋아요. 대신에 줄기 부분은 굵기 때문에 미열로는 익지 않을 수 있어, 한번 끓이시는 게 좋답니다)</li>
 </ol>
 </dd>

 <h1><b>2.마음과 몸을 따숩게 녹여주는 비프스튜,, </b>
 <span>by.서현마미</span>
 <div class="recipe_info">
 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>초급 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/03/18/ff897d4f0d78660856ced608ce18c12b1.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>소고기 등심 1kg, 밑간(소금, 후춧가루, 레드와인 2Ts), 올리브오일 2Ts, 양파 1개, 감자 2개, 당근 1개, 샐러리스틱 2개, 다진마늘 1Ts, 홀토마토 1캔(411g), 토마토페이스트 2Ts, 레드와인 150cc, 물 1500cc, 치킨스톡 1개, 월계수잎 1장, 바질가루 약간, 파슬리가루</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>소고기는 큼직하게 썰어서 소금과 후춧가루, 레드와인을 넣어 밑간을 해두세요.</li>
 
 <li>재료 준비~당근, 감자, 양파, 샐러리는 큼직큼직하게 썰어서 준비해요.야채를 작게 썰면 오랜시간 끓이면서 다 뭉개져서 찾아볼수가 없답니다.</li>
 
 <li>홀토마토는 숟가락으로 대충 으깨어서 준비하고~</li>
 
 <li>큼직한 냄비를 준비해서 달구어주고,올리브오일을 둘러준 다음 밑간해둔 소고기를 넣어서 겉면이 익도록 볶아주세요. 센불에서~겉면이 익은 색이 나면 다른 그릇에 덜어두고,</li>
 
 <li>그 냄비에 다진마늘과 야채를 넣어서 볶아줍니다.</li>
 
 <li>어느정도 볶다가 홀토마토, 토마토페이스트, 레드와인, 물, 치킨스톡, 월계수잎, 바질가루를 넣어서 끓여줘요.</li>
 
 <li>바글바글 끓으면 불을 약하게 줄여서 2시간 정도 두어요.중간중간 거품과 기름을 걷어내 주어야 맑은 국물의 스튜를 맛볼수 있어요.</li>
 
 <li>두어시간 뭉근하게 끓이다가뚜껑을 열고 중강불에서 10분정도 끓여서 국물이 좀 졸아들게 해줍니다.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>3.영양가득 전복밥</b>
 <span>by.예쁜포비</span>
 <div class="recipe_info">
 <b>분량 :</b>3인분 <b>조리시간 :</b>90분 이내 <b>난이도 :</b>중급 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/03/26/99a60aa6ab72828f32e75507f88dff8a1.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>쌀 2컵, 전복 4개, 표고버섯 3개, 은행 10알, 참기름 1큰술, 밥물 1+1/2컵</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>영양부추 1/4줌, 청양고추 1개, 간장 3큰술, 참기름 1큰술</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>쌀은 씻어서 건져 주세요. 불리셔도 되지만.. 전 압력밥솥에서 취사할 것이기 때문에 그냥 사용했습니다. 냄비밥을 하실 분들은 꼭 30분 불리세요.</li>
 
 <li>오늘 가락시장에서 사온 신선한 전복과 표고버섯입니다.</li>
 
 <li>전복 손질하는 법 알고 계시나요? 전복은 안쪽에서 수저를 이용해서 껍질과 살을 분리해 주세요. 이 때 내장이 부서지지 않도록 조심하세요.</li>
 
 <li>떼어낸 전복의 모습입니다. 표시된 부분이 전복 내장입니다. 칼로 내장이 터지지 않도록 잘 분리해 주세요.</li>
 
 <li>내장을 분리하면 사진에서 표시된 입 부분을 잘라 버려 주세요.</li>
 
 <li>전복 내장도 함께 사용할 것입니다. 전복과 표고버섯은 슬라이스해 주세요.</li>
 
 <li>달군 팬에 참기름을 두른 후 전복 내장과 전복을 넣고 잘 볶아 주세요. 이때 내장은 터뜨려 주세요.
쌀도 함께 살짝 볶아 주세요.</li>
 
 <li>밥솥 내솥에 쌀과, 전복, 표고버섯을 넣고 밥 물은 1컵 반을 넣어 주세요. 표고버섯이 있어서 물을 평소보다 적게 잡았습니다. ​압력밥솥의 백미코스를 눌러 취사를 해 주었습니다. 뭐.. 넘 간단하지요.^^</li>
 
 <li>밥이 되는 동안 은행을 달군 팬에 볶아 주었습니다.</li>
 
 <li>구운 은행은 키친타월을 이용해서 껍질을 벗겨 주세요.</li>
 
 <li>그리고 영양부추를 넣은 간장을 만들어 주세요. 영양부추와 청양고추는 잘게 다져서 간장, 참기름과 함께 잘 섞어 주시면 됩니다.</li>
 
 <li>간장을 만드는 사이~~ 밥이 다 되었네요.^^</li>
 
 <li>먹기 좋게 영양부추를 살짝 올린 후 전복을 젓가락으로 골라 위에 담았습니다. 좀~~ 뽀대납니다.ㅋㅋ매콤한 간장도 곁들여 주세요.</li>
 </ol>
 </dd>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}
