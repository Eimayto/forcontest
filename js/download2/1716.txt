function doBlogSelect() {

     if (select_type == 'url') {

function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();

            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}

 <li>분량의 잡채양념재료를 섞어 미리 준비해주세요.</li>
 <li>굴소스도 1큰술 넣어 함께 섞어주세요.</li>
 <li>달궈진 팬에 식용유를 두른 후 채썰어둔 재료들을 모두 넣어 달달 센불에서 볶아주세요. 참고로 오이는 입맛에 따라 가첨하시면 되는데요 오이를 넣으니 아삭함이 살아서 더 맛있게 느껴진답니다.</li>
 <li>채소들이 모두 볶아졌다면 여기에 미리 준비해 둔 잡채양념을 넣어 중불에서 한 번 더 볶아주세요.</li>
 <li>불려둔 당면을 넣어 1분간 볶아주세요. 이미 충분히 불려둔 상태인지라 가볍게만 볶아내도 당면은 충분히 익어요.</li>
 <li>매콤해서 더욱 맛있는 매콤 잡채덮밥 완성이네요.</li>
 </ol>

 <h1><b>3.홍합 매운 볶음 </b>
 <span>by.과일짱</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/22/a248aebea3893d5d06f301bf0ee97afb.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>홍합, 청량고추 2개, 양파 1/2개, 마늘 5~6톨</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>두반장 1큰술, 화이트와인(소주로 대체가능), 후추, 올리브유 2큰술</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>홍합은 겉에 붙은 수염도 떼고 깨끗하게 씻어서 준비해주세요.</li>
 <li>일단 냄비에 물을 끓이고 홍합을 넣고 마늘과 생강즙을 넣어 끓여주었답니다.</li>
 <li>삶아진 홍합은 이렇게 홍합과 국물을 함께담아 냉동실에 보관하면 국물요리할대 사용할수있구요.</li>
 <li>요렇게 살만 발라서 냉동보관하면 부침개나 홍합조림등으로 만들어먹을수 있어요.</li>
 <li>홍합살과 국물을 함께 넣어 냉동실에 보관하면 또 다른 요리에 넣어먹을수 있답니다.</li>
 <li>홍합육수만 보관해두어도 다양한 요리에 쓸수 있어요.</li>
 <li>청량고추와 마늘과 양파는 곱게 다져주시는데요. 야채다지기를 이용애서 다지셔도 됩니다.</li>
 <li>달구어진팬에 올리브유넣고 다진야채를 볶아주세요.</li>
 <li>야채가 볶아지면 홍합을 넣고 화이트와인을 조금 넣어 주면 좋아요. 화이트와인 대신 소주로 대체해도 됩니다.</li>
 <li>두반장을 넣어줍니다.</li>
 <li>센불에 재빠르게 볶아 홍합이 벌어지면 불을 끄심 됩니다.</li>
 <li>매콤한 홍합찜 완성입니다.</li>
 </ol>
 </dd>

 <h1><b>4.치즈를 얹은 매운 등갈비찜</b>
 <span>by.예쁜포비</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>중급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/20/68b9a1b44ec12b18b1b7537c2f878109.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>돼지고기 등갈비 1팩(500g), 떡볶이 떡 2줌, 모짜렐라 치즈 80g, 통후추 1큰술, 월계수잎 4장, 대파 1대, 생강 1쪽, 마늘 5쪽, 소주 약간, 정향 2알</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>[매운소스] 고춧가루 2큰술, 고추장 3큰술, 간장 2큰술, 꿀 2큰술, 맛술 2큰술, 생강즙 1작은술, 다진마늘 1/2큰술, 양파 1/2개, 홍고추(또는 청양고추) 2개, 갈비육수 1컵</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>돼지고기 등갈비는 안쪽의 얇은 막을 제거한 후 찬물에 약 2시간 정도 담가 핏물을 제거해 주세요. 중간중간에 물을 갈아 주시면 더 좋습니다.</li>
 <li>냄비에 돼지고기 등갈비를 넣고 등갈비가 잠길 정도로 물을 넉넉히 붓고 각종 향신재료를 넣은 뒤 50분 정도 삶아 주세요.</li>
 <li>다 삶은 등갈비는 체에 받쳐 물기를 제거해 주세요. 가위나 칼 등을 이용해서 갈비를 잘라 주세요.</li>
 <li>믹서기에 양파와 홍고추를 곱게 갈아 주세요.</li>
 <li>매운 양념소스에 양파와 홍고추 갈은 것, 갈비삶은 육수를 넣고 잘 섞어 주세요. 피자치즈도 준비해 주세요.</li>
 <li>달군 팬에 삶은 등갈비와 매운 양념소스를 넣고 중불에서 졸여 주세요.</li>
 <li>국물이 절반 정도 졸았을 때 떡볶이 떡을 넣고 조려 주세요.</li>
 <li>소스가 자작해질 때까지 조린 후 내열용기에 담아 모짜렐라 치즈를 골고루 뿌려주세요. 200도로 예열된 오븐에서 약 10분정도 구워 주세요. 치즈가 녹기만 하면 되기 때문에 전자레인지 등을 이용하셔도 됩니다.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>5.매운 불닭꼬치</b>
 <span>by.쿡라이프</span>
 <div class="recipe_info">

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>닭가슴살 250g</dd>

 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>고추장 1/2큰술, 고춧가루 2큰술, 간장 3큰술, 매실청 1큰술(또는 설탕), 물엿 1큰술, 청주 2큰술, 사과 1/4쪽, 양파 1/4, 청량고추 3개, 후추, 계피조금, 참기름 1작은술, 고추기름 1큰술, 깨소금</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>고추장 1/2큰술, 고춧가루 2큰술, 간장 3큰술, 매실청 1큰술(또는 설탕), 물엿 1큰술, 청주 2큰술, 사과 1/4쪽, 양파 1/4, 청량고추 3개, 후추, 계피조금을 넣고 믹서기에 갈아 줍니다. 매운게 싫으시면 청량고추를 빼 주세요.</li>
 <li>믹서기에 갈은 양념장에 마지막으로 참기름 1작은술, 고추기름 1큰술, 깨소금을 넣고 섞어 줍니다.</li>
 <li>닭가슴살은 가로, 세로 약 1.5cm로 썰어 줍니다.</li>
 <li>양념장에 닭가슴살을 1시간 동안 재워 둡니다.</li>

 <li>달구어진 팬에 꼬치를 놓아 줍니다.</li>

 </ol>

function getViewRecipeList(page)

    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}

$(document).ready(function () {
    getViewRecipeList(1);
})
