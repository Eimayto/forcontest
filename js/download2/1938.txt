function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }

    $("#txtBlogTail").text(tail);
}

        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');

 <span>by.연년생홀릭마더</span>
 <div class="recipe_info">
 <b>분량 :</b>1인분 <b>조리시간 :</b>10분 이내 <b>난이도 :</b>아무나 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/11/16/6af75233c803ee168ee257cf33f085041.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>알로에 500g, 흑설탕 500g, 현미식초 500g, 파인애플 500g, 흑설탕 500g, 현미식초 500g</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>모든 과일발효식초를 보시면 아시겠지만!
과일:식초:설탕=1:1:1의 비율이랍니다.
그러니 위에 기재해 둔 계량치에 신경쓰지 마시고 비율을 맞춰주세요.
우선 알로에를 준비해주시고</li>
 <li>알로에 식초의 경우 안에 있는 알로에겔을 사용할것이라서 껍질을 벗겨주세요</li>
 <li>다루기 쉽게 썰어주신다음에 세워놓은 상태로 주변의 껍질을 썰어내시면 됩니다.</li>
 <li>그러면 이렇게 알로에겔이 분리가되죠?</li>
 <li>넣기 좋은 크기 그리고 추출되기 좋은 크기로 이렇게 토막토막 썰어주세요.</li>
 <li>그리고 알로에의 양에 맞게 현미식초에 흑설탕을 넣어서 녹여주시고</li>
 <li>알로에겔을 넣어주면</li>
 <li>알로에식초만들기는 끝! 아주 간단해요.</li>
 <li>두번째는 파인애플 식초만들기
파인애플 식초도 마찬자기로 파인애플과 설탕 식초 전부 동일비율로 준비해주시면된답니다.</li>
 <li>저처럼 통 파인애플인 경우 껍질을 벗겨내고 심지를 제거한 다음에 무게를 측정해주세요.</li>
 <li>넣기 좋은 크기로 파인애플을 썰어주신다음에</li>
 <li>마찬가지로 동일비율의 식초</li>
 <li>동일비율의 흑설탕을 준비해주시는데요
제가 사용한것은 머스코바도예요 일반 흑설탕은 카라멜색소를 입힌것이라서 저희가 아는 몸에 좋은 흑설탕이 아니랍니다. 설탕을 사용하실거라면 비정제설탕을 추천하는 바예요^^</li>
 <li>알로에식초만들기처럼 파인애플식초만들기도 식초에 흑설탕을 넣고 녹여주시는데요. 저는 볼에 담은 다음 중탕으로 녹여줬답니다.</li>
 <li>이렇게 중탕으로 녹여주신다음에</li>
 <li>파인애플에 부어주시면</li>
 <li>끝난답니다!
파인애플 식초의 경우 건져서 먹어도 저의 경우 무난했구요. 알로에 식초의 경우 따로 알로에맛이 나는것이라서 요리초로 쓰기에도 좋답니다. 단! 알로에 특유의 그 끈끈한 점액질도 같이 추출되기 때문에 그 점은 있으니 참고하세요^^</li>
 </ol>
 </dd>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}
