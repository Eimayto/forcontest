var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {
    if (typeof q_link == 'undefined' || q_link == '') {
        q_link = $("#listBlog [id^=q_link_]:checked").val();

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1709&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

        $("#top_blog_select_url").show();
    } else {
        $("#top_blog_select_url").hide();
        $("#top_blog_select_id").show();
    }

}

function doCommentSubmit()
{
    var txtVal = $("#cmtTextarea").val();
    if(!txtVal)
    {
        alert('의견을 입력해주세요.');
        return ;
    }
    if(txtVal.length < 4)
    {
        alert('의견은 4자 이상으로 작성해 주세요.');
        return ;
    }
    $.ajax({
        type: "POST",
        url: "/bbs/ajax.html",
        data : $("#insCmtFrm").serialize(),
        success: function(result) {
            if(result == 'SUCCESS')
            {
                $("#cmtTextarea").val('');
                getCommentList();
                var ccnt = parseInt($("#recipeCommentListCount").html(),10);
                $("#recipeCommentListCount").html(ccnt+1);
            }
            else if(result == 'NO_PARAM')
            {
                alert('파라미터 부족. 다시 확인해 주세요.')
            }
            else
            {
                alert('등록 실패! 잠시후 다시 시도해 주세요.')
            }
        },
        error: function () {
            alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
        }
    });
}

function doCommentDelete(commentSeq)
{
    if(confirm('정말 삭제할까요?'))
    {
        $.ajax({
            type: "GET",
            url: "/bbs/ajax.html",
            data: "q_mode=deleteComment&commentSeq="+commentSeq,
            success: function(result) {
                if(result == 'SUCCESS')
                {
                    getCommentList();
                    var ccnt = parseInt($("#recipeCommentListCount").html(),10);
                    $("#recipeCommentListCount").html(ccnt-1);
                }
            },
            error: function () {
                alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
            }
        });
    }
}

 <li>잘 삶아진 국수에 만들어 놓은 국물을 넣고요!! ​오이, 당근 채썰어 놓은 것을 고명으로 올려 주시고 역시나 김치 채썰어 놓은 것도 넣고요. ​마지막으로 수박도 같이 올려 주시면 완성이요!!</li>
 </ol>
 </dd>
 </dl>

 <h1><b>7.수박껍질 오이고추 소박이</b>
 <span>by.쿨캣</span>
 <div class="recipe_info">
 <b>분량 :</b>4인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/08/24/b6a3df742c06dd19648d61057e43a29d1.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>오이고추 10개, 수박껍질 300g, 부추 100g, 양파 1/2개</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>소금 3큰술, 고춧가루 3큰술, 새우젓 2큰술, 설탕 1작은술, 매실청 1큰술, 다진 마늘 1큰술, 다진 생강 1/4 작은술, 통깨 약간</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>고추는 꼭지가 마르지 않고 겉이 단단하며 윤기가 나는 것을 고르시는 것이 좋아요.</li>
 <li>분량의 재료로 양념을 만들어 잠시 숙성을 시켜주세요. 아무래도 고춧가루가 불어야 하는 시간이 필요하니 시간상 젤 먼저 만들어 두는 것이 좋겠지요.</li>
 <li>수박은 달고 맛있는 빨간 과육 부분을 먹고 버리게 되는 하얀 껍질 부분을 이용할 거예요.</li>
 <li>수박 껍질은 곱게 채를 썰어주세요.</li>
 <li>채 썬 수박 껍질은 소금을 약간 뿌려 10여분 정도 절인 후 물기를 꼭 짜주세요. 절여져서 수분이 빠지면 양이 반으로 확 줄어버려요.</li>
 <li>오이 고추는 깨끗이 씻어 꼭지를 뗀 다음 양끝 1cm 정도만 남기고 세로로 길게 칼집을 넣어주세요. 이때 고추씨는 제거하지 않고 그대로 사용해주세요.</li>
 <li>물 4컵에 소금 3큰술을 넣고 섞은 후 칼집을 낸 오이 고추를 넣고 15~20분 정도 절여주세요. 이때 칼집을 낸 부분을 벌려 고추 안쪽까지 소금물이 잘 묻을 수 있도록 해주시구요..</li>
 <li>요렇게 부드럽게 휘어지도록 잘 절여졌어요.</li>
 <li>고추가 부드럽게 절여지면 물에 한번 헹구고 물기를 가볍게 탈탈 털어 낸 다음 채반에 뒤집어 놓고 물기를 완전히 빼주세요.</li>
 <li>양파와 부추는 짧게 2~3cm 내외의 길이로 채를 썰어주세요.</li>
 <li>미리 준비해둔 양념에 절여서 물기를 짠 수박껍질과 채썬 양파를 넣고 먼저 고루 버무려 주세요.</li>
 <li>이어 부추도 넣어 가볍게 버물막 버물락 처음부터 여린 부추를 넣으면 치이게 되니까 나중에 넣어 주는게 좋겠지요.</li>
 <li>물기를 제거한 고추에 김칫소를 가득 가득 미어져라 꾹꾹 눌러 채워주면 고추 소박이 완성, 밀폐용기에 담아서 하루 정도 실온 보관 후 냉장 보관하시면 됩니다. 고추 소박이는 숙성시키지 않고 바로 먹어도 맛나요. 풋풋한 풋맛이 좋으면 바로 냉장보관 하셔도 됩니다.</li>
 </ol>
 </dd>

 <h1><b>8.참치회맛이 나는 수박 스테이크</b>
 <span>by.밥심은국력</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>아무나 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/13/988e5a12eafbf9fa69c41a952504fb0f1.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>수박 직사각형 모양 2-3 조각, 새싹 채소 1/4통, 호두 2조각, 파마산 치즈 가루 약간 등</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>올리브유 1큰술, 발사믹 드레싱 3작은술, 발사믹 진한 소스 1큰술</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>수박의 모양은 원형으로 찍어서 사용하셔도 되고요, 전 사각 모양으로 잘라주었습니다. 두께는 도톰한 것이 좋구요.</li>
 <li>원하는 모양으로 자른 후에 보이는 씨는 제거하는 게 좋답니다.</li>
 <li>키친 타올에 수박을 감아서 수분을 제거한 뒤에 맛소금과 후추를 뿌려줍니다. 소금은 수박의 단맛을 더 상승시켜주는 역할을 한답니다.</li>
 <li>5-10분 정도 재워줍니다. 키친 타올을 깔아두었기 때문에 수분이 더 빠지는 효과도 있구요.</li>
 <li>달군 스테이크 팬에 올리브유를 1큰술 정도 둘러줍니다. 그릴 지국이 날 때까지 구워줍니다.</li>
 <li>앞뒤로 노릇노릇 구워주면 되구요.</li>
 <li>새싹채소를 곁들일텐데, 어린잎 채소도 좋고, 깻잎도 좋답니다. 가능한 집에 있는 채소를 이용하시구요.</li>
 <li>새싹채소 종류별로 잘 얹기 위해서는 채반에 받쳐서 그대로 수돗물을 틀어놓고 세척하면 편하답니다. 물기를 제거하시구요. 그래야 소스가 묽어지지 않는답니다.</li>
 <li>호두도 하나씩 예쁘게 올려주고요. 새싹채소를 올려 준 후에 발사믹 드레싱 3작은술과 발사믹 소스를 한번 더 뿌려줍니다.</li>
 </ol>
 </dd>

 <h1><b>9.바삭바삭 고소한 수박껍질 오징어 부추전</b>
 <span>by.쿨캣</span>
 <div class="recipe_info">
 <b>분량 :</b>4인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/08/22/86abbdd02493195f2968b1b59247dc761.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>부추 두줌, 수박 껍질 2컵, 오징어 다리 2개 분량, 홍고추 1개, 청양고추 2개, 부침가루 1과 1/2컵, 쌀가루 1/2컵, 물, 식용유 적당량</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>냉동 된 오징어 다리는 팔팔 끓는물에 넣어 살짝 데쳐 건져둡니다. 데쳐서 사용하면 전을 부칠 때 물이 생기지 않아 좋아요.</li>
 <li>부추는 먹기 좋은 길이로 썰고 오징어는 큼직큼직하게 썰고 고추는 송송 썰고 수박껍질은 채를 썰어 주세요. 양파나 깻잎 같은 채소를 더 넣어 주셔도 좋아요.</li>
 <li>부침가루와 쌀가루를 섞어 반죽을 만들고 준비한 재료를 모두 넣어 섞어주세요.</li>
 <li>달군팬에 식용유를 넉넉히 두르고 반죽을 떠서 얇고 넓게 펴주고</li>
 <li>앞뒤로 노릇노릇하게 부쳐주면 완성. 전을 부칠때 자꾸 뒤적이면 부스러지고 맛도 덜하니 한면이 완전히 익으면 그때 딱 한번 뒤집어 주는 것이 좋아요.</li>
 <li>오징어 다리가 큼직큼직하게 들어가서 더 먹음직 먹음직오징어를 한번 데친 후 부쳐서 물이 나오지 않아서 더 바삭한 맛이 살아 있어요.</li>
 </ol>
 </dd>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}
