var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {
    if (typeof q_link == 'undefined' || q_link == '') {
        q_link = $("#listBlog [id^=q_link_]:checked").val();

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1941&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

    }

}

function doCommentSubmit()
{
    var txtVal = $("#cmtTextarea").val();
    if(!txtVal)
    {
        alert('의견을 입력해주세요.');
        return ;
    }
    if(txtVal.length < 4)
    {
        alert('의견은 4자 이상으로 작성해 주세요.');
        return ;
    }
    $.ajax({
        type: "POST",
        url: "/bbs/ajax.html",
        data : $("#insCmtFrm").serialize(),
        success: function(result) {
            if(result == 'SUCCESS')
            {
                $("#cmtTextarea").val('');
                getCommentList();
                var ccnt = parseInt($("#recipeCommentListCount").html(),10);
                $("#recipeCommentListCount").html(ccnt+1);
            }
            else if(result == 'NO_PARAM')
            {
                alert('파라미터 부족. 다시 확인해 주세요.')
            }
            else
            {
                alert('등록 실패! 잠시후 다시 시도해 주세요.')
            }
        },
        error: function () {
            alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
        }
    });
}

function doCommentDelete(commentSeq)
{
    if(confirm('정말 삭제할까요?'))
    {
        $.ajax({
            type: "GET",
            url: "/bbs/ajax.html",
            data: "q_mode=deleteComment&commentSeq="+commentSeq,
            success: function(result) {
                if(result == 'SUCCESS')
                {
                    getCommentList();
                    var ccnt = parseInt($("#recipeCommentListCount").html(),10);
                    $("#recipeCommentListCount").html(ccnt-1);
                }
            },
            error: function () {
                alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
            }
        });
    }
}

 <dd>고구마 2개, 사과 1/2개, 체다+모짜렐라치즈 1컵, 우유 3큰술내외, 마요네즈 1큰술내외, 꿀,계피가루 약간씩, 소금,파슬리가루 약간씩, 버터 1/2큰술</dd>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>삶은 고구마는 볼에 담고 포크나 포테이토매셔 등을 이용해

 <h1><b>6.허니버터고구마칩 달달한 간식으로 만들기</b>
 <span>by.완소꽃남매</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>아무나 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/11/13/19b7dd6e37acc969a424eb7a42f3a32d1.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>고구마, 버터, 꿀, 파슬리가루, 식용유</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>고구마를 얇게 채썰어 주세요.
껍질은 깨끗히 씻어 벗겨도 안 벗겨도 괜찮아요 ^^</li>
 <li>찬물에 담궈 전분기를 없애주세요.
전분을 없애줘야 익힐때 서로 달라붙지 않아요.
5분 ~ 10분정도 담궈주시면 되요.</li>
 <li>키친타올에 물기도 제거해주세요.
그래야 튀길때 기름이 튀지 않아요 ^ . ^
요건 기름요리에 기본 상식이죠 !</li>
 <li>팬에 식용유를 적당이 넣어주세요.

얇게 채썬 고구마라 많은 양의 기름이 필요하진 않아요.
그냥 채썬것들이 담길정도로만 되면 된답니당.
0.5cm 높이만 되도 충분하다는 얘기에요! 정확히 잴 순 없으니 눈대중으로 대충 ~ ㅋㅋ</li>
 <li>적정 온도로 기름이 끓으면 채 썬 고구마를 넣어 튀겨주세요.

여기서 언제 넣어야 하냐는 건데요. 확인방법도 아주 간단해요 ^^

모든 튀김요리에 요렇게 하시면 되니깐 기억해두시면 좋아요.

나무젓가락을 기름에 넣었을때 기포가 몰리는 때에 약불로 유지시켜주며 튀겨주시면 된다는 것을요!</li>
 <li>약불로 1분 ~ 2분정도 튀겨주시다 센불로 색감이 살게 튀겨주시면 되요.
노릇노릇 노르스름해야 더 맛깔스러워보이니깐요. 헤헷</li>
 <li>​튀긴 것은 키친타올을 이용해 기름을 빼주세요.
그래야 바삭바삭하니 더 맛있어요.</li>
 <li>팬에 파슬리가루와 버터 그리고 꿀을 1대1 비율로 넣어 녹인다음에
재빨리 고구마를 넣어 버무려주시면 완성이에요 ^^</li>
 </ol>
 </dd>

 <h1><b>7.간단한 아이들 간식/ 고구마맛탕 황금레시피</b>
 <span>by.현짱2</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>15분 이내 <b>난이도 :</b>아무나 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/11/10/10241e7381ed68c9edf80e1484253ac71.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>고구마 3개, 참깨 또는 검은깨 소량</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>설탕 3T, 물 3T, 올리고당 1T</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>먼저 고구마는 깨끗하게 씻은 후 필러를 이용해 껍질을 모두 벗겨주세요!</li>
 <li>먹기 좋은 크기로 썰어줍니다!</li>
 <li>그다음 찬물에 담가 전분을 빼주세요~</li>
 <li>체를 이용해 고구마를 건져내고 물기를 탈탈 털어준 후 물기가 남아있지 않도록 키친타올로 한 번씩 닦아주세요! 번거로우시겠지만 기름에 튀길 것이기 때문에 물기가 남아있으면 안 된답니다..!^^</li>
 <li>이제 기름을 넉넉하게 두른 후 튀김온도를 맞춰주시면 되는데요~! 튀김온도 측정하는 법! 이거 많이들 어려워하시죵..ㅠㅎ 빵가루 조각을 하나 넣었을 때 가라앉았다가 바로 올라오면 적절한 튀김온도예요! 튀김온도가 적절하지 않으면 기름만 많이 먹고 눅눅해지거나 넣자마자 새까맣게 타는 수가 있어요! 튀김온도 적절하게 잘 맞춰주세요ㅎ</li>
 <li>튀김온도가 잘 맞추어졌으면 고구마를 넣고 튀겨줍니다!</li>
 <li>중불과 약불을 번갈아가면서 속까지 익도록 천천히 튀겨주었어요! 어느 정도 익었다 싶었을 때 젓가락으로 콕콕 찔러보세요! 쑥 들어가면 다 익은 거예요ㅎ</li>
 <li>한 번 튀겨낸 고구마를 건져낸 후 튀김온도를 조금 더 높여 한 번 더 튀겨줄 거예요!</li>
 <li>고구마를 다시 한 번 퐁당! 두 번째 튀길 때는 색이 금방 변하기 때문에 10~15초 정도 있다가 금방 건져내셔야 해요! 안 그러면 타버리는 수가 있어요!</li>
 <li>고구마튀김이 겉은 단단하고 속까지 잘 익었네요!ㅎㅎ</li>
 <li>이제 고구마맛탕 시럽소스 황금레시피를 알려드릴게요! 먼저 숟가락 계량으로 설탕 3스푼!</li>
 <li>물도 3스푼!</li>
 <li>올리고당은 1스푼! 설탕, 물, 올리고당의 비율을 3:3:1로만 맞춰주시면 달달하고 딱 적절하게 맛있는 고구마맛탕 소스를 만들 수가 있어요!</li>
 <li>약불에서 설탕이 녹으면서 점점 시럽이 만들어져요! 저어주시지 않으셔도 되고 가만히 내버려 두시면 알아서 만들어진답니다! 설탕이 금방 녹으면서 시럽이 만들어졌어요!</li>
 <li>고구마튀김을 넣고 숟가락 두개로 소스가 잘 버무려지도록 버무려줍니다! 시럽이 졸아들 때까지 버물버물~ ♩ ♪ ♬ ♪ ♩</li>
 <li>완성된 고구마맛탕을 접시에 먹기 좋게 담아준 후 참깨나 검은깨를 솔솔 뿌려주면 고구마맛탕 황금레시피 완성♥</li>
 </ol>

function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}

$(document).ready(function () {
    getViewRecipeList(1);
})
