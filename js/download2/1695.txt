(정확하진 않아욤. 오븐 시간에 맞추질 않고 중간중간 상태를 보아가며 이쑤시게 한번 찔러봐주고 꺼내기도 하구욤) 혹시라도 색이 심각하게 진하게 나오시면 중간에 은박지 같은거 덮어 주셔도 되욤. 식힘망에 쪼르륵 올려 두었다가 드세욤. 저처럼 뜨거울때 한번 드셔보는것도 괜찮구욤! 레시피 상세보기 2.슬라이스치즈 롤샌드위치 만들기 by.예쁜포비 분량 :1인분 조리시간 :15분 이내 난이도 :초급 [주재료] 식빵 4장, 슬라이스 치즈 2장 [양념 및 소스재료] 땅콩버터쨈, 포도쨈 적당량 [요리순서] 재료 참 간단합니다. 식빵은 가장자리를 잘라 내 주세요. 밀대로 식빵을 얇게 밀어 주세요. 식빵에 각각 땅콩버터쨈과 포도쨈을 골고루 펴 발라 주세요. 슬라이스 치즈는 겹쳐서 도톰하게 만든 후 식빵 위에 얹어 돌돌 말아 주세요. 랩으로 단단하게 감싸 주시면 더 좋습니다. 레시피 상세보기 3.슬라이스치즈 블루베리 케이크 by.니코로빈 분량 :4인분 조리시간 :90분 이내 난이도 :중급 [주재료] 높은원형팬, 통밀쿠키 90g, 버터 40g, 설탕 10g, 블루베리 90g [반죽] 우유 100g, 슬라이스치즈 5장, 버터 40g, 옥수수전분 15g, 박력분 35g, 노른자 2알, 레몬즙 1스푼 [양념 및 소스재료] [머랭] 흰자 2알, 설탕 50g [요리순서] 볼에 부숴진 비스킷과 설탕 10g을 넣고 잘 섞은 후, 버터를 넊고 더 섞어주세요. 계란은 흰자와 노른자를 분리 하시고요. 흰자는 깨끗한 큰 볼에 담아 냉장실에 넣어 두세요. 노른자는 작은 그릇에 담아 두세요. 유산지(테프론시트) 를 깐 틀에 비스킷을 담고 숟가락등을 이용해서 꾹꾹 눌러 준 뒤, 블루베리를 얹고 냉동실에 넣어 둡니다. 냄비에 우유 100ml, 슬라이스치즈 5장, 버터 40g을 넣고 약한 불애서 끓여주세요. 보글보글 끓이지 마시고요. 거품기로 저어 가면서 버터와 슬라이스 치즈를 녹이는 과정이예요. 슬라이스치즈가 덩어리없이 잘 녹으면 미리 준비했던 노른자 2개를 넣고 거품기로 재빨리 섞어준 후, 레몬즙 한스푼을 넣어 섞어주세요. 계란 노른자와 레몬즙을 넣을 때는 익거나 덩어리가 생길 수 있으니 재빨리 거품기로 섞어 주세요. 섞은 것을 볼아 담고 옥수수전분 15g, 박력분 35g 을 체에 내려서 거품기로 잘 섞어주세요. 덩어리 없도록 잘 섞어 주시면 되세요. 실온에 잠시 두시고요. 이제 머랭을 올릴 차례예요. 냉장실에서 꺼낸 흰자를 거품이 나게 휘젓고, 설탕을 2~3번에 나눠서 넣으며 머랭을 올려주세요. 거품기를 들어 올렸을 때 빳빳하게 서 있는 상태이면서 끝 부분이 휘는 정도로 올려 주세요. 머랭의 1/3 정도를 반죽에 섞어주고, 완벽하게 섞이면 남은 머랭을 넣어 재빨리 섞어줍니다. 반죽을 섞으실 때는 처음부터 끝까지 거품기로 잘 섞어 주시면 되세요. 비스킷을 넣었던 틀에 완성된 반죽을 모두 부어주세요. 그리고 다른 팬에 따뜻한 물을 붓고(케이크틀의 1/3정도) 160도로 예열된 오븐에서 1시간 20분 중탕으로 굽습니다. 굽는 도중에 윗면에 색이 진하게(약 20분 경과) 나오면 호일을 덮어 구워 주세요. 구워진 케이크가 잘 익었는지~ 케이크 중심에 꼬치를 찔러 넣어 확인 하시고요. 오븐 안에서, 오븐 문을 약간 열어 놓은 상태로 완전히 식혀 주세요. 식히는 시간은 2 ~3시간 걸린답니다. 완전히 식은 케이크는 서브리모 (기타등등 광택제)를 바르고 냉장고에 반나절 정도 넣어 두셨다가 드시면 되세요. 레시피 상세보기 4.슬라이스치즈를 이용한 "치즈볼"♪ by.서재인 분량 :2인분 조리시간 :5분 이내 난이도 :아무나 [요리순서] 슬라이스치즈를 준비합니다. 본인이 좋아하는 아무 치즈나 상관없어요^^ 치즈 비닐을 벗기기 전에 칼등으로 넉넉하게 9등분을 해줍니다. 전자렌지 사용가능한 접시에 종이호일을 깔아주고요~ 등분한 치즈를 넉넉하게 간격을 두고 올려줍니다. (파슬리가루, 깨도 취향껏 뿌려주세요) 그대로 전자렌지에 넣고 1분~1분20초 사이로 돌려줍니다. (정확하진 않아욤. 오븐 시간에 맞추질 않고 중간중간 상태를 보아가며 이쑤시게 한번 찔러봐주고 꺼내기도 하구욤) 혹시라도 색이 심각하게 진하게 나오시면 중간에 은박지 같은거 덮어 주셔도 되욤. 식힘망에 쪼르륵 올려 두었다가 드세욤. 저처럼 뜨거울때 한번 드셔보는것도 괜찮구욤! 레시피 상세보기 2.슬라이스치즈 롤샌드위치 만들기 by.예쁜포비 분량 :1인분 조리시간 :15분 이내 난이도 :초급 [주재료] 식빵 4장, 슬라이스 치즈 2장 [양념 및 소스재료] 땅콩버터쨈, 포도쨈 적당량 [요리순서] 재료 참 간단합니다. 식빵은 가장자리를 잘라 내 주세요. 밀대로 식빵을 얇게 밀어 주세요. 식빵에 각각 땅콩버터쨈과 포도쨈을 골고루 펴 발라 주세요. 슬라이스 치즈는 겹쳐서 도톰하게 만든 후 식빵 위에 얹어 돌돌 말아 주세요. 랩으로 단단하게 감싸 주시면 더 좋습니다. 레시피 상세보기 3.슬라이스치즈 블루베리 케이크 by.니코로빈 분량 :4인분 조리시간 :90분 이내 난이도 :중급 [주재료] 높은원형팬, 통밀쿠키 90g, 버터 40g, 설탕 10g, 블루베리 90g [반죽] 우유 100g, 슬라이스치즈 5장, 버터 40g, 옥수수전분 15g, 박력분 35g, 노른자 2알, 레몬즙 1스푼 [양념 및 소스재료] [머랭] 흰자 2알, 설탕 50g [요리순서] 볼에 부숴진 비스킷과 설탕 10g을 넣고 잘 섞은 후, 버터를 넊고 더 섞어주세요. 계란은 흰자와 노른자를 분리 하시고요. 흰자는 깨끗한 큰 볼에 담아 냉장실에 넣어 두세요. 노른자는 작은 그릇에 담아 두세요. 유산지(테프론시트) 를 깐 틀에 비스킷을 담고 숟가락등을 이용해서 꾹꾹 눌러 준 뒤, 블루베리를 얹고 냉동실에 넣어 둡니다. 냄비에 우유 100ml, 슬라이스치즈 5장, 버터 40g을 넣고 약한 불애서 끓여주세요. 보글보글 끓이지 마시고요. 거품기로 저어 가면서 버터와 슬라이스 치즈를 녹이는 과정이예요. 슬라이스치즈가 덩어리없이 잘 녹으면 미리 준비했던 노른자 2개를 넣고 거품기로 재빨리 섞어준 후, 레몬즙 한스푼을 넣어 섞어주세요. 계란 노른자와 레몬즙을 넣을 때는 익거나 덩어리가 생길 수 있으니 재빨리 거품기로 섞어 주세요. 섞은 것을 볼아 담고 옥수수전분 15g, 박력분 35g 을 체에 내려서 거품기로 잘 섞어주세요. 덩어리 없도록 잘 섞어 주시면 되세요. 실온에 잠시 두시고요. 이제 머랭을 올릴 차례예요. 냉장실에서 꺼낸 흰자를 거품이 나게 휘젓고, 설탕을 2~3번에 나눠서 넣으며 머랭을 올려주세요. 거품기를 들어 올렸을 때 빳빳하게 서 있는 상태이면서 끝 부분이 휘는 정도로 올려 주세요. 머랭의 1/3 정도를 반죽에 섞어주고, 완벽하게 섞이면 남은 머랭을 넣어 재빨리 섞어줍니다. 반죽을 섞으실 때는 처음부터 끝까지 거품기로 잘 섞어 주시면 되세요. 비스킷을 넣었던 틀에 완성된 반죽을 모두 부어주세요. 그리고 다른 팬에 따뜻한 물을 붓고(케이크틀의 1/3정도) 160도로 예열된 오븐에서 1시간 20분 중탕으로 굽습니다. 굽는 도중에 윗면에 색이 진하게(약 20분 경과) 나오면 호일을 덮어 구워 주세요. 구워진 케이크가 잘 익었는지~ 케이크 중심에 꼬치를 찔러 넣어 확인 하시고요. 오븐 안에서, 오븐 문을 약간 열어 놓은 상태로 완전히 식혀 주세요. 식히는 시간은 2 ~3시간 걸린답니다. 완전히 식은 케이크는 서브리모 (기타등등 광택제)를 바르고 냉장고에 반나절 정도 넣어 두셨다가 드시면 되세요. 레시피 상세보기 4.슬라이스치즈를 이용한 '치즈볼'♪ by.서재인 분량 :2인분 조리시간 :5분 이내 난이도 :아무나 [요리순서] 슬라이스치즈를 준비합니다. 본인이 좋아하는 아무 치즈나 상관없어요^^ 치즈 비닐을 벗기기 전에 칼등으로 넉넉하게 9등분을 해줍니다. 전자렌지 사용가능한 접시에 종이호일을 깔아주고요~ 등분한 치즈를 넉넉하게 간격을 두고 올려줍니다. (파슬리가루, 깨도 취향껏 뿌려주세요) 그대로 전자렌지에 넣고 1분~1분20초 사이로 돌려줍니다. (각자 사용하시는 전자렌지 열전도율에 따라 조금씩 차이는 있습니다) 전자렌지에서 나온 치즈볼 모양!! 제가 사용한 전자렌지는 1분20초를 돌리니 사진에서 보듯 모양이 흐트러지기 시작했었어요~ 3~4번 나눠서 돌릴때마다 전자렌지에서 눈을 떼지 않고 관찰했는데요.. 대부분 1분~1분10초 사이로 돌린게 모양이 제일 잘나오더라구요^^ 만들어진 치즈볼은 옮겨담아서 한 김 식혀주고요~ 식으면서 모양도 유지되고, 바삭하면서 쫄깃한 식감이 치즈의 맛을 더 깊게 느낄 수 있게 해줘요. 완성된 치즈볼 그릇에 나눠담아 사진찍어보기!! 통에 담아 놓으니 홈메이드가 순식간에 판매용으로~ ㅋㅋㅋㅋ 레시피 상세보기 " />

function doChangePortal(portal) {

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();

        } else {

function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}

실리콘틀은 조금 더 시간이 걸리는거 아시죵? 저는 25분정도 구운거 같구욤. (정확하진 않아욤. 오븐 시간에 맞추질 않고 중간중간 상태를 보아가며 이쑤시게 한번 찔러봐주고 꺼내기도 하구욤) 혹시라도 색이 심각하게 진하게 나오시면 중간에 은박지 같은거 덮어 주셔도 되욤.</li>
 <li>식힘망에 쪼르륵 올려 두었다가 드세욤. 저처럼 뜨거울때 한번 드셔보는것도 괜찮구욤!</li>

 <h1><b>3.슬라이스치즈 블루베리 케이크</b>
 <span>by.니코로빈</span>
 <div class="recipe_info">
 <b>분량 :</b>4인분 <b>조리시간 :</b>90분 이내 <b>난이도 :</b>중급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/30/393349e023cd5244b361d3925908805f.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>높은원형팬, 통밀쿠키 90g, 버터 40g, 설탕 10g, 블루베리 90g<br>
[반죽] 우유 100g, 슬라이스치즈 5장, 버터 40g, 옥수수전분 15g, 박력분 35g, 노른자 2알, 레몬즙 1스푼</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>[머랭] 흰자 2알, 설탕 50g</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>볼에 부숴진 비스킷과 설탕 10g을 넣고 잘 섞은 후, 버터를 넊고 더 섞어주세요. 계란은 흰자와 노른자를 분리 하시고요. 흰자는 깨끗한 큰 볼에 담아 냉장실에 넣어 두세요. 노른자는 작은 그릇에 담아 두세요.</li>
 <li>유산지(테프론시트) 를 깐 틀에 비스킷을 담고 숟가락등을 이용해서 꾹꾹 눌러 준 뒤, 블루베리를 얹고 냉동실에 넣어 둡니다.</li>
 <li>냄비에 우유 100ml, 슬라이스치즈 5장, 버터 40g을 넣고 약한 불애서 끓여주세요. 보글보글 끓이지 마시고요. 거품기로 저어 가면서 버터와 슬라이스 치즈를 녹이는 과정이예요.</li>
 <li>슬라이스치즈가 덩어리없이 잘 녹으면 미리 준비했던 노른자 2개를 넣고 거품기로 재빨리 섞어준 후, 레몬즙 한스푼을 넣어 섞어주세요. 계란 노른자와 레몬즙을 넣을 때는 익거나 덩어리가 생길 수 있으니 재빨리 거품기로 섞어 주세요.</li>
 <li>섞은 것을 볼아 담고 옥수수전분 15g, 박력분 35g 을 체에 내려서 거품기로 잘 섞어주세요. 덩어리 없도록 잘 섞어 주시면 되세요. 실온에 잠시 두시고요.</li>
 <li>이제 머랭을 올릴 차례예요. 냉장실에서 꺼낸 흰자를 거품이 나게 휘젓고, 설탕을 2~3번에 나눠서 넣으며 머랭을 올려주세요. 거품기를 들어 올렸을 때 빳빳하게 서 있는 상태이면서 끝 부분이 휘는 정도로 올려 주세요.</li>
 <li>머랭의 1/3 정도를 반죽에 섞어주고, 완벽하게 섞이면 남은 머랭을 넣어 재빨리 섞어줍니다. 반죽을 섞으실 때는 처음부터 끝까지 거품기로 잘 섞어 주시면 되세요.</li>
 <li>비스킷을 넣었던 틀에 완성된 반죽을 모두 부어주세요. 그리고 다른 팬에 따뜻한 물을 붓고(케이크틀의 1/3정도) 160도로 예열된 오븐에서 1시간 20분 중탕으로 굽습니다. 

굽는 도중에 윗면에 색이 진하게(약 20분 경과) 나오면 호일을 덮어 구워 주세요. 구워진 케이크가 잘 익었는지~ 케이크 중심에 꼬치를 찔러 넣어 확인 하시고요. 오븐 안에서, 오븐 문을 약간 열어 놓은 상태로 완전히 식혀 주세요. 식히는 시간은 2 ~3시간 걸린답니다.</li>
 <li>완전히 식은 케이크는 서브리모 (기타등등 광택제)를 바르고 냉장고에 반나절 정도 넣어 두셨다가 드시면 되세요.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>4.슬라이스치즈를 이용한 "치즈볼"♪</b>
 <span>by.서재인</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>5분 이내 <b>난이도 :</b>아무나 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2016/04/04/72469f617fbf51c5a2a22393cb75c6331.jpg">
 </h1>


 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>슬라이스치즈를 준비합니다. 본인이 좋아하는 아무 치즈나 상관없어요^^</li>
 <li>치즈 비닐을 벗기기 전에 칼등으로 넉넉하게 9등분을 해줍니다.</li>
 <li>전자렌지 사용가능한 접시에 종이호일을 깔아주고요~ 등분한 치즈를 넉넉하게 간격을 두고 올려줍니다. (파슬리가루, 깨도 취향껏 뿌려주세요)</li>
 <li>그대로 전자렌지에 넣고 1분~1분20초 사이로 돌려줍니다. (각자 사용하시는 전자렌지 열전도율에 따라 조금씩 차이는 있습니다)</li>
 <li>전자렌지에서 나온 치즈볼 모양!!</li>
 <li>제가 사용한 전자렌지는 1분20초를 돌리니 사진에서 보듯 모양이 흐트러지기 시작했었어요~</li>
 <li>3~4번 나눠서 돌릴때마다 전자렌지에서 눈을 떼지 않고 관찰했는데요.. 대부분 1분~1분10초 사이로 돌린게 모양이 제일 잘나오더라구요^^</li>
 <li>만들어진 치즈볼은 옮겨담아서 한 김 식혀주고요~</li>
 <li>식으면서 모양도 유지되고, 바삭하면서 쫄깃한 식감이 치즈의 맛을 더 깊게 느낄 수 있게 해줘요.</li>
 <li>완성된 치즈볼 그릇에 나눠담아 사진찍어보기!!</li>
 <li>통에 담아 놓으니 홈메이드가 순식간에 판매용으로~ ㅋㅋㅋㅋ</li>
 </ol>
 </dd>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}

    getViewRecipeList(1);
