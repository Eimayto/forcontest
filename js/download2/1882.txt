var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1882&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

function doCommentSubmit()
{
    var txtVal = $("#cmtTextarea").val();
    if(!txtVal)
    {
        alert('의견을 입력해주세요.');
        return ;
    }
    if(txtVal.length < 4)
    {
        alert('의견은 4자 이상으로 작성해 주세요.');
        return ;
    }
    $.ajax({
        type: "POST",
        url: "/bbs/ajax.html",
        data : $("#insCmtFrm").serialize(),
        success: function(result) {
            if(result == 'SUCCESS')
            {
                $("#cmtTextarea").val('');
                getCommentList();
                var ccnt = parseInt($("#recipeCommentListCount").html(),10);
                $("#recipeCommentListCount").html(ccnt+1);
            }
            else if(result == 'NO_PARAM')
            {
                alert('파라미터 부족. 다시 확인해 주세요.')
            }
            else
            {
                alert('등록 실패! 잠시후 다시 시도해 주세요.')
            }
        },
        error: function () {
            alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
        }
    });
}

}

 <h1><b>1.유기농 한우사골로 끓여내 구수하고 얼큰한 사골 육개장</b>
 <span>by. 시연성연맘</span>
 <div class="recipe_info">
 <b>분량 :</b>4인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/04/10/c24686bd8aa50a4eab079efd4efb2d201.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>한우 국거리 250g, 한우사골곰탕 1kg, 고사리 150g, 무 3cm 2토막, 대파 2대, 계란 2개, 참기름 1큰술</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>다진마늘 1과 1/2큰술, 고춧가루 6큰술, 참기름 1큰술, 국간장 5큰술, 후춧가루 약간</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>​고사리는 찬물에 담궈둔 후 끓는 물에 넣고 한차례 삶아주세요. 삶는 과정에서 고사리 특유의 쓴 맛이 제거가 된답니다.</li>
 <li>​무는 얇게 썰어주시고 대파는 4cm정도의 길이로 썰어준 후 2~4등분 해주세요. 무와 대파는 많이 들어갈수록 국물맛이 시원하고 개운해지거든요.​ ​</li>
 <li>​볼이 넓은 그릇에 고사리와 대파, 무를 넣은 후 분량의 육개장 양념을 넣고 조물조물 무쳐주세요. ​</li>
 <li>무쳐낸 채소들은 우선 재워두시고...</li>
 <li>​사골 육개장의 육수로 사용 될 산청목장의 유기농 한우사골 곰탕이랍니다. 보통 강한불에서 너무 오랫동안 끓여내는 경우가 있는데 사골특유의 구수함과 단맛이 없어지고 쓴맛과 누린내가 난다고 하더라구요.

강한불에서 약 4~5분간만 팔팔 한소끔 끓을 정도가 좋은듯해요. 그도 아니라면 팩 째 끓는 물에 넣고 데워드시는 것도 좋은 방법이랍니다.</li>
 <li>팬에 참기름을 1큰술 두른 후 중불에서 국거리용 한우소고기를 넣어 달달 볶아주세요. ​</li>
 <li>​양념에 재워둔 고사리와 채소도 넣어 함께 볶아준 후 ​</li>
 <li>​유기농 한우 곰탕 1kg을 넣고 바글바글 끓여주세요^^</li>
 <li>​무와 대파가 푹 익도록 센불에서 한소끔 끓인 후 끓기 시작하면 중불로 조절하고 약 2~30분정도 푹 끓여줍니다.</li>
 <li>​끓고나면 계란 2개를 곱게 풀어 넣어주세요.</li>
 <li>​간을 보시고 부족한 간은 소금이나 국간장으로 해주세요^^ ​ 좀 더 얼큰한 맛을 선호하신다면 고춧가루나 혹은 청양고추를 조금 썰어 넣으셔도 좋아요.</li>
 </ol>
 </dd>

 <h1><b>2.육개장 </b>
 <span>by.풀향기</span>
 <div class="recipe_info">
 <b>분량 :</b>5인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>중급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/05/26/fb499f73864cb8c5cfaebd250b7f2b211.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>쇠고기 610g, 고사리 280g, 숙주나물 270g, 토란대나물 250g, 느타리버섯 220g, 대파 2뿌리, 소금, 고사리, 숙주나물, 토란대나물</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>[양념장] 조선간장 2큰술, 고춧가루 2+1/2큰술, 다진 마늘 1큰술, 물 3큰술 <br>
<br>
[느타리버섯 양념] 고춧가루, 조선간장 1작은술, 다진마늘 약간, 참기름 약간</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>한우국거리를 준비합니다.</li>
 <li>냄비에 물을 적당량 붓고 핏물을 제거한 한우국거리를 넣어 냄비 뚜껑을 덮고 푹 삶아줍니다. 
름기와 중간에 생기는 거품은 걷어냅니다.</li>
 <li>대파는 다듬어서 흐르는 물에 씻은 후 5Cm길이 정도로 썰고 느타리버섯은 끓는 물에 살짝 데친 후 찬물에 헹구어 먹기 좋은 크기로 찢어줍니다.</li>
 <li>느타리버섯은 조선간장, 고춧가루, 다진 마늘, 참기름을 넣고 조물조물 무쳐줍니다. 
숙주나물은 삶아서 흐르는 물에 헹구어 물기를 제거하고 조선간장, 고춧가루,다진 마늘, 참기름 넣고 무쳐줍니다.</li>
 <li>토란대나물과 고사리나물도 삶아서 흐르는 물에 헹군 다음 물기를 제거하고 조선간장, 고춧가루, 다진 마늘, 참기름을 넣고 조물조물 무쳐줍니다.</li>
 <li>분량의 양념장 재료 조선간장, 물, 고춧가루, 다진 마늘을 넣어 고루 섞어줍니다.</li>
 <li>고기 삶은 육수에 양념한 고사리나물, 토란대나물, 숙주나물, 느타리버섯을 넣고 끓여줍니다.</li>
 <li>양념장을 넣고 끓여줍니다.</li>
 <li>썰어 놓은 대파를 넣고 끓인 후 소금으로 간을 맞춰주고 불에서 내려요.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>3.양지 넣고 끓인 한우 육개장</b>
 <span>by.히로</span>
 <div class="recipe_info">
 <b>분량 :</b>4인분 <b>조리시간 :</b>90분 이내 <b>난이도 :</b>중급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/08/12/9b2ed03ef51deda986169a0517f41b571.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>양지 500g, 숙주, 고사리, 대파 ​</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>다진 마늘, 고춧가루, 국간장, 고추기름, 참기름, 소금 ​<br>
</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>양지 500g입니다.</li>
 <li>핏물은 찬물에 빼주세요.</li>
 <li>커다란 솥에 파뿌리와 파를 좀 넣고 물 붓고 끓입니다. 파뿌리 없으면 그냥 물만 붓고 해도 상관없어요. 40분 정도 양지를 삶아서 먹기 좋게 찢어서 준비합니다.</li>
 <li>고사리는 불려서 삶아서 준비하고 숙주와 대파도 씻어서 데쳐서 준비합니다. 참고로 양지 삶은 물은 걸러서 육수로 사용합니다. 대파를 데칠 때는 양지 삶은 물에 데치면 파는 파대로 맛있고 육수에는 대파의 향이 스며들어 맛있습니다.</li>
 <li>모든 재료를 커다란 양푼에 담고 다진 마늘, 고춧가루, 고추기름, 참기름, 국간장을 넣고</li>
 <li>조물조물 무쳐줍니다. 저는 이 상태에서 육수를 붓고 끓였는데 어떤 고수님들은 이 포인트에서 이대로 한번 볶다가 육수를 붓기도 하시더군요.</li>
 <li>냄비에 담고 양지 삶을 때 만들어진 육수를 붓어 팔팔 끓여줍니다. 팔팔 끓어오르면 중불로 은근하게 좀 더 끓여서 완성합니다.</li>
 </ol>
 </dd>
 </dl>

 </ol>

 <h1><b>5.육개장 만들기</b>
 <span>by.근땡이</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>아무나 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/12/14/d833fe98822677e137739212d908ccbf1.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>사태 500g, 양지 500g, 양파 1/2개, 삶은고사리(말린것 60g) 600g, 대파 300g, 양파 1/2개, 숙주 1봉지</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>[양념] 고춧가루 6Ts, 고추장 1Ts, 소금 1Ts, 마늘 3Ts, 국간장 3Ts</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>전 사태와 양지 두가지를 준비했는데요. 각각 사태나 양지를 준비해서 넣으셔도 좋아요.</li>
 <li>먼저 고기는 냄비에 물을 넣고 끓을때 고기를 넣고 삶아주세요. 끓을때 넣어야 고기겉이 빨리 익어 거품도 생기지 않고 국물도 깔끔하게 우러나온답니다. (핏물빼지 않고 조리합니다~!) 이때 양파도 같이 넣고 끓여 고기를 푹~ 익혀주세요. 육수에서 거품이나 기름등은 다 건져주세요. 고기가 푹 익어 손으로 찢어질수 있을때까지 익히고 육수는 체에 걸러 준비해주세요.</li>
 <li>말린 고사리는 물을 끓여준후 마른고사리를 넣고 불려주면 따로 삶지 않아도 되요. 불린고사리는 물에 헹궈 물기를 짜서 준비해주세요.</li>

var curVRPage = 1;

function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')

        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
