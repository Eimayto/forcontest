[건새우,표고는 생략가능] 볶아진 멸치에 물 2대접을 넣어 뚜껑을 덮고 끓인다. 육수가 팔팔 끓기 시작하면 15분가량 더 끓여준다. 끓인 육수는 뚜껑을 열어 맛술 1스푼을 넣고 1분정도만 더 뚜껑을 열고 끓여준다. 육수에 들어갔던 재료들을 건져내주고 준비한 채소들 중 청양고추 , 파의 초록부분 , 쑥갓을 제외한것들을 넣어준다. 국간장 1스푼 , 진간장 1스푼을 넣고 한소큼 끓인 후 소금으로 간을 맞춘다. 만두를 넣어 만두속이 국물에 풀어질 것 같아 약간 싱거운 정도로 간을 했다. 육수가 다시 끓으면 준비한 칼국수 면을 흐르는물에 살짝만 헹구어 겉면의 밀가루를 제거해주고 만두와 함께 육수에 넣어 함께 끓여준다. [만두가 녹아있으면 면먼저 넣고 1분 후 만두를 넣어준다.] 면을 넣고 다시 육수가 끓어오르면 대파의 초록부분과 청양고추를 넣고 면이 다 익을때까지 끓여준다. 모든 재료가 다 익은 칼국수를 그릇에 담고 위에 김가루를 듬뿍 얹고 쑥갓도 얹어주고 , 취향에 맞게 고춧가루도 올려준다. 국물은 뽀얗지만 청양고추와 고춧가루가 들어가 꽤 칼칼한 맛을 낸다. 처음엔 국물이 조금 심심하지만 만두와 함께 국수를 먹다보면 만둣소의 양념이 국물에 풀어져 간이 딱 맞는다 :-) 레시피 상세보기 5.비오는날 딱인 초간단 동죽탕 by.카푸치노 분량 :2인분 조리시간 :30분 이내 난이도 :초급 [주재료] 동죽, 청양고추 [양념 및 소스재료] 마늘 [요리순서] 소금물에 동죽을 넣어서 해감을 시켜주는데 십원짜리 동전을 넣으면 해감이 잘 된다. 해감시킬 때는 신문지로 덮어주거나 뚜껑을 덮어서 하는게 좋다. 냄비에 물을 넣어 끓어오르면 동죽을 넣어 입이 벌어질 때까지 끓여준다, 이 때 거품이 올라오면 걷어주어야 깔끔한 조개탕이 된다. 거품을 완전히 깔끔하게 걷어준 후에 마지막에 청양고추, 마늘을 넣어 준다. 레시피 상세보기 " />

var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {

        if (!q_link) {

            return;

    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1680&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

냉동상태의 모듬해물을 소금물에 담가 해동하고</li>
 <li>해동된 모듬해물에 물기를 빼고 맛술을 1스푼 뿌려두었다.
(생물로 이용시 생략)</li>
 <li>팔팔 끓는물에 준비한 해물을 넣고 1분간 데쳐주고</li>
 <li>다시 물기를 빼 준비한다.</li>
 <li>계란은 풀어서 준비한다.</li>
 <li>당근은 조금 채썰어 준비하고 청양고추도 
잘게 썰어 준비한다.
색감을 위해 청 , 홍고추를 섞어서 준비했다.</li>
 <li>팬을 약불로 달구어 식용유 3스푼을 넣고 고루 펼쳐준 뒤 
(계속 약불로 조리한다.)</li>
 <li>준비한 쪽파를 반죽에 담갔다가 달구어진 팬에 
가지런히 펼쳐준다.</li>
 <li>그 위에 데쳐둔 해물을 펼쳐 올리고 , 
그 위로 남은 반죽을 고루 부어준다.</li>
 <li>그 위로 당근과 청양고추를 적당히 올려주고</li>
 <li>풀어둔 계란을 고루 뿌려준 뒤 식용유 2스푼을 계란물 위로 둘러준다.</li>
 <li>아랫부분이 적당히 익었으면 뒤집어주고</li>
 <li>앞뒤로 노릇노릇하게 익히면 완성 !</li>
 </ol>

 <h1><b>4.비오는날엔 만두 칼국수 :)</b>
 <span>by.달콤한꿀호떡</span>
 <div class="recipe_info">
 <b>분량 :</b>1인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/08/25/5820cf9c7d8bcf440f69edaf53b3c4c41.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>칼국수면 1인분, 냉동만두 3개, 양파 반개, 호박 한줌, 대파 한줌, 청양고추 2개, 마늘 1개, 쑥갓 반줌</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>육수용 멸치 (건새우,건표고) 한줌, 국간장 1스푼, 진간장 1스푼, 맛술 1스푼, 고춧가루 1스푼, 김가루 한줌, 소금 조금</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>마늘은 다지고 다른 채소들은 먹기좋은 크기로 썰어둔다.
멸치는 머리와 내장을 제거해야 깔끔한 육수를 낼 수 있다.</li>
 <li>냄비에 멸치 , 건새우 , 건표고를 넣어 고소한 냄새가 날 때까지 볶아준다.
[건새우,표고는 생략가능]</li>
 <li>볶아진 멸치에 물 2대접을 넣어 뚜껑을 덮고 끓인다.
육수가 팔팔 끓기 시작하면 15분가량 더 끓여준다.</li>
 <li>끓인 육수는 뚜껑을 열어</li>
 <li>맛술 1스푼을 넣고 1분정도만 더 뚜껑을 열고 끓여준다.</li>
 <li>육수에 들어갔던 재료들을 건져내주고</li>
 <li>준비한 채소들 중 청양고추 , 파의 초록부분 , 쑥갓을 제외한것들을 넣어준다.</li>
 <li>국간장 1스푼 , 진간장 1스푼을 넣고 한소큼 끓인 후</li>
 <li>소금으로 간을 맞춘다.
만두를 넣어 만두속이 국물에 풀어질 것 같아 약간 싱거운 정도로 간을 했다.</li>
 <li>육수가 다시 끓으면 준비한 칼국수 면을 흐르는물에 살짝만 헹구어 겉면의 밀가루를 제거해주고</li>
 <li>만두와 함께 육수에 넣어 함께 끓여준다.
[만두가 녹아있으면 면먼저 넣고 1분 후 만두를 넣어준다.]</li>
 <li>면을 넣고 다시 육수가 끓어오르면</li>
 <li>대파의 초록부분과 청양고추를 넣고 면이 다 익을때까지 끓여준다.</li>
 <li>모든 재료가 다 익은 칼국수를 그릇에 담고</li>
 <li>위에 김가루를 듬뿍 얹고 쑥갓도 얹어주고 , 취향에 맞게 
고춧가루도 올려준다.

국물은 뽀얗지만 청양고추와 고춧가루가 들어가 
꽤 칼칼한 맛을 낸다.
처음엔 국물이 조금 심심하지만 
만두와 함께 국수를 먹다보면 만둣소의 양념이 
국물에 풀어져 간이 딱 맞는다 :-)</li>
 </ol>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}

    getViewRecipeList(1);
})
