var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {
    if (typeof q_link == 'undefined' || q_link == '') {
        q_link = $("#listBlog [id^=q_link_]:checked").val();

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1076&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}
function doChangeBlogSelectType(select_type) {

        $("#top_blog_select_url").show();
    } else {
        $("#top_blog_select_url").hide();
        $("#top_blog_select_id").show();
    }

}

function doCommentSubmit()
{
    var txtVal = $("#cmtTextarea").val();
    if(!txtVal)
    {
        alert('의견을 입력해주세요.');
        return ;
    }
    if(txtVal.length < 4)
    {
        alert('의견은 4자 이상으로 작성해 주세요.');
        return ;
    }
    $.ajax({
        type: "POST",
        url: "/bbs/ajax.html",
        data : $("#insCmtFrm").serialize(),
        success: function(result) {
            if(result == 'SUCCESS')
            {
                $("#cmtTextarea").val('');
                getCommentList();
                var ccnt = parseInt($("#recipeCommentListCount").html(),10);
                $("#recipeCommentListCount").html(ccnt+1);
            }
            else if(result == 'NO_PARAM')
            {
                alert('파라미터 부족. 다시 확인해 주세요.')
            }
            else
            {
                alert('등록 실패! 잠시후 다시 시도해 주세요.')
            }
        },
        error: function () {
            alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
        }
    });
}

function doCommentDelete(commentSeq)
{

        $.ajax({
            type: "GET",
            url: "/bbs/ajax.html",

            error: function () {
                alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
            }
        });
    }
}

 <h1><b>1.바게트보트 미트볼파스타</b>
 <span>by.서현마미</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>2시간 이상 <b>난이도 :</b>고급 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/08/ef7434017cb64a2c8ac04c2d62563c1a.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>미니 바게트 - 우리밀 중력분 300g, 인스턴트 드라이이스트 4g, 천일염 6g, 물 200~210cc</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>미트볼파스타 - 스파게티 80g, 미트볼 6개, 토마토소스 160g, 마늘 2쪽, 페페론치노 3~4개,<br>
<br>
올리브오일 적당량, 소금, 후춧가루, 파르메산 치즈가루</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>먼저 바게트를 만들어 볼께요. 바게트는 다른 빵 반죽보다 덜 치대줘야 하는데, 저는 반죽기로 8~9분 정도만 반죽을 하고 꺼내줬어요. 반죽이 서로 매끈하게 한덩어리가 되어 손에 달라붙지 않을 정도만 반죽해주시면 될것 같아요.</li>
 
 <li>반죽을 둥글려 볼에 담아주고 윗면에 랩을 씌워 숨구멍을 3~4군데 뽕뽕 뚫어줍니다. 실온에 그대로 두어 1차 발효를 해주세요. 90분정도 발효를 시키다가 주먹으로 눌러서 펀칭을 해주고, 다시 윗면을 덮어 발효를 시켜줍니다.</li>
 
 <li>1시간 넘게 발효를 시켜주면 반죽의 부피가 처음보다 2배 이상 늘어나요. 그때까지 발효를 시켜주시면 되어요.</li>
 
 <li>반죽을 작업대에 올려놓고 손으로 눌러서 공기를 빼준다음, 3등분을 해서 타원형으로 둥글려주세요. 윗면에 젖은 면보를 덮어서 실온에서 20분정도 중간휴지를 시켜줍니다.</li>
 
 <li>반죽을 조금 길게 펴서 위에서 1/3정도를 접어 꾹꾹 눌러주고, 아래에서도 1/3정도를 접어 꾹꾹 눌러주세요.</li>
 
 <li>그런다음 반절로 접어서 끝부분을 손으로 눌러 붙여 마무리 해줍니다.</li>
 
 <li>간격을 두고 팬에 얹어서 2차 발효를 시작합니다. 저는 반죽의 분량을 좀 다르게 해서 하나는 길게 만들어주고, 나머지 두개는 통통하고 조금 짧게 성형해주었어요.</li>
 
 <li>윗면에 면보를 덮어 90분정도 실온에서 2차 발효를 시켜주고, 반죽이 부풀어오르면 적당히 칼집을 넣어주세요.</li>
 
 <li>240℃로 미리 예열해둔 오븐에 스팀을 넣어주고, 반죽을 넣은다음 온도를 210℃로 낮추어 약 20분정도 구워줍니다.</li>
 
 <li>짧게 구워낸 미니바게트에 미트볼파스타를 담아내려구요. 빵의 윗면을 잘라내주고, 칼로 속을 도려내어 보트처럼 만들어주세요.</li>
 
 <li>달군팬에 기름을 둘러주고, 미트볼을 넣어서 둥글려가며 속까지 익혀줍니다. 센불에서 익히게되면 겉면만 익고 속은 잘 익지 않을수도 있으니 중약불 정도에서 시간을 두고 익혀주시면 좋아요. 미트볼이 다 익으면 다른 그릇에 따로 덜어놓아둡니다. 미트볼을 익히면서 스파게티 면도 삶아서 체에 밭쳐 물기를 빼주세요.</li>
 
 <li>다른 팬에 올리브오일을 둘러주고, 편으로 썬 마늘과 페페론치노를 넣어서 향이 나도록 볶아주세요.</li>
 
 <li>마늘향이 나면 토마토소스를 넣어서 뜨겁게 데워주고, 삶아둔 면을 넣어서 고루 섞어줍니다.</li>
 
 <li>익혀둔 미트볼을 넣어서 같이 버무려주고 불에서 내려주세요. 이제 빵을 그릇삼아 이 파스타를 담아주시면 되요.</li>
 
 <li>파르메산 치즈를 얇게 갈아서 얹어주면 완성입니다.</li>
 </ol>
 </dd>

 <h1><b>2.닭가슴살 가지말이</b>
 <span>by.이슈아</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>15분 이내 <b>난이도 :</b>아무나 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/12/01/93d4c4e1a4ca8a5baa58b9c7e47cc9ad1.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>가지 1개, 닭가슴살 1캔, 피망 1/4개, 무순 약간, 양파 1/4개 </dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>다진마늘 1t, 식초 1t, 간장 1T, 참기름&amp;통깨 약간</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>가지는 깨끗하게 씻어서 가로로 길게 슬라이스하구요. 허브솔트 약간과 올리브유에 밑간 해두세요.</li>
 
 <li>되도록 얇게 썰은 가지는 프라이팬에 살짝 살짝 익혀줬다는 느낌이 들 정도로만 구워서 식혀주세요.</li>
 
 <li>그리고 그 외의 재료들은 한꺼번에 잘 섞어주는데요. 닭가슴살을 삶고 또 찢는 수고를 덜어주기 위해서 저는 동원 순닭가슴살을 사용했어요. 손으로 찢어서 결이 그대로 살아있기 때문에 그냥 캔을 톡 따서 사용하면 모양도 예쁘고 원하는 양만큼 사용할 수 있어 무척 편리하답니다.</li>
 
 <li>피망과 양파는 채썰어주고 참기름과 통깨도 솔솔솔 넣어</li>
 
 <li>모든 재료가 골고루 섞이도록 잘 섞어주세요.</li>
 
 <li>구워둔 가지를 길에 펴놓고 그 위에 적당량의 속을 올려 돌돌돌돌 말아주면 끝.</li>
 
 <li>속을 듬뿍 넣어야 더 맛이 좋구요. 든든하면서도 야채가 듬뿍 들어가 있어서 속도 편안한 메뉴에요. 하나하나 모양도 예쁘면서 만드는 과정이 크게 수고롭지도 않으면서 야채, 특히 가지 싫어하는 분들도 거부감 없이 드실 수 있어 더 좋답니다. 원하는 야채가 있다면 취향껏 더 추가해서 만드셔도 좋구요. 든든한 닭가슴살, 건강한 야채 듬뿍. 요즘들어 자꾸 허기지는 제게 딱 어울리는 음식이었어요.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>3.소고기숙주나물볶음 굴소스요리 파티요리</b>
 <span>by.전라도언니루비씨</span>
 <div class="recipe_info">
 <b>분량 :</b>3인분 <b>조리시간 :</b>15분 이내 <b>난이도 :</b>아무나 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/12/22/f6a4be07cb39ba7876806e049ce7b54f1.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>꽃등심(미국산) 300g, 숙주나물 1봉지, 식용유 약간</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>

 <dd>
 <ol>
 
 <li>먼저 팬에 식용유를 여유있게 둘러주시고 중불로 달궈줍시다잉~</li>
 
 <li>씻어서 물기를 뺀 숙주나물을 넣어주고</li>
 
 <li>굴소스 2아빠 숟가락을 넣어 부러요.

루비씨는 걍 눈대중으로 ㅎㅎ

많이 넣음 짭니다잉~~</li>

허벌라게 볶아불면 나중에 죽탱이 됩니다요.</li>
 
 <li>숙주에서 자연스럽게 물이 나오니까 재빨리 볶아분디요,

 눈으로 볼때 숙주가 살아서 기어나올것같아도 

한김빠지면 다 죽는다고요.

근께 언능 건집시다.</li>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}

    getViewRecipeList(1);
})
