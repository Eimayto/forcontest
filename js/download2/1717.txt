시도할때 당신이 기억하고 있는 것이 큰 도움이 될것이다.7. 각 나라의 음식을 무시하지 말라특정한 요리법을 배울 때 이러한 마음가짐은 시야를 흐리게 만들 수 있다. 당신이 지금 배우고 있는 요리법과 맞지 않더라도 다른 분야의 조리법을 배제하지 말아라. 샤퀴테리를 배운 패이스트리셰프는 달콤한 디저트뿐만 아니라 입맛을 돋구는 디저트 또한 만들 수 있을 것이다. var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}

    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1717&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

}

border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">본인이 일했던 모든 주방엔 큰 나이프 보관함이 있었고 오직 하나의 주방만이 보관함도 없었고 나이프들을 방치하기 일쑤였는데 그곳에선 오래 머물고 싶지도 않았고 오래 일하지도 않았다. 만약 자신이 매일 혹은 하루종일 사용하는 도구가 있다면 항상 최상의 상태로 보관하고 있어야 한다. 칼날은 항상 세워둘 것.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">2. 지저분하게 일하지 말 것</strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">항상 작업대 주위를 청결하게 하고 정리정돈 상태를 유지해야 한다. 만약 자신이 일하고 있는 곳이 지저분하다면 그것이 바로 당신 정신상태이다. 이 규율은 당신 작업대뿐만 아니라 카운터, 냉장고, 주방바닥에도 해당하는 이야기이다. 만약 당신이 깨끗하게 일하지 않는다면 위험할 뿐만 아니라 당신 혹은 다른 동료들까지도 다치게 할 수 있다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">3. 당신의 상관을 존중하라</strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">최고들과 일하고 최고가 되길 원한다면 명령받은 대로 일하고 상관이 당신에게 하는 말을 귀담아 들음으로써 그들의 다년간의 경험과 노하우를 전수받아라. 거인의 어깨 위에 올라서면 더 넓은 시야를 가지고 더 멀리 볼 수 있다는 말이 있듯이 당신의 상관이 바로 그 거인이다. 또한, 사공이 많으면 배가 산으로 가듯이 하나의 주방엔 한 명의 주방장만 존재한다. 설사 당신이 여러 주방에서 일한 경험이 있다 하더라도 상관을 존중하고 배움의 자세를 가져라. 아마 당신의 이력과 능력을 쌓는 데 큰 도움이 될 것이다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">4. 낭비하지 말라</strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">모든 것을 사용하라. 짐승의 모든 부위도 낭비하지 말고 사용하라. 오직 계란 흰자만 필요한 레시피가 있다 하더라도 크렘뷜레와 같은 요리에 남은 노른자를 사용할 수 있다. 먹다 남은 케이크와 빵이 있다면 우선 말린다음 가루로 내서 보관해 두었다가 빵가루같은 재료가 필요한 레시피에 이용할 수 있다. 세상 모든 조리법에는 모든 재료를 경제적으로 이용할 수 있는 방법이 있다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">5. 고전을 존중하라</strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">당신의 요리경력의 배경이 어찌됐든 고전 요리를 존중하라. 집 한 채를 지을때도 기반이 가장 중요하듯 고전요리의 레시피, 조리스킬을 통해 당신의 테크닉을 완벽하게 만들어라. 일단 당신의 테크닉이 완벽해지면 조리법의 창작이든 변형은 언제든지 가능하다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">6. 기록하라</strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">내가 일했던 주방에는 항상 레시피북이 잘 분류되어 있었다. 당신의 하나의 레시피를 배우면 바로 기록하라. 아마 몇 년이 지나도 당신은 잊지 않을 것이고 무언가 새로운 것을 시도할때 당신이 기억하고 있는 것이 큰 도움이 될것이다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">7. 각 나라의 음식을 무시하지 말라</strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">특정한 요리법을 배울 때 이러한 마음가짐은 시야를 흐리게 만들 수 있다. 당신이 지금 배우고 있는 요리법과 맞지 않더라도 다른 분야의 조리법을 배제하지 말아라. 샤퀴테리를 배운 패이스트리셰프는 달콤한 디저트뿐만 아니라 입맛을 돋구는 디저트 또한 만들 수 있을 것이다. 아시안 요리에 대해 지식이 있는 프렌치 셰프 또한 자신이 알고 있는 것을 이용해 전통 프랑스요리를 약간 변형시켜 새로운 요리나 독특한 요리를 만들어 낼 수 있을 것이다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">8. 자신의 미각을 존중하라</strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">나는 자신의 요리를 맛보지 않는 요리사는 믿지 않는다. 오직 맛을 봄으로써 항상 최상의 결과를 이끌어 낼 수 있다. 적절하게 간하는 방법을 배우고 만약 적절한 맛이 나지 않는다면 고객에게 접대하지 말 것.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">9. 불을 존중하라</strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">불의 사용법을 마스터하고 열의 속성에 대해 이해하라. 이러한 스킬이 당신의 질 좋은 재료를 아주 놀라운 한 끼 식사로 만들어 줄 것이다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">10. 비평을 무시하지 말아라</strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">건설적인 비평은 매우 창조적인 힘이 될 수 있다. 많은 자아와 감정을 요리에 담아내면 비평에 대해 매우 예민해질 수 있다. 그러나 당신이 비평으로 부터 배울 수 있다면 다음 번에 그 비평으로 부터 얻은 지식을 이용해 좋은 결과를 이끌어 낼 수 있을 것이다. 1번 규칙에서 언급했던 무딘칼을 아무렇지도 않게 사용하던 주방의 주방장은 항상 자기확신이 없었고 고객과 동료들의 조언을 귀담아듣지도 않았다. 그저 화만 낼뿐이었다. 이러한 태도는 문제를 해결하는데 아무런 도움도 되지 않는다. 단지 냉담하고 적의에 가득찬 분위기의 주방만 만들뿐이다.</p><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">만약 자신이 그런 주방에서 일하고 있다고 생각된다면 당장 때려치고 나와라. 그렇지 않으면 앞에 언급했던 3번 규칙을 따르기가 사실상 불가능 할 것이다.</p><div class="clear" style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; clear: both; color: rgb(51, 51, 51); font-family: Signika; font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);"></div><div class="divider divider-normal" style="padding: 0px; margin: 10px 0px; outline: none; list-style: none; border-width: 1px 0px 0px; border-style: solid none none; border-top-color: rgb(204, 204, 204); box-sizing: border-box; overflow: hidden; clear: both; color: rgb(51, 51, 51); font-family: Signika; font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);"></div><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);"><em style="padding: 0px; margin: 0px; var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}

    getViewRecipeList(1);
