var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {
    if (typeof q_link == 'undefined' || q_link == '') {
        q_link = $("#listBlog [id^=q_link_]:checked").val();

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1718&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}
function doChangeBlogSelectType(select_type) {
    if (select_type == 'url') {
        $("#top_blog_select_id").hide();
        $("#top_blog_select_url").show();
    } else {
        $("#top_blog_select_url").hide();
        $("#top_blog_select_id").show();
    }

}

function doCommentSubmit()
{
    var txtVal = $("#cmtTextarea").val();
    if(!txtVal)
    {
        alert('의견을 입력해주세요.');
        return ;
    }
    if(txtVal.length < 4)
    {
        alert('의견은 4자 이상으로 작성해 주세요.');
        return ;
    }
    $.ajax({
        type: "POST",
        url: "/bbs/ajax.html",
        data : $("#insCmtFrm").serialize(),
        success: function(result) {
            if(result == 'SUCCESS')
            {
                $("#cmtTextarea").val('');
                getCommentList();
                var ccnt = parseInt($("#recipeCommentListCount").html(),10);
                $("#recipeCommentListCount").html(ccnt+1);
            }
            else if(result == 'NO_PARAM')
            {
                alert('파라미터 부족. 다시 확인해 주세요.')
            }
            else
            {
                alert('등록 실패! 잠시후 다시 시도해 주세요.')
            }
        },
        error: function () {
            alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
        }
    });
}

function doCommentDelete(commentSeq)
{

        });
    }
}

 <div class="recipe_info">
 <b>분량 :</b>4인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/07/30/262d41f98337c1f0ccc8b4c4bf1797561.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>두부 1모, 물 두부곽으로 3번, 볶은 통깨 5TS, 소금 1TS, 설탕 1TS, 땅콩버터 1.5TS, 소면 4인분, 계란 취향것, 오이 취향것, 방울토마토 취향것</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>믹서기에 소면과 고명을 제외한 모든 재료를 넣고 잘 갈아준다</li>
 <li>갈아져서 만들어진 콩국물은 용기에 담아 냉장고에 넣어둔다

※ 조금 더 깔끔하게 먹고 싶을 경우 국물을 체에 한번 걸러낸다</li>
 <li>고명으로 올릴 방울 토마토는 반으로 가르고 오이는 돌려깎아 채 썰고 계란 1개는 완숙으로 삶아 반으로 가른다</li>
 <li>끓는 물에 소면을 넣고 찬물 1컵을 물이 끓어오르면 2-3번에 걸처 나누어 넣어가며 소면을 삶는다

삶아진 소면은 찬물에 헹군 뒤 체에 받쳐 물기를 털어낸다</li>
 <li>접시에 소면을 담고 콩국물을 부어준다</li>
 <li>고명을 올리고 통깨를 조금 더 뿌려 마무리 한다</li>
 </ol>
 </dd>
 </dl>

 <h1><b>3.상큼한 냉모밀 만들기 </b>
 <span>by.DanielSong</span>
 <div class="recipe_info">
 <b>분량 :</b>1인분 <b>조리시간 :</b>15분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/10/426bcb9e1e185f3bf52e7aa5d92311d31.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>메멜면 150g, 로즈마리 20g, 찬물 150ml, 얼음 7조각, 라임 1/5개, 깻잎 3장</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>메밀소바쯔유 200ml</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>냄비에 물을 넣고 면 삶을 물을 끓이기 시작해 주세요.</li>
 <li>깻잎은 서너장 돌돌 겹쳐서 말아 0.5cm 간격으로 슬라이스 해서 준비해 주세요.</li>
 <li>시판되는 메밀소바 쯔유를 활용해 보았는데요, 단맛이 거의 없는 제품이라 꿀이나 올리고당을 1T정도 추가 해 주시거나 배를 1/5개 정도 갈아 넣으셔도 좋습니다.</li>
 <li>냉모밀을 더욱 향긋하게 만들어 줄 로즈마리는 손으로 끊어서 육수에 넣어주세요.</li>
 <li>그리고 라임1/5개의 즙을 넣어 가볍게 섞어주세요. 완성된 육수는 냉동고에 넣고 차게 보관해 주세요.</li>
 <li>끓는물에 메밀국수 면을 넣고 휘휘 저어가면서 익히기 시작해주세요. 삶아진 냉모밀면을 흐르는 찬물에 수차례 헹구어 미끄덩한 전분기를 모두 씻어주세요. 이때, 얼음을 같이 활용해헹궈주시면 쫄깃함도 더해지고 더욱 시원하게 즐기실 수 있답니다.</li>
 <li>이제 마지막 단계!! 차게 냉동고에 보관한 허브 냉모밀 육수를 면이 70%정도 잠기도록 부어 얼음까지 동동 띄워서 시원하고 먹음직스럽게 담아내시면 냉모밀 만들기 끝~</li>
 </ol>
 </dd>
 </dl>

 <h1><b>4.가슴속까지 얼얼 물냉면</b>
 <span>by.칼스버그</span>
 <div class="recipe_info">
 <b>분량 :</b>3인분 <b>조리시간 :</b>2시간 이상 <b>난이도 :</b>중급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/09/21/2eb0b633abbe76c796c47d3e19ce6989.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>양지머리 600g, 물 5리터, 월계수잎 1잎, 감초 2알, 대파 1줄기, 양파 1개, 무우 1cm 두께 1토막, 통마늘 3개, 통생강 반쪽, 냉면, 오이, 계란, 배</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>국간장 3큰술, 소금 2작은술, 포도당(설탕) 1작은술, 구연산 1작은술 (구연산이 없을때는 식초 2큰술)</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>먼저 양지살 600g을 준비하여 핏물을 제거합니다. 물에 약 1시간 담가두면 핏물이 어느정도 제거가 되지요. 핏물을 제거한 양지살에서 기름부위를 제거합니다. 기름부위가 많으면 냉면육수에 기름기가 많아진답니다.</li>
 <li>물 5리터를 들통에 붓고 양지살을 넣은 다음 아주 센불에서 한번 푹 끓입니다. 10여분이 지나면 중불로 조절을 하여 은은하게 끓입니다.처음 끓을때에 생기는 거품은 걷어냅니다. 몇 분 정도 지나면 거품은 생기지가 않습니다.여기서 주의할 점은 물이 줄어든다고 찬물을 더 부어주는데 절대로 물을 추가로 넣으면 안된답니다. 끓는 물에 찬물이 들어가면 고기가 노린내가 나고 단맛이 나오게 됩니다.</li>
 <li>거품이 더이상 생기지 않을때에 월계수잎 한 장, 감초 2알을 넣고 1시간 동안 중간불에서 푹 고아야 합니다. 1시간 경과후 월계수잎과 감초를 꺼내고 다시 30분정도 푹 고아 준답니다. 월계수잎과 감초의 역활은 고기 노린내 제거와 함께 육수의 맛을 살려 줍니다. 총 1시간 30분정도 끓인 후에 양지머리를 건져내고 면보로 싸서 무거운 것으로 눌러 놓습니다. 물은 양은 절반으로 줄어들었습니다.</li>
 <li>고기를 건져낸 국물에 대파 1줄기. 양파 1개, 무우 1cm 두께 1토막, 통마늘 3개, 통생강 반쪽을 넣고 다시 30분정도 끓여야 합니다. 30분정도가 지나면 채로 모든 야채들을 건져내어서 육수를 최대한 깔끔하게 합니다. 그리고 국간장 3큰술, 소금 2작은술, 포도당(설탕) 1작은술, 구연산 1작은술 (구연산이 없을때는 식초 2큰술)을 넣고 한소끔 끓인후 불에서 내립니다.</li>
 <li>기름은 국자로 최대한 건져내고, 식으면 냉장고에 보관하여 기름을 응고 시킨다음 채에 면보를 얹어서 걸러내면 더욱 깔끔한 육수가 됩니다. 밑바닥에는 고기와 야채의 침전물이 가라앉아 있으니 조심스럽게 따르고 침전물은 버립니다. 육수의 양은 약 2리터 정도 확보가 되었습니다. 아래 사진은완성된 육수에 동치미 국물을 섞어서 (육수 70 : 동치미국물 30) 냉동실에서 보관한 사진입니다.</li>
 <li>삶아서 잘 씻은 다음 그릇에 담고 고명을 올리고 육수를 부어주면 맛있는 물냉면이 완성되겠죠.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>5. 오이냉국&amp;오이 냉국수</b>
 <span>by.민쿡스</span>
 <div class="recipe_info">
 <b>분량 :</b>1인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/12/369b066a70366211948c3b293b0bab14.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>불린미역 두줌, 오이 1개, 당근 1/3개, 양파 1/2개, 대파흰부분 1대, 청양고추 1 개, 홍고추 1개, 소면 한줌, 다시마우린 물 6컵, 얼음 조금</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>다진마늘 1작은술, 통깨 조금, 국간장 2큰술, 소금 조금, 설탕 1t, 고춧가루 1작은술, 식초 3큰술 </dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>미역은 미리 불려 주고요...</li>
 <li>오이는 소금으로 문질러 비빈 다음 깨끗하게 씻어 채썰고, 양파는 가늘게, 대파는 곱게... 고추도 썰어주고 마늘은 곱게 다져 줍니다.</li>
 <li>불린미역과 준비한 야채에 다진마늘 1작은술, 통깨 조금, 국간장 2큰술, 소금 조금, 설탕 1t, 고춧가루 1작은술, 식초 3큰술 넣어 주고...</li>
 <li>고루 섞어 냉장실에 차갑게 넣어 둡니다.</li>
 <li>먹기 직전 다시마 우린물을 넣고, 소금으로 간을 맞춘 뒤 얼을을 띄워 시원하게 드시면 돼요.</li>
 <li>소면도 삶아서...</li>
 <li>오이냉국에 담궈 드시면 별미음식이 된답니다.</li>
 </ol>
 </dd>
 </dl>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}
