var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1694&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

    }

}

function doCommentSubmit()

 <h1><b>1.콩나물 농어찜</b>
 <span>by.유현맘</span>
 <div class="recipe_info">
 <b>분량 :</b>4인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>중급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/08/27/96da368da00ecdadb1d133c4a2bfbff4.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>농어(반건조500g) 2마리, 콩나물 160g, 만드기(또는 미더덕) 170g, 쑥갓 30g, 대파 1/2대, 참기름 1T, 통깨 1T, 물 200ml<br>
[녹말물 재료] 녹말 2T, 물 2T<br>
[농어 데쳐낼 재료] 통마늘 3알, 된장 1/2t, 맛술(청주) 1T, 물 500ml</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>고춧가루 5T, 간장 3T, 맛술(청주) 2T, 생강가루(또는 생강즙 1T) 1/4t , 다진마늘 2T, 매실청 1T, 올리고당 1T </dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>제일 먼저 해야 할 일은 양념장 만들기에요!! 분량의 재료를 모두 넣어 양념장을 만들어주세요. 매실청이 없으시다면 올리고당으로 대체해주어도 되구요 .그도 없다면 설탕 1T를 넣어주세요.</li>
 <li>콩나물은 찜용 콩나물을 사서 머리를 떼어내주세요. 콩나물 머리는 조리시 떨어져서 좀 지져분해보이거든요..머리째 넣어서 드시고 싶으시면 그렇게 하셔도 무방합니다~ 전 머리는 다른곳에 쓸 예정이라 다 떼어냈어요~~ 마트에서 포장해서 파는 콩나물이 찜용이 새로 나왔더라구요.. 그래도 큰 통에 담아서 찜용으로 파는 콩나물보다는 굵기가 얇아요.</li>
 <li>미나리를 넣어야 하지만 집에 쑥갓이 있어서 대신 사용했어요. 쑥갓은 콩나물 길이로 썰어주고 대파는 어슷썰어주세요..</li>
 <li>이건 만드기라고 하는데요. 미더덕 친구쯤으로 생각하시면 되요.</li>
 <li>배를 갈라 내장을 제거하고 소금을 뿌려 반건조시켜둔 농어에요 그래서 생선살에 밑간이 다 되어있답니다. 비늘은 벗겨서 말리는데 배부분과 꼬리부분, 머리부분에는 약간씩 비늘이 남아있으므로 깨끗하게 비늘을 제거해주고 흐르는 물에 구석구석 깨끗하게 씻어내주세요..</li>
 <li>가위로 지느러미부분을 다 잘라내주세요.</li>
 <li>먹기 좋은 크기로 토막을 내주세요. 머리부터 꼬리까지 4토막을 냈어요.</li>
 <li>냄비에 물을 넣고 된장을 풀어 통마늘을 넣고 한소끔 끓여주세요..</li>
 <li>농어를 넣고 맛술(청주)을 넣어 한소끔 끓여 농어를 건져주세요. 잡내가 잡아진답니다. 농어를 넣고 오래 끓이지말고 뜨거운 물로 샤워시켜준다는 느낌으로 한번 후르르 끓으면 건져주세요.</li>
 <li>냄비에 물을 넣고 농어를 넣어 양념장을 2/3 넣어 뚜껑을 닥고 농어를 익혀주세요.</li>
 <li>농어가 반쯤 익을때까지 끓여주면 되요.. 중불에서 조리했어요..</li>
 <li>만드기를 넣고 그 위에 콩나물을 올려주세요..</li>
 <li>남은 양념장을 콩나물 위에 골고루 올리고 뚜껑을 닫아 콩나물을 익혀주세요..</li>
 <li>콩나물이 익으면 녹말물을 조금씩 넣어가면서 걸쭉하게 농도를 맞춰주세요. 콩나물은 너무 익히면 수분이 다 빠져서 질겨져요.. 숨이 죽을 정도로만 익혀주면 되요.</li>
 <li>불을 켜놓은 상태에서 쑥갓과 대파를 넣고 버무려준 다음 불에서 내려주면 완성이에요.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>2.부드러운 속살~ 병어조림</b>
 <span>by.노란장미</span>
 <div class="recipe_info">
 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>초급 </div>

 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>병어 2마리, 감자 3개,양파 반개, 대파 1/2대, 청양고추 2개, 홍고추 1개, 간장 4T, 고추장 1T, 고추가루 1t, 식용유 1T, 매실청 1T, 후추조금</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>병어 물로 씻어 지느러미 잘라야 깔끔하지만 오늘은 모양 고대로 담을거라 칼집만 앞뒤로 내주었어요~</li>
 <li>감자는 껍질벗겨 굵게 편 썰어요~ 양파도 굵게 채썰고 청양고추, 홍고추, 대파 어슷썰고 양념장 미리 섞어둬요~</li>
 <li>감자깔고~ 병어 얹고 양념얹고~ 파,고추 얹고 잘박하니 물을 부어요~
양파도 얹었구요~</li>
 <li>뒤집지않을 거라 병어 위로 물이 올라 오게 넉넉히 부어요~ 
뚜껑열고 센불에 끓이다 끓기 시작하면 불 줄이고 국물이 졸기 시작하면 수저로 국물 끼얹어가며 졸여요~</li>
 <li>양념이 너무 없어도 맛없으니 양념이 넉넉할 때 불을 꺼요~</li>
 <li>접시에 병어 살포시 담고 감자도 곁들여요~</li>
 </ol>
 </dd>

 <h1><b>3.얼큰 꽃게찌개 만들기</b>
 <span>by.민쿡스</span>
 <div class="recipe_info">
 <b>분량 :</b>3인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>중급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/15/112136495d61c6cc3483042c20a63426.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>꽃게 3 마리, 바지락 10 ~ 15 개, 홍고추 1 개, 청양고추 1 개, 무 1/4개, 애호박 1/2개, 물 10컵, 소금 조금, 쑥갓 조금</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>[양념장] 고춧가루 1큰술, 된장 1큰술, 고추장 1작은술, 바지락 육수 1/2컵, 다진마늘 1큰술, 다진생강 1작은술 소금, 후춧가루 조금</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>물을 팔팔 끓여 바지락을 넣고 끓여 주는데 입이 벌어지면 건져 놓고 육수는 계속 끓여요.</li>
 <li>끓이는 사이 꽃게손질을 하는데 솔로 구석구석 닦아내고, 가운데 배딱지와 등딱지를 떼어내고 꽃게양쪽에 붙어있는 털도 제거해 줍니다.</li>
 <li>물로 깨끗이 헹군 뒤, 꽃게를 2 등분으로 자르고, 집게발쪽은 칼등으로 두드려 줍니다.</li>
 <li>양념장 재료를 모두 섞어주고, 야채를 썰어 준비 하고요.</li>
 <li>푹 끓인 육수에 양념장을 넣고 끓여 줍니다.</li>
 <li>양념장이 한소끔 끓어 오르면 무우를 넣고 끓여 주고요.</li>
 <li>무우가 투명하게 익어가기 시작하면 꽃게와 건져놓은 바지락, 야채를 모두 넣고 끓여 줍니다.</li>
 <li>해물찌개를 끓일때 된장을 조금 풀어 끓이면 해물비린내가 안나고 시원하니 맛다답니다.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>5.춘곤증 예방에 도움을 주고 입맛 돋우는 냉이무침</b>
 <span>by.풀향기</span>
 <div class="recipe_info">
 <b>조리시간 :</b>10분 이내 <b>난이도 :</b>아무나 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/04/09/cde43ae059d586de74e99f44911b15ed1.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>냉이 200g, 소금 약간, 물</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>고추장 2큰술, 매실액 2큰술, 다진 파 1큰술, 다진 마늘 1/2큰술,<br>
식초 1작은술, 통깨 2작은술, 참기름 1작은술</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>냉이는 흙을 털어내고 누런 잎을 떼어낸 후 맑은 물이 나올 때까지 여러 번 씻어서 물기를 빼주세요.</li>
 <li>끓는 물에 소금을 조금 넣고 냉이를 넣어 데친 후</li>
 <li>찬물에 헹구어 물기를 빼주세요.</li>
 <li>분량의 양념장재료 고추장, 매실액, 다진 파, 다진 마늘, 식초, 참기름을 골고루 섞어주세요. 통깨는 마지막에 넣으려고 남겨두었습니다.</li>
 <li>냉이를 볼에 담고 양념장을 넣어 간이 골고루 배도록 조물조물 버무려주세요. 냉이가 큰 것은 반으로 썰어서 사용했습니다.</li>
 <li>통깨를 뿌려 다시 한번 골고루 버무려주세요.</li>
 <li>냉이무침을 접시에 담아냅니다. 딱 한 접시가 나오더군요.</li>
 </ol>
 </dd>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}
