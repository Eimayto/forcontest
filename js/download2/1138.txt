var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1138&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

}

 <h1><b>1.딸기롤샌드위치 만들기</b>
 <span>by.완소꽃남매</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>15분 이내 <b>난이도 :</b>아무나 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2016/01/16/b79e79a4642c20653ad278ff19cba82b1.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>식빵 3장, 딸기 9~12개 정도, 크림츠 OR 생크림, 랩</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>식빵의 테두리를 잘라주세요.
이때 자르신 테두리는 버리지 마시고 모아서 크루통 해드셔요^^</li>
 
 <li>​반죽 밀대를 이용해 밀어주세요.
밀대 없으시면 굴러다니는 병도 괜찮..?
아니면 일정하게 생긴 무언가로 슥슥 삭 삭 밀어주시면 돼요.</li>
 
 <li>밀어둔 식빵 위로 크림치즈를 듬뿍 발라주세요.
알차게 골고루 +. +
개인적으로 좀 더 두텁게 발라주시면
이 이후에 잘랐을 때 더 예쁠 거예요.</li>
 
 <li>비슷한 크기의 딸기를 골라 양쪽 끝을 잘라 중간쯤에 올려 돌돌 말아주세요.
너무 알찬 녀석들로는 하지 마세요.
적당히 작은 걸로 하셔야 예쁘게 잘 말려요.
그리고 양쪽 끝을 잘라주셔야 울퉁불퉁하지 않고 속을 채워주면서 일정해요^^

​</li>
 
 <li>모양 유지를 위해 말아둔 식빵딸기롤에 랩으로 꽁꽁 둘러주세요.
전 랩이 떨어져서 비닐팩 찢어다가 했네요 ㅋㅋ
요렇게 해주시면 모양도 유지되구요.
예쁘게 자르실 수 있어요^^</li>
 </ol>

 <h1><b>2.딸기 핫케이크 팬케이크 만들기,생크림만드는법 </b>
 <span>by.꽃청춘이주부</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>아무나 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2016/01/08/0b1724f656eff74ef83b9518e2734baa1.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>핫케익가루 200g, 딸기 10개, 계란 1개, 우유 100ml, 생크림 500ml, 설탕 1t, 꿀 1t, 식용유</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>재료: 핫케익가루200g,딸기10알,계란1개,우유100ml,생크림500ml,설탕1스푼,꿀1스푼,식용유조금</li>
 
 <li>마트가면 요렇게 생크림이나 휘핑크림을 팔더라구요~</li>
 
 <li>집에 핸드믹서기가 있으면 좋으련만....없으니 잇몸으로 대신해야겠어요.

 

믹서기에 생크림이나 휘핑크림을 넣어준뒤 설탕1스푼을 넣어서 갈아줍니다.

 

 

여기서 설탕1스푼을 넣었더니 전~~~혀 달지 않았어요

 

달달한맛을 좋아하시면 3-4스푼 넣어도 될거 같더라구요^^</li>
 
 <li>​믹서기에 1분정도 갈아주니까 생크림으로 완성!

집에서 간단하게 생크림을 만들수 있어요 ㅎㅎㅎㅎ

​

저처럼 핸드믹서기가 없어도 가능하답니다. ㅎㅎㅎㅎㅎㅎ​

​

​요과정이 귀찮으시면 큰 마트가시면 스프레이 생크림이나, 파바 가시면 천원에 팔더라구요~</li>
 
 <li>자, 이제 딸기를 깨끗하게 손질한뒤, 꼭지를 제거해주고, 반으로 잘라서 준비합니다</li>
 
 <li>부지런한 분들은 밀가루에 머 섞어서 핫케익 가루를 직접 만드시던데...

 

저는 간편하게 ㅋㅋㅋ 마트에서 핫케익크 가루를 구입해왔어요!

 

1인분 -100g 

2인분-200g

5인분 500g 

 

이라고 하니 참고하세요~

라면이나, 스프나 이렇게 정해진 메뉴얼 대로 해야 제일 맛난거 같아요

 

밑에 칼로리는 우리 못본걸로 해요.

음식할맛 떨어지니까요 ㅋㅋㅋㅋㅋ</li>
 
 <li>큰볼에 핫케익가루를 200g을 넣고, 우유를 100ml넣어준뒤 ,계란을 1개를 깨서 넣어줍니다.</li>
 
 <li>​그리고 잘 섞어주세요.</li>
 
 <li>​가열한 후라이팬에 식용유를 살짝 넣고 기름을 키친타올로 슬쩍 딱아주세요

​

(사진에는 너무 많이 뿌렸는데 조금만 뿌려도 된답니다)</li>
 
 <li>핫케익반죽을 한국자 떠서 후라이팬에 올려주세요

 

금방 탈수 있으므로 제일 약한불로 조리해줍니다.</li>
 
 <li>앞뒤로 노릇노릇하게 구워줍니다.</li>
 
 <li>한국자씩 만들어서 요렇게 접시에 담아 차곡차곡 쌓아줍니다.

 

 

저는 욕심부려서 7장을 했는데 4장정도 했을때가 모양도 제일 이쁜거 같아요.</li>
 
 <li>잘 쌓은 팬케이크위에 꿀1스푼을 골고루 뿌려준뒤 생크림을 듬뿍올려줍니다.</li>
 
 <li>잘라두었던 딸기를 차곡차곡 쌓으면서 올려주세요.</li>
 
 <li>저는 멋부린다고 슈가파우더를 좀 뿌려봤어요.</li>
 </ol>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}

    getViewRecipeList(1);
