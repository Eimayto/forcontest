[요리순서] 우선, 팥을 푸욱 삶아준후.. 블랜더로 약간의 물과 함께 갈아줍니다... 그런다음, 냄비에 담아... 소금약간과 설탕을 넣고 볶아주듯이 끓인후... 농도가 된듯하게 쫄았으면 불을 끄고 식혀줍니다... 팥이 완전히 식은후... 한입크기의 타원형모양으로 빚어 놓습니다. 방앗간에서 빻아온 쑥+쌀가루를... 체망에 한번 내려서 곱게 준비해 놓은후... 물을 여러차례 뿌려가며 두손으로 가루를 비비면서 잘 섞어주면, 소보루느낌의 가루가 됩니다. 그 가루를.. 김이 오른 찜통에서 20분가량 쪄낸후...바로 주물주물 반죽을 시작합니다... 겁나게 뜨겁슴다 ~ 조심조심... 떡이 거의 식을때까지 주물주물 반죽을 해주면.. 엄청시리 쫄깃쫄깃해지지요...ㅎ 그런다음, 밀대로 반죽을 밀어주고.. 준비해논 팥소를 올린후... 반죽을 반으로 접어... 둥근컵으로 찍어줍니다. 이렇게~ ^^ 마지막으로 참기름 or 들기름을 발라주시면 맛있는 바람떡 완성!...^^ 레시피 상세보기 3.아이반찬으로 좋은 참치쑥전 by.최쓰리 분량 :2인분 조리시간 :30분 이내 난이도 :초급 [주재료] 참치 1캔, 쑥, 계란 [요리순서] 잘 다듬은 쑥은 쌀뜨물에 1시간 정도 담궈주었어요. 담궈두면 안 좋은 성분들이 물안에 녹아 들어간다고 하더라고요. 1시간여 지난 후에,쌀뜨물에서 건져낸 다음 흐르는 찬물에 다시 한번 헹궈 먹기 좋은 크기로 썰어줍니다. 참치 한캔을 꺼내어 채에 넣고 기름기를 빼주어요. 아이반찬으로 만들 때에는 기름기를 쪽 빼서 담백하게 만드는게 더 좋아요. 먹기 좋게 썰어 놓은 쑥을 볼에 담고 그 위에 기름기를 쪽 뺀 참치를 올려요. 여기에 계란 두개를 넣고 잘 섞어줍니다. 계란의 양은 쑥과 참치의 양에 따라 가감해주시면 되요. 팬에 기름을 두르고 수저로 듬뿍 듬뿍 떠 올리고 지글지글 앞뒤로 부쳐주면 됩니다. 레시피 상세보기 4.향긋한 애쑥탕 by.풀향기 분량 :3인분 조리시간 :60분 이내 난이도 :중급 [주재료] 쑥 200g, 쇠고기(우둔살) 다짐육 190g, 두부 330g, 밀가루 4큰술, 소금 적당량, 대파 약간 [멸치다시마 국물 재료] 무 1/4개, 다시마 3조각, 양파 1/2개, 다시멸치 10마리, 물 6컵 [요리순서] 냄비에 물을 붓고 무, 다시멸치, 다시마, 양파를 넣어 멸치다시마 국물을 내어 줍니다. 다시마는 흐르는 물에 헹구어서 사용하고 멸치다시마 국물이 한소끔 끓으면 다시마는 먼저 건져 냅니다. 쑥은 흐르는 물에 씻은 후 체에 밭쳐 물기를 제거하고 잘게 썰어 줍니다. 고명으로 사용할 쑥은 약간 남겨 두었어요. 쇠고기 우둔살은 다짐육으로 준비합니다. 두부는 흐르는 물에 헹군 다음 베보자기를 이용해 물기를 꼭 짜서 으깨어 줍니다. 쑥과 쇠고기, 두부, 밀가루, 소금을 넣고 잘 섞은 다음 치대어서 동그랗게 완자로 빚어 줍니다. 접시에 완자가 달라붙지 않도록 밀가루를 살짝 뿌려 주었어요. 건더기를 건져 내고 한소끔 끓인 후 완자를 넣어 끓여 주면서 거품은 걷어 내고 소금으로 간을 맞춘 후 불에서 내려요. 애쑥탕을 그릇에 담고 쑥과 어슷 썬 대파를 고명으로 얹어 냅니다. 레시피 상세보기 5.활력이 쑥쑥! 봄내음 가득 바지락 쑥수제비 by.에브리맘 분량 :2인분 조리시간 :60분 이내 난이도 :초급 [요리순서] 바지락은 소금물에 해감하고, 쑥은 깨끗이 씻어서 준비한다. 애호박과 당근, 양파는 채썰어 준비하고, 감자는 은행잎 썰기를 해둔다. 손질한 쑥과 물 100ml를 넣고 믹서에 간 다음, 밀가루에 섞어 만든 반죽을 10분 정도 랩을 씌워 휴지시킨다. 냄비에 바지락과 물을 넣고 바지락이 입을 벌릴 때가지 팔팔 끓인 뒤 감자를 넣어준다. 감자가 살짝 익었을 때 모든 채소를 넣어 끓인다. 수제비 반죽을 조금씩 떼어서 넣고 끓이다가 반죽이 동동 떠오르면 다진마늘과 국간장, 소금을 넣어 간한다. 마지막으로 쑥을 넣고 한소끔 끓여주면 바지락 쑥 수제비 완성! 레시피 상세보기 " />

var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {
    if (typeof q_link == 'undefined' || q_link == '') {
        q_link = $("#listBlog [id^=q_link_]:checked").val();

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1692&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

    }

}

function doCommentSubmit()
{
    var txtVal = $("#cmtTextarea").val();
    if(!txtVal)
    {
        alert('의견을 입력해주세요.');
        return ;
    }
    if(txtVal.length < 4)
    {
        alert('의견은 4자 이상으로 작성해 주세요.');
        return ;
    }
    $.ajax({
        type: "POST",
        url: "/bbs/ajax.html",
        data : $("#insCmtFrm").serialize(),
        success: function(result) {
            if(result == 'SUCCESS')
            {
                $("#cmtTextarea").val('');
                getCommentList();
                var ccnt = parseInt($("#recipeCommentListCount").html(),10);
                $("#recipeCommentListCount").html(ccnt+1);
            }
            else if(result == 'NO_PARAM')
            {
                alert('파라미터 부족. 다시 확인해 주세요.')
            }
            else
            {
                alert('등록 실패! 잠시후 다시 시도해 주세요.')
            }
        },
        error: function () {
            alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
        }
    });
}

}

 <h1><b>1.향긋한 쑥계란말이</b>
 <span>by.과일짱</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>15분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/05/1296019820fac05b76d0ac14e5b159a31.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>데친쑥, 달걀, 식초</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>포도씨유(식용유), 소금</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>데쳐둔 쑥은 물기를 제거해줍니다.</li>
 <li>계란을 잘 풀어줍니다.</li>
 <li>데친 쑥을 넣고 식초를 조금 넣어 주세요. 소금도 살짝 넣어줍니다.</li>
 <li>달군 팬에 포도씨유를 두르고 중불로 계란물을 부어 돌돌 말아주세요.</li>
 <li>앞뒤로 뒤집어가며 잘 익혀 줍니다.</li>
 <li>한김 식으면 계란말이를 썰어냅니다.</li>
 </ol>
 </dd>

 <h1><b>2.쑥향이 은은한 쑥바람떡만들기</b>
 <span>by.돌하우스</span>
 <div class="recipe_info">
 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/04/22/62c7e584c97eee72e1b888a2ceefe99f1.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>쑥+쌀가루, 물, 팥소, 참기름(or 들기름)...</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>우선, 팥을 푸욱 삶아준후.. 블랜더로 약간의 물과 함께 갈아줍니다... 
그런다음, 냄비에 담아... 소금약간과 설탕을 넣고 볶아주듯이 끓인후... 농도가 된듯하게 쫄았으면 불을 끄고 식혀줍니다...</li>
 <li>팥이 완전히 식은후... 한입크기의 타원형모양으로 빚어 놓습니다.</li>
 <li>방앗간에서 빻아온 쑥+쌀가루를...</li>
 <li>체망에 한번 내려서 곱게 준비해 놓은후...</li>
 <li>물을 여러차례 뿌려가며 두손으로 가루를 비비면서 잘 섞어주면, 소보루느낌의 가루가 됩니다.</li>
 <li>그 가루를.. 김이 오른 찜통에서 20분가량 쪄낸후...바로 주물주물 반죽을 시작합니다...</li>
 <li>겁나게 뜨겁슴다 ~ 조심조심...
떡이 거의 식을때까지 주물주물 반죽을 해주면.. 엄청시리 쫄깃쫄깃해지지요...ㅎ</li>
 <li>그런다음, 밀대로 반죽을 밀어주고.. 준비해논 팥소를 올린후...</li>
 <li>반죽을 반으로 접어... 둥근컵으로 찍어줍니다.</li>
 <li>이렇게~ ^^</li>
 <li>마지막으로 참기름 or 들기름을 발라주시면 맛있는 바람떡 완성!...^^</li>
 </ol>
 </dd>

 <h1><b>3.아이반찬으로 좋은 참치쑥전 </b>
 <span>by.최쓰리</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/05/08/216b7a5f10cf638b31c7ac7eb44211ad1.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>참치 1캔, 쑥, 계란</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>잘 다듬은 쑥은 쌀뜨물에 1시간 정도 담궈주었어요. 담궈두면 안 좋은 성분들이 물안에 녹아 들어간다고 하더라고요.</li>
 <li>1시간여 지난 후에,쌀뜨물에서 건져낸 다음 흐르는 찬물에 다시 한번 헹궈 먹기 좋은 크기로 썰어줍니다.</li>
 <li>참치 한캔을 꺼내어 채에 넣고 기름기를 빼주어요. 아이반찬으로 만들 때에는 기름기를 쪽 빼서 담백하게 만드는게 더 좋아요.</li>
 <li>먹기 좋게 썰어 놓은 쑥을 볼에 담고 그 위에 기름기를 쪽 뺀 참치를 올려요.</li>
 <li>여기에 계란 두개를 넣고 잘 섞어줍니다. 계란의 양은 쑥과 참치의 양에 따라 가감해주시면 되요.</li>
 <li>팬에 기름을 두르고 수저로 듬뿍 듬뿍 떠 올리고 지글지글 앞뒤로 부쳐주면 됩니다.</li>
 </ol>
 </dd>

 <h1><b>4.향긋한 애쑥탕</b>
 <span>by.풀향기</span>
 <div class="recipe_info">
 <b>분량 :</b>3인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>중급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/07/28/455c5484c3d33081759f7a386add00d5.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>쑥 200g, 쇠고기(우둔살) 다짐육 190g, 두부 330g, 밀가루 4큰술, 소금 적당량, 대파 약간<br>
<br>
[멸치다시마 국물 재료] 무 1/4개, 다시마 3조각, 양파 1/2개, 다시멸치 10마리, 물 6컵</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>냄비에 물을 붓고 무, 다시멸치, 다시마, 양파를 넣어 멸치다시마 국물을 내어 줍니다. 다시마는 흐르는 물에 헹구어서 사용하고 멸치다시마 국물이 한소끔 끓으면 다시마는 먼저 건져 냅니다.</li>
 <li>쑥은 흐르는 물에 씻은 후 체에 밭쳐 물기를 제거하고 잘게 썰어 줍니다. 고명으로 사용할 쑥은 약간 남겨 두었어요.</li>
 <li>쇠고기 우둔살은 다짐육으로 준비합니다.</li>
 <li>두부는 흐르는 물에 헹군 다음 베보자기를 이용해 물기를 꼭 짜서 으깨어 줍니다.</li>
 <li>쑥과 쇠고기, 두부, 밀가루, 소금을 넣고 잘 섞은 다음 치대어서 동그랗게 완자로 빚어 줍니다. 접시에 완자가 달라붙지 않도록 밀가루를 살짝 뿌려 주었어요.</li>
 <li>건더기를 건져 내고 한소끔 끓인 후 완자를 넣어 끓여 주면서 거품은 걷어 내고 소금으로 간을 맞춘 후 불에서 내려요. 애쑥탕을 그릇에 담고 쑥과 어슷 썬 대파를 고명으로 얹어 냅니다.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>5.활력이 쑥쑥! 봄내음 가득 바지락 쑥수제비 </b>
 <span>by.에브리맘</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2016/04/06/0827c99f7b67960cfa7c301d6f93ceb51.jpg">
 </h1>


 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>바지락은 소금물에 해감하고, 쑥은 깨끗이 씻어서 준비한다.</li>
 <li>애호박과 당근, 양파는 채썰어 준비하고, 감자는 은행잎 썰기를 해둔다.</li>
 <li>손질한 쑥과 물 100ml를 넣고 믹서에 간 다음, 밀가루에 섞어 만든 반죽을 10분 정도 랩을 씌워 휴지시킨다.</li>
 <li>냄비에 바지락과 물을 넣고 바지락이 입을 벌릴 때가지 팔팔 끓인 뒤 감자를 넣어준다.</li>
 <li>감자가 살짝 익었을 때 모든 채소를 넣어 끓인다.</li>
 <li>수제비 반죽을 조금씩 떼어서 넣고 끓이다가 반죽이 동동 떠오르면 다진마늘과 국간장, 소금을 넣어 간한다.</li>
 <li>마지막으로 쑥을 넣고 한소끔 끓여주면 바지락 쑥 수제비 완성!</li>
 </ol>
 </dd>
 </dl>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}
