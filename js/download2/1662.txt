function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');

    _blogIsSubmit = true;

    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);

            return html;
        },

            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;

    });
}

 <li># 애호박 영양상식

미네랄을 함유하고 있어

 <h1><b>3.간단한 밑반찬 만들기 - 시금치무침</b>
 <span>by.반짝이마망</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>10분 이내 <b>난이도 :</b>아무나 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/10/19/66bc2f1bb90b26870d9996015f2f1c351.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>시금치 1단, 파 1/3</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>소금 1/3스푼, 마늘 1/3스푼, 깨 약간, 참기름 약간</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>전 요즘 수유중이기 때문에 매운음식이나 고추가루가 들어간 음식은 제한하고 있어요. 
그래서 반찬으로 먹을만한게 마땅치 않은데 그래도 엄마가 종류별로 돌아가면서 만들어주셔서 질리지 않고 먹고있어요~
그래서 오늘은 간단한 밑반찬인 시금치나물 무침을 해봤어요. 재료는: 시금치1단, 파 약간, 소금, 마늘, 깨, 참기름 
만 있으면 되는 간단한 레시피랍니다.
일단 시금치는 뿌리쪽 밑둥을 잘라내고 깨끗하게 씻어주셔야해요.
시금치는 특히 흙이 많이 묻어있는 채소이기 때문에 깨끗한 물에 두세번정도 잘 씻어주세요.</li>
 <li>시금치를 씻기전에 물을 올려놓고 끓여주면 시간이 절약되겠죠?
잘 씻은 시금치는 물이 끓기를 기다리게 대기해 주시구요~~</li>
 <li>끓는 물에 시금치를 퐁당 넣어서 뒤적뒤적 해주세요.
숨이 팍팍 죽고 살짝 익을때까지 한번씩 뒤적여주시구요~</li>
 <li>요정도 익으면 다 익은거예요.
시금치가 금방 익기때문에 시간은 오래걸리지 않아요.
살짝 데쳐서 꺼내주시면돼요.</li>
 <li>꺼내서 차가운 물에 두세번정도 헹궈주세요.</li>
 <li>그리고 물기를 꼭~꼭~ 짜서 커다란 볼에 넣어주세요.
이제 양념을 해서 무쳐야해요!</li>
 <li>소금 1/3 마늘 1/3 (밥스푼기준) 깨소금 왕창 파약간을 넣어주시고</li>
 <li>고소~한 참기름도 넣어주세요.
조물조물 손으로 무쳐주시고 간을 한번 봐서 간이 부족하면 소금간을 더 하시면돼요.
나물종류는 무쳐놓으면 심심해 지니까 약간 삼삼하게 하시면됩니당~!</li>
 <li>맛있는 시금치 나물 완성이예요!
몸에도 좋고 맛도 좋은 시금치무침. 밥반찬으로 딱 좋아요.
고소한 냄새가 솔솔 나는거 같지 않나요?^^</li>
 </ol>
 </dd>
 </dl>

 <h1><b>5.아이들도 좋아하는 쉽고 빠른 밑반찬 감자볶음만들기</b>
 <span>by.심술이네집밥이야기</span>
 <div class="recipe_info">
 <b>분량 :</b>3인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/09/17/fe639e06f6995815f3aef9d517b069ea1.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>감자, 양파, 파프리카</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>대파, 다진마늘, 소금, 후추가루, 참기름, 통깨</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>칼륨이 풍부한 감자는 체내의 나트륨 배출을 도와 고혈압 환자에게도 좋은 식재료이지요~
감자는 바람이 잘 통하는 곳에 보관하여 주시고
사과와 함께 보관을 하시면 감자의 싹이 돋는것을 막아준다고하니 참고하세요^^</li>

 <li>파프리카와 양파는 감자와 같은 굵기로 채썰어주시고
대파는 어슷썰어 준비합니다~</li>
 <li>팬에 올리브유나 식용유를 두르시고
감자를 볶기 시작합니다~</li>
 <li>감자가 반투명하게 익어갈때쯤
다진마늘과 양파를 넣어 볶아주세요~
다진마늘을 처음부터 넣고 볶으시면 마늘이 타버릴수가 있어 색이 예쁘게 나오질 않아요ㅠㅠ</li>
 <li>파프리카와 대파를 넣어주시고
소금으로 간을 합니다~</li>
 <li>가스의 불을 끄시고 후추가루, 참기름, 통깨를 넣어 마무리합니다~</li>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}
