(소금 6T+ 물 600cc) 냄비에 밀가루 5T와 물 200cc를 넣고 끓여 밀가루풀을 만들어주세요. 큰 볼에 밀가루풀을 담아줍니다. 여기에 고추가루 10T를 넣고 섞어주세요. 밀가루풀이나 찹쌀풀은 왜요? 왜? 양념이 열무에 더 착착 붙으라고 밀가루풀이나 찹쌀풀을 만들어 양념합니다. 귀찮으시면 패스~! 홍고추 3개, 대파 3대, 청양고추 3개를 어슷 썰어줍니다. 아까의 고추가루섞은 밀가루풀에 어슷썰은 대파와 고추들을 넣고 다진생강 1T 와 다진마늘 1T, 멸치액젓 2T 를 넣고 섞어 양념을 만들어주세요. 15분을 절인 열무는 물에 한번 헹궈 꼭 짜주세요. 이제 열무를 양념위에 놓고 구석 구석 양념을 발라줍니다. 양념을 묻힌 열무를 통에 차곡 차곡 엇갈리게 넣으세요. 남은 양념.. 을 열무위에 뿌려주세요. 전 2큰술 남았네요.^^ 실온에 3일 놔뒀다가 냉장고에 넣었어요. 얼른 먹고싶어가지고~^^ 맛있는 열무김치 넣고 꽁치찌개 해먹고싶어지네요~! 레시피 상세보기 5.열무김치 담그기 by.예쁜포비 분량 :6인분 이상 조리시간 :60분 이내 난이도 :중급 [주재료] 열무 1단 (2kg), 소금 200g, 물 2L (채소 무게의 10% 정도) [양념 및 소스재료] 홍고추 250g, 양파 200g, 쪽파 100g, 보리밥 80g, 다진 마늘 3큰술, 생수 3컵, 배즙 1컵, 까나리액젓 2/3컵, 매실 청 3큰술, 새우젓 3큰술, 생강 1쪽 [요리순서] 열무는 풋내가 날 수 있으므로 잎이 많은 끝 부분을 살짝 자르고 떡잎은 떼어 낸다. 5cm 길이로 자른 후 살살 흔들어 씻으세요. 물에 소금을 녹여 소금물을 만들어주세요. 열무를 넣고 약 2시간 정도 절인다. 중간에 한번 뒤집어 주세요. 절인 열무는 서너 번 씻은 후 물기를 빼 주세요. 쪽파는 5cm 길이로 자르고 양파는 채썰어 주세요. 홍고추, 배, 액젓, 새우젓, 매실청은 믹서기에 넣고 곱게 갈아 주세요. 김치 담글 때에는 주로 찹쌀 풀을 쑤지만.. 여름김치에는 밀가루 풀이나 보리밥을 이용하면 더 맛있다고 합니다. 전 집에 보리쌀이 많아서 보리밥을 조금 지었습니다. 보리밥에 물 1컵을 믹서기에 넣고 곱게 갈아 주세요. 다진마늘, 다진생강, 믹서기에 간 재료를 잘 섞은 후 쪽파와 양파를 넣어 주세요. 생수 2컵도 더 넣어 잘 섞어 주세요. 물기를 뺀 열무를 넣고 잘 버무려 주시면 끝... 한나절 정도 상온에 두어 익힌 다음 냉장고에 넣어 주세요. 열무김치 담그기 끝~ 레시피 상세보기 " />

    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {
    if (typeof q_link == 'undefined' || q_link == '') {
        q_link = $("#listBlog [id^=q_link_]:checked").val();

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1792&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}
function doChangeBlogSelectType(select_type) {
    if (select_type == 'url') {

        $("#top_blog_select_url").show();
    } else {
        $("#top_blog_select_url").hide();
        $("#top_blog_select_id").show();
    }

}

 <span>by.예쁜포비</span>
