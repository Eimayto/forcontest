(판젤라틴의 경우 물기를 꼭 짜서 넣으세요.) 젤라틴과 섞은 우유를 6에 넣고 잘 저으세요. 거품기 자국이 살짝만 남을 정도로 생크림을 휘핑하세요. 8에 생크림을 넣으세요. 적당하게 자른 통조림 황도를 넣고 섞으세요. 무스링에 팬닝한 다음 냉동실에서 2~3시간 정도 굳히세요. 레시피 상세보기 4.화이트 초코 치즈케이크 by.메기맹이 분량 :6인분 이상 조리시간 :60분 이내 난이도 :중급 [주재료] 크림치즈 250g, 사워크림 50g, 플레인요거트 50g, 설탕 50g, 레몬즙 50g, 판젤라틴 12g, 화이트초콜릿 130g, 생크림 250g 쿠키바닥 : 곡물쿠키 120g, 버터 53g [요리순서] 곡물쿠키는 비닐팩에 넣고 잘게 부순다음 실온의 버터를 넣고 골고루 잘 섞어줍니다. 무스링 바닥을 호일로 막거나 받침을 둔 다음 쿠키바닥을 넣고 꼭꼭꼭 잘다져가면서 펼쳐주고 치즈케이크 반죽하는 동안 냉동에 굳혀줍니다. 실온의 크림치즈를 잘 풀어 크림화 해준 다음설탕을 넣고 잘그락 거리지 않을 정도로 충분히 휘핑해줍니다. 치즈반죽 중 주의하실 점은 젤라틴이 들어가기때문에 반죽온도가 차가우면 절대 안된다는 것입니다. 저는 가끔 뜨거운물 옆에두고 살짝씩 중탕해가면서 반죽온도를 높여주고 재료들도 모두 실온 준비해서 넣고 작업했답니다. 사워크림과 플레인 요거트를 각각 따로넣고 잘 섞어줍니다.한쪽이 다 섞인 다음 다른쪽을 넣고 섞으시구요. 플레인 요거트는 체에 받쳐서 하루정도 뒀다 쓰면 더 좋아요. 사워크림이 따로없을 때는 물기를 뺀 플레인 요거트를 분량만큼 넣으세요. 레몬즙을 넣어 잘섞은 다음 찬물에 불려둔 젤라틴을 물기 꼭짜서 따뜻한 물에 중탕으로 덩어리 없이 녹인 다음 반죽에 넣고 잘 섞어줍니다. 화이트 초콜릿을 중탕으로 잘 녹여서 반죽에 넣고 섞어줍니다. 다른 볼에 차가운 생크림을 60~70% 휘핑해줍니다. 60~70%면 대략 휘핑했을 때 거품기가 지나간 흔적이 생기지만놔두면 흔적이 퍼지는 상태입니다. 반죽을 두번에 나눠 주걱으로 바닥을 긁어서 위로 올리는 큼직한 동작으로 재빨리 섞어줍니다. 한번 넣은게 다 섞인 다음 생크림을 넣으세요. 쿠키바닥에다가 만들어준 치즈 반죽을 부어준 다음 젓가락으로 휘휘 저어서 기포를 빼준 다음냉동에 3시간 정도 굳힌 다음 냉장에 내립니다.냉장에서 하룻밤정도 굳혀줘도 무관합니다. 레시피 상세보기 5.블루베리 치즈케이크 by.서현마미 분량 :2인분 조리시간 :60분 이내 난이도 :중급 [주재료] [20cm 정사각틀 1개분량] 통밀쿠키 90g, 버터 35g, 생블루베리(또는 냉동블루베리) 100g, 설탕 약간 [양념 및 소스재료] [치즈반죽] 크림치즈 350g, 설탕 95g, 달걀 2개, 옥수수전분 30g, 사워크림 100g, 생크림 150g, 레몬즙 1Ts [요리순서] 베이스에 들어가는 통밀쿠키는 비닐에 넣어 밀대로 밀어서 고운 가루로 만들어주세요. 거기에 실온버터를 넣고 주물주물해서 잘 섞어줍니다. 유산지를 깔아둔 틀에 베이스반죽을 넣고 밑면이 편편한 그릇이나 스패츌러로 꾹꾹 눌러주세요. 냉동이나 생블루베리는 설탕에 살짝 굴려서 설탕을 털어내고 깔아둔 베이스반죽 위에 올려줍니다. 그리고 치즈반죽을 만들동안 냉장고에 넣어두세요~ 실온에 둔 크림치즈를 부드럽게 풀어준다음, 설탕을 넣어 충분히 섞어줍니다. 달걀을 하나씩 넣어 그때마다 충분히 섞어주세요~ 체친 옥수수전분을 넣고 섞어줍니다. 실온에 두었던 사워크림을 넣고 고루 섞어줍니다. 생크림을 조금씩 넣어주면서 섞다가 레몬즙을 넣어 섞어주시면 되요. 사진처럼 부드러운 반죽이 되면 완성이예요~ 블루베리 깔아둔 팬에 반죽을 부어 윗면을 평평하게 해주고, 데워둔 물을 베이킹팬에 부어 중탕으로 구워주세요. 처음엔 175℃에서 20분정도 구워주다가 온도를 160℃로 내려서 40~45분정도 구워줍니다. 다 구워지면 틀째로 식혀주세요. 실온에서 식힌다음 냉장고에 3~4시간정도 넣어두어 차게해서 드시면 되요. 드실때 틀에서 꺼내어 유산지를 제거해주시면 된답니다. 레시피 상세보기 " />

(판젤라틴의 경우 물기를 꼭 짜서 넣으세요.) 젤라틴과 섞은 우유를 6에 넣고 잘 저으세요. 거품기 자국이 살짝만 남을 정도로 생크림을 휘핑하세요. 8에 생크림을 넣으세요. 적당하게 자른 통조림 황도를 넣고 섞으세요. 무스링에 팬닝한 다음 냉동실에서 2~3시간 정도 굳히세요. 레시피 상세보기 4.화이트 초코 치즈케이크 by.메기맹이 분량 :6인분 이상 조리시간 :60분 이내 난이도 :중급 [주재료] 크림치즈 250g, 사워크림 50g, 플레인요거트 50g, 설탕 50g, 레몬즙 50g, 판젤라틴 12g, 화이트초콜릿 130g, 생크림 250g 쿠키바닥 : 곡물쿠키 120g, 버터 53g [요리순서] 곡물쿠키는 비닐팩에 넣고 잘게 부순다음 실온의 버터를 넣고 골고루 잘 섞어줍니다. 무스링 바닥을 호일로 막거나 받침을 둔 다음 쿠키바닥을 넣고 꼭꼭꼭 잘다져가면서 펼쳐주고 치즈케이크 반죽하는 동안 냉동에 굳혀줍니다. 실온의 크림치즈를 잘 풀어 크림화 해준 다음설탕을 넣고 잘그락 거리지 않을 정도로 충분히 휘핑해줍니다. 치즈반죽 중 주의하실 점은 젤라틴이 들어가기때문에 반죽온도가 차가우면 절대 안된다는 것입니다. 저는 가끔 뜨거운물 옆에두고 살짝씩 중탕해가면서 반죽온도를 높여주고 재료들도 모두 실온 준비해서 넣고 작업했답니다. 사워크림과 플레인 요거트를 각각 따로넣고 잘 섞어줍니다.한쪽이 다 섞인 다음 다른쪽을 넣고 섞으시구요. 플레인 요거트는 체에 받쳐서 하루정도 뒀다 쓰면 더 좋아요. 사워크림이 따로없을 때는 물기를 뺀 플레인 요거트를 분량만큼 넣으세요. 레몬즙을 넣어 잘섞은 다음 찬물에 불려둔 젤라틴을 물기 꼭짜서 따뜻한 물에 중탕으로 덩어리 없이 녹인 다음 반죽에 넣고 잘 섞어줍니다. 화이트 초콜릿을 중탕으로 잘 녹여서 반죽에 넣고 섞어줍니다. 다른 볼에 차가운 생크림을 60~70% 휘핑해줍니다. 60~70%면 대략 휘핑했을 때 거품기가 지나간 흔적이 생기지만놔두면 흔적이 퍼지는 상태입니다. 반죽을 두번에 나눠 주걱으로 바닥을 긁어서 위로 올리는 큼직한 동작으로 재빨리 섞어줍니다. 한번 넣은게 다 섞인 다음 생크림을 넣으세요. 쿠키바닥에다가 만들어준 치즈 반죽을 부어준 다음 젓가락으로 휘휘 저어서 기포를 빼준 다음냉동에 3시간 정도 굳힌 다음 냉장에 내립니다.냉장에서 하룻밤정도 굳혀줘도 무관합니다. 레시피 상세보기 5.블루베리 치즈케이크 by.서현마미 분량 :2인분 조리시간 :60분 이내 난이도 :중급 [주재료] [20cm 정사각틀 1개분량] 통밀쿠키 90g, 버터 35g, 생블루베리(또는 냉동블루베리) 100g, 설탕 약간 [양념 및 소스재료] [치즈반죽] 크림치즈 350g, 설탕 95g, 달걀 2개, 옥수수전분 30g, 사워크림 100g, 생크림 150g, 레몬즙 1Ts [요리순서] 베이스에 들어가는 통밀쿠키는 비닐에 넣어 밀대로 밀어서 고운 가루로 만들어주세요. 거기에 실온버터를 넣고 주물주물해서 잘 섞어줍니다. 유산지를 깔아둔 틀에 베이스반죽을 넣고 밑면이 편편한 그릇이나 스패츌러로 꾹꾹 눌러주세요. 냉동이나 생블루베리는 설탕에 살짝 굴려서 설탕을 털어내고 깔아둔 베이스반죽 위에 올려줍니다. 그리고 치즈반죽을 만들동안 냉장고에 넣어두세요~ 실온에 둔 크림치즈를 부드럽게 풀어준다음, 설탕을 넣어 충분히 섞어줍니다. 달걀을 하나씩 넣어 그때마다 충분히 섞어주세요~ 체친 옥수수전분을 넣고 섞어줍니다. 실온에 두었던 사워크림을 넣고 고루 섞어줍니다. 생크림을 조금씩 넣어주면서 섞다가 레몬즙을 넣어 섞어주시면 되요. 사진처럼 부드러운 반죽이 되면 완성이예요~ 블루베리 깔아둔 팬에 반죽을 부어 윗면을 평평하게 해주고, 데워둔 물을 베이킹팬에 부어 중탕으로 구워주세요. 처음엔 175℃에서 20분정도 구워주다가 온도를 160℃로 내려서 40~45분정도 구워줍니다. 다 구워지면 틀째로 식혀주세요. 실온에서 식힌다음 냉장고에 3~4시간정도 넣어두어 차게해서 드시면 되요. 드실때 틀에서 꺼내어 유산지를 제거해주시면 된답니다. 레시피 상세보기 " />

function doChangePortal(portal) {
    var header = '';
    var tail = '';

        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {

function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}

}

 <h1><b>2.오븐없이 만드는 [딸기크림치즈케이크]</b>
 <span>by.달콤스윗맘</span>
 <div class="recipe_info">
 <b>분량 :</b>3인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>아무나 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2016/04/06/eca0524d33a56d54e1e8a3a6446067f71.jpg">
 </h1>


 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>딸기는 원하는 만큼의 양을 잘게 썰어 설탕1/2컵에 재워둡니다.</li>
 <li>크림치즈 6큰술을 듬뿍~ 퍼서 잘 풀어 준 다음 생크림 1/2컵을 넣고 잘 섞어줍니다.</li>
 <li>잘 풀어준 크림치즈와 생크림이 만나 되직한 느낌으로 섞입니다.</li>
 <li>설탕에 재워두었던 딸기를 넣고 잘 섞어줍니다.</li>
 <li>아주 여리여리하게 이쁜 핑크~ 딸기크림치즈가 완성됩니다.</li>
 <li>[크러스트 만들기] 단단한 쿠키를 비닐에 넣고 잘게 부숴줍니다. "오레오쿠키를 가운데 크림을 제거하고 사용했는데 바삭한 식감이 참 좋았습니다."</li>
 <li>녹인버터 1큰술을 넣고 잘 섞어줍니다.</li>
 <li>딸기크림치즈케이크를 만들 그릇바닥에 크러스트(버터섞은 쿠키)를 넣고 꾹꾹 눌러 담습니다.</li>
 <li>크러스트위에 딸기크림을 부어줍니다.</li>
 <li>용기에 담은 딸기크림을 골고루 펴 준 다음 랩을 씌워 냉동실에서 6시간 정도 굳혀줍니다. 남은 딸기크림을 빵에 발라먹음 부드러움이~~~ 참 맛있습니다.</li>
 <li>짜잔~~~ 6시간 냉동실에서 굳혀 나 온 [딸기크림치즈케이크]</li>
 <li>그릇에서 딸기크림치즈케이크를 빼 낸 다음 접시에 담고 딸기를 얹어 장식합니다.</li>

 <h1><b>3.상큼한 복숭아 레어치즈케이크</b>
 <span>by.분홍돌고래</span>
 <div class="recipe_info">

 <li>전자렌지에 버터를 녹인 다음 통밀비스킷과 섞으세요. 바닥면에 랩을 씌운 무스링에 꾹꾹 눌러가며 무스링 바닥에 깔아 냉장고에 넣어두세요.</li>

 <li>따뜻하게 데운 우유를 불려둔 젤라틴에 넣고 섞으세요. (판젤라틴의 경우 물기를 꼭 짜서 넣으세요.) 젤라틴과 섞은 우유를 6에 넣고 잘 저으세요.</li>
 <li>거품기 자국이 살짝만 남을 정도로 생크림을 휘핑하세요. 8에 생크림을 넣으세요.</li>
 <li>적당하게 자른 통조림 황도를 넣고 섞으세요. 무스링에 팬닝한 다음 냉동실에서 2~3시간 정도 굳히세요.</li>
 </ol>

 <h1><b>4.화이트 초코 치즈케이크</b>
 <span>by.메기맹이</span>
 <div class="recipe_info">
 <b>분량 :</b>6인분 이상 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>중급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/08/729590d3e6041abf72fc3ff3dd81cb41.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>크림치즈 250g, 사워크림 50g, 플레인요거트 50g, 설탕 50g, 레몬즙 50g, 판젤라틴 12g, 화이트초콜릿 130g, 생크림 250g<br>
쿠키바닥 : 곡물쿠키 120g, 버터 53g</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>곡물쿠키는 비닐팩에 넣고 잘게 부순다음 실온의 버터를 넣고 골고루 잘 섞어줍니다.</li>
 <li>무스링 바닥을 호일로 막거나 받침을 둔 다음 쿠키바닥을 넣고 꼭꼭꼭 잘다져가면서 펼쳐주고 치즈케이크 반죽하는 동안 냉동에 굳혀줍니다.</li>
 <li>실온의 크림치즈를 잘 풀어 크림화 해준 다음설탕을 넣고 잘그락 거리지 않을 정도로 충분히 휘핑해줍니다. 치즈반죽 중 주의하실 점은 젤라틴이 들어가기때문에 반죽온도가 차가우면 절대 안된다는 것입니다. 저는 가끔 뜨거운물 옆에두고 살짝씩 중탕해가면서 반죽온도를 높여주고
재료들도 모두 실온 준비해서 넣고 작업했답니다.</li>
 <li>사워크림과 플레인 요거트를 각각 따로넣고 잘 섞어줍니다.한쪽이 다 섞인 다음 다른쪽을 넣고 섞으시구요. 플레인 요거트는 체에 받쳐서 하루정도 뒀다 쓰면 더 좋아요. 사워크림이 따로없을 때는 물기를 뺀 플레인 요거트를 분량만큼 넣으세요.</li>
 <li>레몬즙을 넣어 잘섞은 다음 찬물에 불려둔 젤라틴을 물기 꼭짜서 따뜻한 물에 중탕으로 덩어리 없이 녹인 다음 반죽에 넣고 잘 섞어줍니다.</li>
 <li>화이트 초콜릿을 중탕으로 잘 녹여서 반죽에 넣고 섞어줍니다.</li>
 <li>다른 볼에 차가운 생크림을 60~70% 휘핑해줍니다. 60~70%면 대략 휘핑했을 때 거품기가 지나간 흔적이 생기지만놔두면 흔적이 퍼지는 상태입니다.</li>
 <li>반죽을 두번에 나눠 주걱으로 바닥을 긁어서 위로 올리는 큼직한 동작으로 재빨리 섞어줍니다. 한번 넣은게 다 섞인 다음 생크림을 넣으세요.</li>
 <li>쿠키바닥에다가 만들어준 치즈 반죽을 부어준 다음 젓가락으로 휘휘 저어서 기포를 빼준 다음냉동에 3시간 정도 굳힌 다음 냉장에 내립니다.냉장에서 하룻밤정도 굳혀줘도 무관합니다.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>5.블루베리 치즈케이크</b>
 <span>by.서현마미</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>중급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/08/b9d3df5da329ffd685c5c45712fb407d.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>[20cm 정사각틀 1개분량] 통밀쿠키 90g, 버터 35g, 생블루베리(또는 냉동블루베리) 100g, 설탕 약간</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>[치즈반죽] 크림치즈 350g, 설탕 95g, 달걀 2개, 옥수수전분 30g, 사워크림 100g, 생크림 150g, 레몬즙 1Ts</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>베이스에 들어가는 통밀쿠키는 비닐에 넣어 밀대로 밀어서 고운 가루로 만들어주세요.</li>
 <li>거기에 실온버터를 넣고 주물주물해서 잘 섞어줍니다.</li>
 <li>유산지를 깔아둔 틀에 베이스반죽을 넣고 밑면이 편편한 그릇이나 스패츌러로 꾹꾹 눌러주세요.</li>
 <li>냉동이나 생블루베리는 설탕에 살짝 굴려서 설탕을 털어내고 깔아둔 베이스반죽 위에 올려줍니다. 그리고 치즈반죽을 만들동안 냉장고에 넣어두세요~</li>
 <li>실온에 둔 크림치즈를 부드럽게 풀어준다음, 설탕을 넣어 충분히 섞어줍니다.</li>
 <li>달걀을 하나씩 넣어 그때마다 충분히 섞어주세요~</li>
 <li>체친 옥수수전분을 넣고 섞어줍니다.</li>
 <li>실온에 두었던 사워크림을 넣고 고루 섞어줍니다.</li>
 <li>생크림을 조금씩 넣어주면서 섞다가 레몬즙을 넣어 섞어주시면 되요.</li>
 <li>사진처럼 부드러운 반죽이 되면 완성이예요~</li>
 <li>블루베리 깔아둔 팬에 반죽을 부어 윗면을 평평하게 해주고, 데워둔 물을 베이킹팬에 부어 중탕으로 구워주세요. 처음엔 175℃에서 20분정도 구워주다가 온도를 160℃로 내려서 40~45분정도 구워줍니다. 다 구워지면 틀째로 식혀주세요. 실온에서 식힌다음 냉장고에 3~4시간정도 넣어두어 차게해서 드시면 되요. 드실때 틀에서 꺼내어 유산지를 제거해주시면 된답니다.</li>
 </ol>
 </dd>
 </dl>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}
