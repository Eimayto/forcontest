쌓아오면서 했던 경험들을 통해 이 레스토랑을 운영하는 데 필요한 모든 수완을 갖추었다는 것을 보여주어야 한다. 대부분의 독립적인 투자자들은 돈보다 팀이 얼마나 잘 갖추어졌는지를 더 중요시하므로 당신이 일을 떠나서 무엇을 가치 있게 여기며 어떤 사람인지에 대해 간접적으로 보여주는 것이 도움된다.6.디자인레스토랑의 느낌, 로고 등을 표현할 수 있는 이미지들을 하나의 보드에 모아서 보여주면, 투자자들이 레스토랑에 대해 좀 더 명확한 그림을 그릴 수 있다. 장작 오븐을 사용할 계획이라면 그 이미지도 포함해라. 당신이 만들어 나갈 브랜드와 비슷한 콘셉트를 가진 다른 레스토랑이 있다면 관련된 작은 기사나 이미지 등을 포함하는 것도 도움이 된다.7.목표 시장누가 당신의 레스토랑에서 식사할 것인가? 주 고객층의 직업은 무엇이며, 나이는 어떻게 되는지, 평균 수입은 어느 정도인지 조사해야 한다. 당신의 목표 고객층에 대해 자세히 설명한 후, 이 고객들에게 당신의 콘셉트가 매력적일 수밖에 없음을 강조한다.8.위치목표 시장에서 언급한 내용과 위치는 밀접한 연관이 있다. 당장 확실히 정해진 위치가 없을 가능성이 더 크지만, 당신이 원하는 장소 주변이 어떤 조건을 갖추고 있는지에 대해 자세히 설명해야 한다. 투자자들이 그 지역에 대해 아무것도 모른다는 가정하에 어떤 사람들이 목표 지역에 사는지에 대한 자세한 설명이 필요하다. 이 부분을 읽은 투자자들이 목표 고객과 당신이 목표로 하는 지역이 딱 맞아떨어진다고 확신하게끔 해야 한다.마음에 둔 정확한 지점이 없다면, 원하는 규모, 유동인구, 주차공간, 고속도로 접근 용이성 그리고 그 외 중요 사항들에 대해 자세히 설명하면 된다.9.시장 전반에 대한 설명레스토랑을 오픈 할 장소의 시장 환경에 관해 설명한다. 큰 의미에서는, 지역의 전반적인 경제 상황은 어떤지, 그곳에 현존하는 레스토랑들이 성공을 이루지 못하고 있다면 당신은 그들과 달리 성공할 수 있는 이유는 무엇인지, 그들이 성공적으로 운영을 잘하고 있다면 당신은 어떻게 그들과 경쟁해 나갈 것인지에 대해 설명한다. 작은 의미에서는, 당신의 직접적인 경쟁자들을 지목하고, 그들과 어떤 차별화를 하여 고객을 끌어올 것인지 설명한다.10.마케팅 &amp; 홍보레스토랑 시장은 점점 더 경쟁이 치열해지고 있다. 오픈 전까지 어떻게 사람들의 관심을 끌 것인지 그리고 개업 후에도 어떻게 그 관심을 유지할 것인지에 대해 설명한다. 마케팅 및 홍보 회사를 고용할 계획이라면, 그들을 소개하고 이 회사를 선택한 이유를 밝힌다. 이 회사가 성공적으로 홍보한 다른 고객들을 예로 든다면 큰 도움이 될 것이다. 스스로 마케팅과 홍보를 책임질 것이라면 소셜 미디어, 홈페이지, 대중매체 등을 이용해 목표 고객의 관심을 집중시킬 수 있는 계획이 있음을 보여주어야 한다.11.전문가 &amp; 컨설턴트추가로 계약을 할 전문가 방법이다. 사업계획서을 쓰는 것 다음으로 중요한 것은 좋은 변호인을 고용하는 것이라고 말하기도 한다. 유능한 변호사들은 이미 수많은 사업계획서들을 읽어 보았고, 제3자 관점에서 사업을 바라볼 수 있기 때문이다. 분명 당신의 레스토랑에 맞는 지속 가능한 경영 구조를 택하는 데 큰 도움이 될 것이다.13.재정이 부분은 전문가인 회계사의 도움을 받는 것을 있는 서비스 방식을 취할 것인가? 아니면 서버들이 직접 손님 앞에 주문한 음식을 놓을 것인가? 다양한 와인들을 와인 메뉴에 올리려 한다면, 소믈리에를 고용할 것인가?만약 당신 레스토랑에 있어 서비스가 그리 중요한 부분이 아니라면 그 이유를 2번 콘셉트 파트에서 간단히 설명하면 된다.5.팀 구성자신을 소개하고 함께 레스토랑을 꾸려나갈 팀원들을 소개한다. 현재까지 경력을 쌓아오면서 했던 경험들을 통해 이 레스토랑을 운영하는 데 필요한 모든 수완을 갖추었다는 것을 보여주어야 한다. 대부분의 독립적인 투자자들은 돈보다 팀이 얼마나 잘 갖추어졌는지를 더 중요시하므로 당신이 일을 떠나서 무엇을 가치 있게 여기며 어떤 사람인지에 대해 간접적으로 보여주는 것이 도움된다.6.디자인레스토랑의 느낌, 로고 등을 표현할 수 있는 이미지들을 하나의 보드에 모아서 보여주면, 투자자들이 레스토랑에 대해 좀 더 명확한 그림을 그릴 수 있다. 장작 오븐을 사용할 계획이라면 그 이미지도 포함해라. 당신이 만들어 나갈 브랜드와 비슷한 콘셉트를 가진 다른 레스토랑이 있다면 관련된 작은 기사나 이미지 등을 포함하는 것도 도움이 된다.7.목표 시장누가 당신의 레스토랑에서 식사할 것인가? 주 고객층의 직업은 무엇이며, 나이는 어떻게 되는지, 평균 수입은 어느 정도인지 조사해야 한다. 당신의 목표 고객층에 대해 자세히 설명한 후, 이 고객들에게 당신의 콘셉트가 매력적일 수밖에 없음을 강조한다.8.위치목표 시장에서 언급한 내용과 위치는 밀접한 연관이 있다. 당장 확실히 정해진 위치가 없을 가능성이 더 크지만, 당신이 원하는 장소 주변이 어떤 조건을 갖추고 있는지에 대해 자세히 설명해야 한다. 투자자들이 그 지역에 대해 아무것도 모른다는 가정하에 어떤 사람들이 목표 지역에 사는지에 대한 자세한 설명이 필요하다. 이 부분을 읽은 투자자들이 목표 고객과 당신이 목표로 하는 지역이 딱 맞아떨어진다고 확신하게끔 해야 한다.마음에 둔 정확한 지점이 없다면, 원하는 규모, 유동인구, 주차공간, 고속도로 접근 용이성 그리고 그 외 중요 사항들에 대해 자세히 설명하면 된다.9.시장 전반에 대한 설명레스토랑을 오픈 할 장소의 시장 환경에 관해 설명한다. 큰 의미에서는, 지역의 전반적인 경제 상황은 어떤지, 그곳에 현존하는 레스토랑들이 성공을 이루지 못하고 있다면 당신은 그들과 달리 성공할 수 있는 이유는 무엇인지, 그들이 성공적으로 운영을 잘하고 있다면 당신은 어떻게 그들과 경쟁해 나갈 것인지에 대해 설명한다. 작은 의미에서는, 당신의 직접적인 경쟁자들을 지목하고, 그들과 어떤 차별화를 하여 고객을 끌어올 것인지 설명한다.10.마케팅 &amp; 홍보레스토랑 시장은 점점 더 경쟁이 치열해지고 있다. 오픈 전까지 어떻게 사람들의 관심을 끌 것인지 그리고 개업 후에도 어떻게 그 관심을 유지할 것인지에 대해 설명한다. 마케팅 및 홍보 회사를 고용할 계획이라면, 그들을 소개하고 이 회사를 선택한 이유를 밝힌다. 이 회사가 성공적으로 홍보한 다른 고객들을 예로 든다면 큰 도움이 될 것이다. 스스로 마케팅과 홍보를 책임질 것이라면 소셜 미디어, 홈페이지, 대중매체 등을 이용해 목표 고객의 관심을 집중시킬 수 있는 계획이 있음을 보여주어야 한다.11.전문가 &amp; 컨설턴트추가로 계약을 할 전문가 및 상담사를 나열한다. 예를 들어:– 회계사– 변호인– 설계자– 디자이너– 건설업자– 홍보 마케팅이들을 통해 받을 서비스, 선택한 이유, 이전에 성공한 사례에 대해 간단히 설명한다.12.경영 구조한눈에 경영 구조를 알 수 있도록 간략하게 써야 하며, 본 구조를 택한 이유도 함께 설명한다. 어떤 경영 체제를 택할지 모르겠다면 변호인과 함께 상의하는 것도 좋은 방법이다. 사업계획서을 쓰는 것 다음으로 중요한 것은 좋은 변호인을 고용하는 것이라고 말하기도 한다. 유능한 변호사들은 이미 수많은 사업계획서들을 읽어 보았고, 제3자 관점에서 사업을 바라볼 수 있기 때문이다. 분명 당신의 레스토랑에 맞는 지속 가능한 경영 구조를 택하는 데 큰 도움이 될 것이다.13.재정이 부분은 전문가인 회계사의 도움을 받는 것을 추천한다. 회계사를 고용할 수 없다면, 레스토랑 재정을 담당한 경험이 풍부한 회계사의 조언을 구하는 정도로 일을 진행하는 것도 좋다. 이들은 레스토랑 재정의 특성을 잘 알고 있어서 사업계획서의 이 파트에서 어떤 설명이 필요하며 당신에게 어떤 질문을 해야 하는지도 잘 알고 있다.실제 재무 계획을 만들기 전에 회계사는 레스토랑의 총 좌석, 고객 일 인당 평균 소비액, 하루에 받을 고객 수 등을 알고 싶어 할 것이다. 이런 수치들은 실제 당신의 콘셉트가 재정상 실현 가능한지를 결정하는 기반이 되므로 실제 수보다는 좀 더 적게 잡는 것이 중요하다.“많은 회계사가 다양한 레스토랑과 일을 해보았다고 말할 것입니다. 하지만 그중에서도 레스토랑 재정에 대한 확고한 전문 지식이 있고, 실질적인 function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';

    }

    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}

margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">레스토랑의 느낌, 로고 등을 표현할 수 있는 이미지들을 하나의 보드에 모아서 보여주면, 투자자들이 레스토랑에 대해 좀 더 명확한 그림을 그릴 수 있다. 장작 오븐을 사용할 계획이라면 그 이미지도 포함해라. 당신이 만들어 나갈 브랜드와 비슷한 콘셉트를 가진 다른 레스토랑이 있다면 관련된 작은 기사나 이미지 등을 포함하는 것도 도움이 된다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">7.목표 시장<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;"></strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">누가 당신의 레스토랑에서 식사할 것인가? 주 고객층의 직업은 무엇이며, 나이는 어떻게 되는지, 평균 수입은 어느 정도인지 조사해야 한다. 당신의 목표 고객층에 대해 자세히 설명한 후, 이 고객들에게 당신의 콘셉트가 매력적일 수밖에 없음을 강조한다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">8.위치<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;"></strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">목표 시장에서 언급한 내용과 위치는 밀접한 연관이 있다. 당장 확실히 정해진 위치가 없을 가능성이 더 크지만, 당신이 원하는 장소 주변이 어떤 조건을 갖추고 있는지에 대해 자세히 설명해야 한다. 투자자들이 그 지역에 대해 아무것도 모른다는 가정하에 어떤 사람들이 목표 지역에 사는지에 대한 자세한 설명이 필요하다. 이 부분을 읽은 투자자들이 목표 고객과 당신이 목표로 하는 지역이 딱 맞아떨어진다고 확신하게끔 해야 한다.<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">마음에 둔 정확한 지점이 없다면, 원하는 규모, 유동인구, 주차공간, 고속도로 접근 용이성 그리고 그 외 중요 사항들에 대해 자세히 설명하면 된다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">9.시장 전반에 대한 설명<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;"></strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">레스토랑을 오픈 할 장소의 시장 환경에 관해 설명한다. 큰 의미에서는, 지역의 전반적인 경제 상황은 어떤지, 그곳에 현존하는 레스토랑들이 성공을 이루지 못하고 있다면 당신은 그들과 달리 성공할 수 있는 이유는 무엇인지, 그들이 성공적으로 운영을 잘하고 있다면 당신은 어떻게 그들과 경쟁해 나갈 것인지에 대해 설명한다. 작은 의미에서는, 당신의 직접적인 경쟁자들을 지목하고, 그들과 어떤 차별화를 하여 고객을 끌어올 것인지 설명한다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">10.마케팅 &amp; 홍보<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;"></strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">레스토랑 시장은 점점 더 경쟁이 치열해지고 있다. 오픈 전까지 어떻게 사람들의 관심을 끌 것인지 그리고 개업 후에도 어떻게 그 관심을 유지할 것인지에 대해 설명한다. 마케팅 및 홍보 회사를 고용할 계획이라면, 그들을 소개하고 이 회사를 선택한 이유를 밝힌다. 이 회사가 성공적으로 홍보한 다른 고객들을 예로 든다면 큰 도움이 될 것이다. 스스로 마케팅과 홍보를 책임질 것이라면 소셜 미디어, 홈페이지, 대중매체 등을 이용해 목표 고객의 관심을 집중시킬 수 있는 계획이 있음을 보여주어야 한다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">11.전문가 &amp; 컨설턴트<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;"></strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">추가로 계약을 할 전문가 및 상담사를 나열한다. 예를 들어:</p><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">– 회계사<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">– 변호인<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">– 설계자<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">– 디자이너<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">– 건설업자<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">– 홍보 마케팅</p><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">이들을 통해 받을 서비스, 선택한 이유, 이전에 성공한 사례에 대해 간단히 설명한다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">12.경영 구조<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;"></strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">한눈에 경영 구조를 알 수 있도록 간략하게 써야 하며, 본 구조를 택한 이유도 함께 설명한다. 어떤 경영 체제를 택할지 모르겠다면 변호인과 함께 상의하는 것도 좋은 방법이다. 사업계획서을 쓰는 것 다음으로 중요한 것은 좋은 변호인을 고용하는 것이라고 말하기도 한다. 유능한 변호사들은 이미 수많은 사업계획서들을 읽어 보았고, 제3자 관점에서 사업을 바라볼 수 있기 때문이다. 분명 당신의 레스토랑에 맞는 지속 가능한 경영 구조를 택하는 데 큰 도움이 될 것이다.</p><h4 style="padding: 0px; margin: 25px 0px 10px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-weight: normal; font-family: Helvetica; line-height: 1em; font-size: 18px; color: rgb(51, 51, 51); background-color: rgb(255, 255, 255);"><span style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(255, 102, 0);"><strong style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">13.재정<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;"></strong></span></h4><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">이 부분은 전문가인 회계사의 도움을 받는 것을 추천한다. 회계사를 고용할 수 없다면, 레스토랑 재정을 담당한 경험이 풍부한 회계사의 조언을 구하는 정도로 일을 진행하는 것도 좋다. 이들은 레스토랑 재정의 특성을 잘 알고 있어서 사업계획서의 이 파트에서 어떤 설명이 필요하며 당신에게 어떤 질문을 해야 하는지도 잘 알고 있다.<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">실제 재무 계획을 만들기 전에 회계사는 레스토랑의 총 좌석, 고객 일 인당 평균 소비액, 하루에 받을 고객 수 등을 알고 싶어 할 것이다. 이런 수치들은 실제 당신의 콘셉트가 재정상 실현 가능한지를 결정하는 기반이 되므로 실제 수보다는 좀 더 적게 잡는 것이 중요하다.<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">“많은 회계사가 다양한 레스토랑과 일을 해보았다고 말할 것입니다. 하지만 그중에서도 레스토랑 재정에 대한 확고한 전문 지식이 있고, 실질적인 재정을 꾸려 갈 수 있는 사람을 고용해야 합니다.”라고 컨설턴트 회사 Kross의 대표 루는 말한다. 준비해야 하는 재정 계획은 아래와 같은 것들이 있다.</p><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">– 향후 3~5년간의 손익 계산 견적서<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">– 손익 분기점 분석<br style="padding: 0px; margin: 0px; outline: none; list-style: none; border: 0px none; box-sizing: border-box;">– 필요한 예상 자본</p><p style="padding: 0px; margin: 0px 0px 20px; outline: none; list-style: none; border: 0px none; box-sizing: border-box; font-family: nanumbarungothic; color: rgb(51, 51, 51); font-size: 13px; line-height: 22px; background-color: rgb(255, 255, 255);">탄탄한 사업계획서은 위의 모든 혹은 항목 대부분을 항상 포함하고 있다. 처음으로 레스토랑을 오픈한다면 다른 레스토랑뿐만 아니라 다른 사업의 사업계획서들도 많이 읽어보고 내용을 어떤 식으로 배치했는지, 어떤 문체로 썼는지, 콘셉트가 얼마나 명확한지 등을 확인해보는 것이 도움된다.</p><p style="font-size: 14px;">· 셰프뉴스에서 보기 : http://chefnews.kr/archives/12296</p>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}
