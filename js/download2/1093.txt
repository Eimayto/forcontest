var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {
    if (typeof q_link == 'undefined' || q_link == '') {
        q_link = $("#listBlog [id^=q_link_]:checked").val();

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1093&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

        $("#top_blog_select_id").show();
    }

}

function doCommentSubmit()
{
    var txtVal = $("#cmtTextarea").val();
    if(!txtVal)
    {
        alert('의견을 입력해주세요.');
        return ;
    }
    if(txtVal.length < 4)
    {
        alert('의견은 4자 이상으로 작성해 주세요.');
        return ;
    }
    $.ajax({
        type: "POST",
        url: "/bbs/ajax.html",
        data : $("#insCmtFrm").serialize(),
        success: function(result) {
            if(result == 'SUCCESS')
            {
                $("#cmtTextarea").val('');
                getCommentList();
                var ccnt = parseInt($("#recipeCommentListCount").html(),10);
                $("#recipeCommentListCount").html(ccnt+1);
            }
            else if(result == 'NO_PARAM')
            {
                alert('파라미터 부족. 다시 확인해 주세요.')
            }
            else
            {
                alert('등록 실패! 잠시후 다시 시도해 주세요.')
            }
        },
        error: function () {
            alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
        }
    });
}

        });
    }
}

            <p>오늘은 소한이다. 소한은 작은 추위라는 뜻이지만 실제로는 일 년 중에 가장 추운 날씨를 보이고 있다. 요즘 같이 밖에서 추운 날씨를 뚫고 집에 들어와 가볍게 스프 한 그릇을 먹으면 몸이 녹아내릴 듯 하다.&nbsp;스프는 출근하기 전에 아침대용으로 먹을 수 있고, 속에 부담스럽지 않은 음식이기에 아무때나 잘 넘어가는 음식이다. 집에서 다양한 재료로 만드는 특별한 스프로 올 겨울을 이겨내보자. </p><p><br></p><p><br></p><p><p> <div class="recipe_wrap">
 <h1><b>1.[고구마요리]고구마 스프 만들기,고구마죽</b>
 <span>by.꽃청춘이주부</span>
 <div class="recipe_info">
 <b>분량 :</b>3인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>아무나 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/12/17/6355ec4bd41a1dc6fe1b4b89daddfc531.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>고구마 4개, 버터 3T, 우유 500ml</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>소금 약간, 파마산치즈 1T, 없어도되고 있으면 간지나는재료: 파슬리가루, 없어도되고 있으면 맛있는재료:설탕or후추</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>재료: 고구마4개,버터 3스푼,우유 500ml ,소금약간,파마산치즈조금

 

없어도되고 있으면 간지나는재료: 파슬리가루

 

없어도되고 있으면 맛있는재료:설탕or후추</li>
 
 <li>고구마스프를 하기전 우선 고구마를 찜기에 넣고 쪄줍니다.</li>
 
 <li>​아시죠?

고구마가 익었는지 안익었는지는 젓가락으로 꾸욱 눌러봐서 쉽게 들어가면 익은거랍니다^^

​</li>
 
 <li>잘익은 고구마를 껍질을 모두 벗겨 줍니다.</li>
 
 <li>버터 3스푼을 준비해주세요.</li>
 
 <li>큰볼에 고구마와 버터를 넣어 마구 으깨줍니다.

 

여기서 버터는 고구마가 뜨거워서 저절로 녹더라구요~</li>
 
 <li>좀더 부드럽게 고구마 스프를 만들기 위해 으깬고구마를 믹서기에 넣어줍니다.</li>
 
 <li>이대로 갈아버리면 퍽퍽해서 잘 갈리지가 않으니까~

우유를 1컵(종이컵기준)을 넣어서 갈아줍니다.</li>
 
 <li>갈아주니 좀더 부드러운 죽처럼 되었어요.</li>
 
 <li>​요걸 다시 냄비에 옮겨줍니다.</li>
 
 <li>남아 있던 우유를 다넣어주세요.</li>
 
 <li>개인의 입맛에 따라 소금으로 간을 해주고,</li>
 
 <li>마지막에 파마산치즈가루를 1스푼정도 뿌려줍니다.</li>
 
 <li>그리고 잘섞어주면~고구마 스프 완성이예요!</li>
 
 <li>여기서 멋좀 부려봐야겠죠?

파슬리 가루를 준비해서 뿌려주세요.</li>
 </ol>
 </dd>

 <h1><b>2.맛없는 브로콜리 브로콜리스프로 맛있게 변신!! </b>
 <span>by.빵굽는쪼꼬맹이</span>
 <div class="recipe_info">
 <b>분량 :</b>1인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>중급 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/11/10/2c2ffdb336cbec4892bbaf1e995625f91.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>​감자 250g(조금 작은사이즈 3개), 양파 90g(반개), 데친 브로콜리160g(한줌) ​</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>우유 1컵(200ml), 생크림 반컵(100ml), 버터1큰술, 선택재료 - 치킨스톡 반개, 파마산치즈가루 2큰술, 후추</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>달군팬에 버터를 한큰술 넣고 녹여 주구요.</li>
 
 <li>감자와 양파는 채썰어 넣고 반투명해 질때까지 볶아주세요.</li>
 
 <li>감자와 양파가 반투명해지면 데쳐놓은 브로콜리는 넣고 잠길 만큼 물을 넣고</li>
 
 <li>치킨스톡 반개를 넣어 주세요.</li>
 
 <li>그리고 뚜껑을 덮고 보글보글 끓여줍니다. 한번씩 저어줘야해요. 안그럼 눌러붙어요. 감자를 눌럿을때 툭툭 부서질 만큼 익히면 되요. 그리고 도깨비 방망이로 윙 갈아주세요. 저는 너무 고운 스프보다 건더기 좀 느껴지는 스프가 좋거든요 그래서 조금은 건더기가 느껴지게 갈았어요. 취향껏 갈아주세요.</li>
 
 <li>우유와 생크림을 넣고 중불에서 보글보글 끓여 줍니다.</li>
 
 <li>다 끓으면 파마산치즈가루롸 후추가루를 톡톡 넣고 마무리 해주세요. 치킨스톡 안넣으신 분들은 이때 소금을 넣어서 간을 맞춰주세요.</li>
 
 <li>완성입니다.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>3.다이어트요리/ 이하늬레시피/ 단호박스프</b>
 <span>by.현짱2</span>
 <div class="recipe_info">
 <b>분량 :</b>3인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>아무나 </div>
 
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/12/23/f733531538e79fc710f334c50e4d5c991.jpg">
 </h1>

 
 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>단호박 1개, 양파 1/2개, 우유 적당량</dd>
 </dl>

 
 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 
 <li>먼저 양파를 잘게 썰어주기! 딱히 크기는 상관없답니당</li>
 
 <li>그다음 단호박을 손질해줄거예요~! 이때 이하늬 씨가 알려준 방법은 단호박을 3분간 전자레인지에 돌려주면 감자칼로 껍질을 벗기기가 쉽다는 정보! 저도 그 방법으로 했어요! 그 다음 숟가락으로 단호박 속을 퍼내서 깔끔하게 비워주기!</li>
 
 <li>그다음 단호박이 잘 익도록 얇게 얇게 썰어주기!</li>
 
 <li>다시 한 번 조각조각 내주세요~</li>
 
 <li>모든 준비가 끝났으면 그다음! 냄비에 기름 또는 버터를 살짝 두르고 양파가 투명해질 때까지 볶아주기!</li>
 
 <li>그다음 단호박을 넣고</li>
 
 <li>우유를 부어주면 끝! 우유의 양은 단호박의 양을 고려해서 넣어서 끓여주다가 농도를 봐주면서 우유를 첨가해주시면 될 것 같아요!</li>
 
 <li>단호박이 익으면서 점차 색이 노래져요!</li>
 
 <li>요로케 요로케 ~ 
이때 참고로 점도가 높아서 막 여기저기 팡팡 튀기기 때문에 나뭇 주걱으로 계속 휘휘 저어주셔야 해요!</li>
 
 <li>계속 끓이다 보니까 농도는 어느 정도 맞춰졌는데 건더기들이 잘게 잘게 남아있어서 핸드믹서로 깔끔하게 한 번 더 갈아주었어요!</li>
 
 <li>한 숟가락 드셔보시고 입맛에 따라 소금으로 간을 해주시면 될 것 같아요! 저는 간을 전혀 안 했는데도 맛있었어요ㅎ
정말 곱게 갈린 단호박! 헤헿 색 너무 예뻐요~</li>
 
 <li>그릇에 담아내면 초간단 다이어트요리 이하늬레시피로 만든 단호박스프 완성! 집에 브로콜리나 파슬리가루가 있다면 데코를 해주세요~</li>
 </ol>
 </dd>

<script>
var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}

$(document).ready(function () {
    getViewRecipeList(1);
})
