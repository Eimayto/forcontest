function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}

 <h1><b>11.콩국수 만드는 법</b>
 <span>by.김성심김치요리1</span>
 <div class="recipe_info">
 <b>분량 :</b>1인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/24/08c2601d651c612ea469e893f8917f3f1.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>검은콩 2컵, ​호두 1컵, 검은깨 3큰술, ​소면 800g​, 건포도, 오이채, 체리</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>소금, 꿀</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>검은콩은 깨끗이 씻어 3배의 물을 부어 물이 끓고 콩이 떠오르기 시작하면서부터 15분 정도 뚜껑을 덮고 삶아준다. 콩은 너무 삶으면 고소한 맛이 없어지니까 적당히 삶아 주는 게 포인트.</li>
 <li>삶은 콩물은 따라 놓고(믹서에 갈 때 사용) 껍질을 제거한다.</li>
 <li>호두는 끓는 물에 살짝 데쳐 물기를 뺀다. 그래야 떪은맛이 없어진다.</li>
 <li>검은깨는 고소하게 볶아 놓는다.</li>
 <li>콩, 호두, 검은깨를 넣고 물과 콩의 비율을 3:1로 잡아 믹서에 곱게 갈아준다. 이 때 삶을 때 나온 콩물도 같이 부어준다.</li>
 <li>콩국은 냉장고에 넣어 시원하게 만든다.</li>
 <li>끓는 물에 소금 1/2큰술을 넣고 소면을 넣어 삶다가 끓어오르면 찬물 1/2컵을 붓고 또 끓어오르면 찬물 1/2컵을 부어 끓여 국수가 익으면(투명하고 맑갛게 되면) 건져 찬물(얼음 물)에 바락바락 여러번 씻어준다.</li>
 <li>소면을 그릇에 담고 진한 콩국물과 얼음을 넣어준다. 고명으로 건포도, 오이채, 체리를 올려주면 진하고 고소한 콩국수 완성. 진하고 고소한 콩국수에 소금이나 꿀을 조금 넣어서 먹으면 더욱 맛있다.</li>
 </ol>
 </dd>

var curVRPage = 1;

function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)

        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;

    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;

    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
