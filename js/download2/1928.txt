var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {
    if (typeof q_link == 'undefined' || q_link == '') {

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1928&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

    }

}

function doCommentSubmit()

 <h1><b>8.계란말이 커틀렛 만들기</b>
 <span>by.모모둥이</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/05/26/7c6d5f96c60e05a4f8787d478d9cb3941.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>스팸 20g, 대파 10g, 양파 30g, 당근 10g 계란 4개, 빵가루</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>소금 찻스푼으로 2스푼</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>스팸20g, 대파10g, 양파30g, 당근10g을 사진처럼 썰어 준비하고, 계란4개를 풀어 준비합니다. 계란물에는 소금을 찻스푼으로 2숟가락 넣었다고 해요.</li>
 <li>후라이팬을 달군 후, 오일을 두르고 계란물을 1/2만 부어준 후에 가장자리가 익어올라오면 한번 휙~ 접어주고 접어진 끝에 나머지 계란물을 다 부어서 이어서 한번 더 말아 계란말이를 완성.</li>
 <li>넙적하고 두톰한 계란말이를 완성하여 계란물 조금 남긴걸 입혀준 다음 파슬리가루 섞은 빵가루를 고루고루 발라 주세요.</li>
 <li>후라이팬에 기름을 부어주는데 잠길정도로 붓지 않아도 되고, 사진처럼 계란말이의 1/2반 담겨도 될 정도로. 그리고 바삭하게 튀겨냅니다.</li>
 </ol>
 </dd>

 <h1><b>9.초간단 든든한 계란말이 국수 </b>
 <span>by.송아송송</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/11/19/eec05aad8591a14dbc3923afa38538bf1.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>새송이버섯 1개, 애호박 1/3개, 계란 3개</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>대파 1/2개, 청양고추 1개, 마늘 2-3개, 참기름 1큰술, 고춧가루 1큰술, 진간장 4큰술, 국간장 1큰술, 설탕 약간, 깨소금 약간</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>육수는 멸치육수 등 취향껏 준비해주세요. 저는 멸치육수에 멸치액젓을 살쩍 넣어서 감칠맛을 더해줬어요. 양념장 재료를 준비해서 양념장을 만들어둡니다. 시간이 많이 걸리는 것들을 미리 준비해두면 조리시간이 단축되어요. 육수와 양념장 준비가 끝났나요? 고명을 만들어 볼게요. 호박과 새송이버섯은 얇게 채썰어둡니다.</li>
 <li>팬에 기름을 살짝 두르고 새송이 버섯을 먼저 볶아요. 소금 간 살짝 해주세요.</li>
 <li>버섯이 어느정도 익었으면 애호박도 같이 넣어줍니다. 소금 간 살짝 해주세요. 간을 겹겹이 입힌다는 느낌으로 약하게 여러번 해주면 간이 적당히 잘 베어요.</li>
 <li>애호박이 숨이 죽고 어느정도 익을때 쯤 깨소금뿌려서 마무리해줍니다.</li>
 <li>완성된 애호박버섯고명은 잠시 식혀둡니다.</li>
 <li>오늘의 주인공 계란말이를 만들어볼거예요 계란 3개에 소금 약간, 멸치액젓 약간 넣어줍니다. 비린내가 잡히고 감칠맛이 생겨요.</li>
 <li>파를 잘라서 넣고 섞어줍니다.</li>
 <li>계란말이 하나 휘리릭 만들어줍니다.</li>
 <li>완성된 계란말이는 식혀서 예쁘게 잘라주세요.</li>
 <li>삶아둔 면을 그릇에 면을 담고 애호박버섯고명과 계란말이를 올려줍니다.</li>
 <li>육수를 부어주면 간단하고 든든한 한끼, 계란말이 국수 완성이예요!</li>
 </ol>
 </dd>
 </dl>

 <h1><b>10.계란말이 두부김밥 </b>
 <span>by.쿡라이프</span>
 <div class="recipe_info">
 <b>분량 :</b>2인분 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/09/01/939cf424be9eb5860a2bc400eae17e9b.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>두부 2/3모, 시금치 30g, 당근 1/4, 게맛살 3개, 계란 2개, 김 한장, 녹말 1큰술, 물 1큰술</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>두부, 소금, 참기름, 깨소금, 집간장(국간장)</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>당근은 채를 썰어 주시고, 게맛살은 큼직큼직하게 뜯어 줍니다.</li>
 <li>두부는 팔팔 끓는 물에 통째로 넣어 1분간 데칩니다. 데친 두부를 면보에 넣어 물기를 꼭 짭니다. 그릇에 두부, 소금, 참기름, 깨소금을 넣고 섞어 줍니다.</li>
 <li>시금치는 끓는 물에 살짝 데쳐서 집간장(또는 국간장), 참기름, 깨소금을 넣고 조물조물 무칩니다.</li>
 <li>당근은 달군 팬에 기름을 두르고 소금을 넣고 살짝 볶습니다.</li>
 <li>위에 두부를 놓고 그 위에 게맛살, 시금치, 당근순으로 올려 줍니다. 그리고 김밥 말듯이 두부도 돌돌 말아 주세요. 김위에 두부 놓는 법 : 김의 반만 두부를 놓아 주시고 양옆으로 1. 5 ~ 1.7cm 여백을 두세요. 그리고 두부를 손으로 가볍게 눌러 주세요.</li>
 <li>계란은 체에 한번 걸러 녹말물(녹말 1큰술 + 물 1큰술)을 넣고 섞습니다. 달군팬에 기름을 조금 두르고 계란을 동그랗게 만듭니다. 계란이 익기 시작할때 말아 두었던 두부김밥을 올리고 계란을 돌돌 말아 주세요.</li>
 <li>계란을 돌돌 말은 두부김밥을 김발로 다시 말아서 한금시킨 후, 한입 크기로 잘라 드시면 됩니다.</li>
 <li>저는 삼각형김밥을 만들고 싶어서 10cm꼬치를 이용해서 삼각대를 만들어 보았습니다. 이렇게 해서 10분간 두면 삼각형이 김밥이 됩니다. 저는 시간을 짧게 두었는데, 시간을 더 두시면 삼각형 모양이 더 확실하게 나온답니다. 삼각대 만드는 방법 : 꼬치 3개를 고무줄 3개를 이용해서 사진 처럼 1번과 2번을 묶습니다. 김발에 고무줄로 묶은 꼬치를 끼워고 나서 3번을 묶으면 끝입니다. 썰때 두부라 모양이 흐트러질수 있으니 조심해서 썰어주세요. 폼나게 담아 맛나게 드시면 OK.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>11.찬밥해결 # 계란말이밥</b>
 <span>by.숑하우스</span>
 <div class="recipe_info">
 <b>분량 :</b>1인분 <b>조리시간 :</b>10분 이내 <b>난이도 :</b>아무나 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2016/07/02/645c7f6d598e9d76bff1f3173cf79da41.jpg">
 </h1>


 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>뜨거운 밥은 지금 퍼서 식혀주어요.
양파, 당근 애호박을 잘게 썰어 달군팬에 소금 한꼬집 넣고 한번 볶아주세요.
* 안 볶고 바로 섞어줘도 되지만 나중에 계란말이 할때만 굽는거라 당근이 넘 안 익더라구요.</li>
 <li>계란 2개(1인분) 풀고 소금 1/2t 간 해주고 밥 1공기 넣고 -&gt;</li>
 <li>볶아둔 채소, 청주 1T, 카레가루 1/2T(생략 가능) 모두 넣고 섞어주세요. 쉐키쉐키</li>
 <li>사각 계란말이 팬 이용했어요.~
구운 팬에 기름을 두르고 달걀밥물 1/2분량을 부어 약불로 얇게 부처 익힌 후 천천히 계란말이처럼 돌돌 말아주세요. 
한번 돌돌 말고 빈 공간에 나머지 1/2 분량을 부어 붙여 또 얇게 천천히 이어 돌돌 말아주세요.
* 달군 팬에 올려야 말때 잘 뭉쳐요.</li>
 <li>한소끔 식힌 후 먹기 좋게 한 입 크기로 잘라주세요.</li>
 </ol>
 </dd>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}

$(document).ready(function () {
    getViewRecipeList(1);
})
</script>
