function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();

            _blogIsSubmit = false;
            return html;
        },

            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}

<br>

닭가슴살 추파춥스 : 닭가슴살 100g, 감자 1개, 청피망 1/2개, 당근 조금<br>
만두 깐풍기 : 작은 만두 한봉지, 다진 양파 2T, 다진 파 2T, 다진 청고추 1T, 다진 홍고추 1T<br>
고구마 또띠아 : 고구마 1개, 청, 홍피망 1/2개, 양파 1/4개, 모짜렐라 치즈 원하는 만큼, 또띠아 2장, 피클</dd>

<br>
[당면 양념] 간장 3T, 설탕 2 + 1/2T<br>
<br>
[만두 깐풍기 양념] 고추기름, 다진마늘 1T, 간장 2T, 매실청 1T(또는 설탕), 물엿 1T, 청주 1T<br>
<br>
[고구마 또띠아 양념]케찹 3T, 매실청 2T(또는 설탕), 간장 1T, 머스타드 1t, 물엿 1t</dd>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>베이컨 계란말이 밥 : 당근, 양파, 애호박은 잘게 썰어 줍니다.</li>
 <li>팬에 기름을 두르고 양파, 당근, 애호박을 볶아 줍니다.</li>
 <li>밥에 당근, 애호박, 양파를 넣고 섞어 줍니다.</li>
 <li>계란은 지단을 구워 베이컨 넓이 보다 조금만 더 크게 썰어 줍니다. 밥을 뭉쳐(지름 2cm 정도의 원통) 계란 지단 위에 놓고 돌돌 말아 줍니다.</li>
 <li>다시 베이컨으로 돌돌 말아 줍니다. 팬에 살짝 구워 주시면 완성입니다.</li>
 <li>버섯 잡채 : 표고버섯은 채를 썰어 간장 1 + 1/2T, 재래간장 1t(또는 국간장), 매실청 2T(또는 설탕) 다진마늘 1/2T, 후추를 넣고 섞어 줍니다.</li>
 <li>느타리버섯은 졸졸 뜯어 줍니다. 양파는 채 썰어 주시고, 고추는 씨를 뺀 다음 채를 썰어 줍니다.</li>
 <li>팬에 표고버섯을 볶아 접시에 담아 둡니다.</li>
 <li>팬에 참기름을 살짝 두르고 양파, 느타리버섯을 넣고 볶다가 양파가 익으면 소금 간을 합니다. 홍고추를 마지막으로 넣어 살짝 한번더 볶은 다음 불을 꺼 줍니다. (일반 기름 식용유나 올리브오일을 사용하시는 것 보다 참기름을 넣고 볶으시면 나중에 드실때 느낌함을 덜하게 된답니다.)</li>
 <li>끓는물에 당면을 6분간 삶은 뒤 깨끗히 헹구어 줍니다. 팬에 당면을 넣고 간장 3T, 설탕 2 + 1/2T을 넣고 면이 꼬들꼬들 할때까지 볶아 줍니다.</li>
 <li>그릇에 볶은 당면, 표고버섯, 채소를 넣고 묻혀 줍니다. 마지막에 통깨 넣고 간이 싱거우시면 간장을 넣지 마시고, 소금을 넣어 간을 맞추어 주세요.</li>
 <li>감자, 닭가슴살 추파춥스 : 감자는 삶거나 쪄 으깨어 줍니다. 닭가슴살은 잘게 다져 줍니다. 당근은 채칼로 벗긴다음 잘게 다져 줍니다. 피망도 잘게 다져 줍니다. 그릇에 감자, 닭가슴살, 당근, 피망, 감자전분가루 3T을 넣고 끈기있게 치대어 줍니다.</li>

 <li>다 구운 감자과 치킨볼을 긴 꼬치에 꽂아 주시면 완성입니다.</li>
 <li>만두 깐풍기 : 팬에 기름을 넉넉하게 두르고 만두를 노릇노릇하게 구워 줍니다.</li>
 <li>다진 양파 2T, 다진 파 2T, 다진 청고추 1T, 다진 홍고추 1T을 준비해 줍니다.</li>
 <li>팬에 고추기름을 넉넉히 두르고 다진마늘 1T과 위에 재료들인 다진양파, 파, 청, 홍고추를 넣고 볶아 줍니다.</li>
 <li>코끝에서 매콤한 냄새가 나면 간장 2T, 매실청 1T(또는 설탕), 물엿 1T, 청주 1T을 넣고 끓어 오르면 만두를 넣고 섞어 주시면 완성입니다.</li>
 <li>고구마는 동글동글하게 썰어 기름에 처음부터 넣고 불 위에서 서서힌 튀겨 줍니다. (달구어진 기름이 아닙니다. 그리고 고구마는 껍질을 벗기고 바로 넣어 주세요. 물에 씻지 않습니다.)</li>
 <li>다 튀겨진 고구마입니다.</li>

 <li>팬에 청, 홍피망, 당근을 넣고 볶다가 양파가 다 익으면 케찹 3T, 매실청 2T(또는 설탕) 간장 1T, 머스타드 1t, 물엿 1t을 넣고 끓어오르면 불을 꺼 줍니다.</li>
 <li>또띠아에 모짜렐라치즈, 양념소, 피클을 가지런히 놓아 줍니다.</li>
 <li>피클 위에 튀겨 놓은 고구마를 가전히 놓고 또띠아를 반으로 접어 줍니다.</li>
 <li>약한불에서 천천히 앞, 뒤로 구워 줍니다.</li>
 </ol>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}

    getViewRecipeList(1);
})
