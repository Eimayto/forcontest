var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {
    if (typeof q_link == 'undefined' || q_link == '') {
        q_link = $("#listBlog [id^=q_link_]:checked").val();

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1813&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}
function doChangeBlogSelectType(select_type) {
    if (select_type == 'url') {
        $("#top_blog_select_id").hide();
        $("#top_blog_select_url").show();
    } else {
        $("#top_blog_select_url").hide();
        $("#top_blog_select_id").show();
    }

}

function doCommentSubmit()
{
    var txtVal = $("#cmtTextarea").val();
    if(!txtVal)
    {
        alert('의견을 입력해주세요.');
        return ;
    }
    if(txtVal.length < 4)
    {
        alert('의견은 4자 이상으로 작성해 주세요.');
        return ;
    }
    $.ajax({
        type: "POST",
        url: "/bbs/ajax.html",
        data : $("#insCmtFrm").serialize(),
        success: function(result) {
            if(result == 'SUCCESS')
            {
                $("#cmtTextarea").val('');
                getCommentList();
                var ccnt = parseInt($("#recipeCommentListCount").html(),10);
                $("#recipeCommentListCount").html(ccnt+1);
            }
            else if(result == 'NO_PARAM')
            {
                alert('파라미터 부족. 다시 확인해 주세요.')
            }
            else
            {
                alert('등록 실패! 잠시후 다시 시도해 주세요.')
            }
        },
        error: function () {
            alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
        }
    });
}

function doCommentDelete(commentSeq)
{
    if(confirm('정말 삭제할까요?'))
    {
        $.ajax({
            type: "GET",
            url: "/bbs/ajax.html",

                    $("#recipeCommentListCount").html(ccnt-1);

            },
            error: function () {
                alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
            }
        });
    }
}

 <b>분량 :</b>6인분 이상 <b>조리시간 :</b>15분 이내 <b>난이도 :</b>초급 </div>

 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>

 <li>고추와 파를 송송 썰어 줍니다.</li>

 <h1><b>3.꼬들꼬들한 오이지 담그는 법</b>
 <span>by.맑음입니다</span>
 <div class="recipe_info">
 <b>분량 :</b>6인분 이상 <b>조리시간 :</b>60분 이내 <b>난이도 :</b>아무나 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/06/25/3f18a94c56233c3bfb51a61e4ab4ad901.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>오이 10개</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>소금 2컵, 설탕 2컵, 식초 2컵​</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>오이 농약을 제거하려 물에 30분 이상을 담가두었다. 이러면 왠만한 농약이 거의 다 빠져나간단 얘기를 오래전 tv에서 봤음. 그리곤 깨끗이 씻어</li>
 <li>채반에 띄엄띄엄 자리 잡아 물기를 말려주었다.</li>
 <li>물기가 다 마르면 오이에 소금 넣고 골고루 섞어 하루쯤 절여주기. 중간에 한번 뒤집어 주면서</li>
 <li>하루가 지나면 오이가 이렇게 낭창하게 휘어진다. 대충 절여졌다는 신호.</li>
 <li>팬에 식초와 설탕을 넣고 설탕이 완전히 녹을때까지 끓여준다.</li>
 <li>그런 다음 뜨거울 때 소금에 절여진 오이에 부어준다.</li>
 <li>누름돌로 눌러주었다. 처음엔 물이 별로 안나왔지만 실온에 일주일정도 두니 물이 흥건하게 고인 것이 제법 맛있게 익은 오이지 내가 솔솔 난다. 이때부터 냉장고에 넣어 필요한만큼 꺼내먹으면 된다.</li>
 </ol>
 </dd>
 </dl>

 <h1><b>4.오이지 담그는 법(오이지무침)</b>
 <span>by.행운77</span>
 <div class="recipe_info">
 <b>분량 :</b>5인분 <b>조리시간 :</b>2시간 이상 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/07/27/e69d72a2037c09cf6f158c6cd42abce61.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>[재료] 오이 3개, 물 5컵, 천일염 1/2컵<br>
 <br>
</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>[오이지3개기준무침 양념] 설탕1t, 고춧가루1T, 참기름, 깨, 다진마늘, 파, 청양고추<br>
<br>
<br>
[양념] 설탕 1t, 고춧가루 1T, 참기름, 깨, 다진마늘, 파, 청양고추</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>물5컵에 굵은소금1/2컵을 넣어 끓인다.</li>
 <li>세척한 오이를 반 잘라 끓는 소금물에 넣고 불을 끈다.
무거운 냄비로 오이를 3시간 이상 눌러준다. 시간이 오래 경과할수록 소금의 양이 더 많아질수록 오이지는 더 잘 만들어져요~</li>
 <li>그런데 그렇게 하면 오이지가 짜져서 물에 담갔다가 소금기 우려내 무쳐내셔야 해요~ 위의 염도로 5시간까지는 물에 담가놓지 않아도 제 입맛에는 맞더라구요~</li>
 <li>오이지를 썰어 맛을 보아 짜면 찬물에 담가 우려낸 후 면보에 싸서 물기 제거한다.</li>
 <li>위의 양념 모~두 넣고 조물~조물~무친다. 부족한 간은 소금으로 한다.​ 
이렇게 속성으로 오이지를 만들어 오이지무침이 완성되었네요~</li>
 </ol>
 </dd>
 </dl>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}
