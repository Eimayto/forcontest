물을 80% 버려주세요. 뜨거우니 조심~ 조심~ 라면스프 반절과 크림치즈 2큰술 넣어주세요. 오메~! 냄새가 끝내줘요~ 아웃* 투움바 파스타 같은 느낌! 그릇에 담고 파슬리를 조금 뿌려주니 이건 완전 스파게티같은 느낌~ 단점은 반 이상 먹고 나니 느끼함이 쫘악 올라 오더라구요. 그 전 까지는 진짜 맛있게 먹었답니다. 레시피 상세보기 5.매콤한 두반장라면볶음 by.민쿡스 분량 :1인분 조리시간 :30분 이내 난이도 :초급 [주재료] 라면 1개, 대파 흰부분 2대, 죽순 작은것 1개, 청고추 1개, 홍고추 1개, 마른홍고추 작은것 3개정도, 마늘 10~15개 [양념 및 소스재료] 두반장 1큰술반, 고추기름 2큰술, 고춧가루 1작은술, 간장 1큰술, 매실액 1큰술, 물 1/3컵, 물녹말 1큰술, 통깨 조금, 소금 조금 [요리순서] 채소를 손질해 먹기좋은 크기로 썰어 줍니다. 고춧가루 1작은술, 간장 1큰술, 매실액 1큰술, 물 1/3컵을 섞어 양념장을 만들어 주세요. 물을 팔팔 끓여 라면을 먼저 삶아 줍니다. 삶아낸 라면은 찬물에 헹궈 물기를 빼둡니다. 달군팬에 고추기름을 두르고 편으로 썬 마늘을 반정도 덜어 노릇하게 튀겨내 키친타월에 올려 기름기를 빼줍니다. 남은 고추기름에 마늘과 마른 홍고추를 넣고 향을 내 볶아줍니다. 소금과 후춧가루를 넣고 볶아주고요. 양파, 죽순, 고추를 넣고 볶다 미리 만들어 두었던 고춧가루 양념을 넣고 고루 섞어 볶아 주세요. 양파빛이 투명해지기 시작하면 두반장을 넣고 계속 볶아 주세요. 계속해서 물녹말을 넣어 주고요. 국물이 자작하게 생기면서 걸죽해지기 시작합니다. 좀 더 끓여가면서 볶아주고요. 간을 보아 싱거우면 소금으로 간을 조금 해주고 찬물에 담궈 물기를 빼주었던 라면을 넣고 고루 볶아 줍니다. 라면은 너무 오래 볶으면 다시 힘이 빠지니까 전체적으로 양념과 섞이면 대파를 넣고 섞어 바로 불을 꺼주었어요 레시피 상세보기 6.맛있고 푸짐한 부대찌개라면 by.자몽의요리 분량 :2인분 조리시간 :15분 이내 난이도 :초급 [요리순서] 청량고추는 어슷썰기 해주시고, 소시지는 + 모양으로 잘라주시고, 스팸은 적당한 크기로 잘라주세요 냄비에 대파30g과 고춧가루2T, 올리브오일2T를 넣어주세요 그리고 볶아주세요 타지 않을 정도로 조금 볶아주세요 볶다가 고추기름이 조금 돈다싶으면 물500ml를 부어주세요 물이 끓기 시작하면 면을 넣어주세요 라면스프도 넣어줍니다 스팸과 소시지도 넣어줍니다 면이 반쯤 익었을때, 계란 하나를 톡 터트려줍니다 계란은 깨트리지 않을꺼예요 청량고추도 넣어주세요 계란이 반쯤 익었을때 불을 꺼주면 됩니다 레시피 상세보기 7.담백한 샐러드라면 레시피 by.숄02 분량 :1인분 조리시간 :15분 이내 난이도 :아무나 [요리순서] 채소는 적당한 크기로 썰고 삶은 계란은 반 잘라 고명을 준비한다. 분량의 소스를 모두 넣고 잘 섞어둔다. 물이 팔팔 끓으면 라면사리를 넣고 4분 정도 익힌다. 이따 완전 차가운 물에 헹구면 면이 좀 딱딱해지니 완전히 익혀준다. 삶은 사리는 찬물에 씻어 준비해둔다. 사리는 물기 탈탈 털어서 그릇에 담고 고명도 올린다. 소스는 뿌리기 전에 다시 한 번 섞어서 샤샥 뿌려주면 완성. 한번에 다 넣으면 짤 수 있으니 조금씩 넣어가며 비빈다. 깨도 뿌려주면 화룡점정! var _blogIsSubmit = false;
function doChangePortal(portal) {
    var header = '';
    var tail = '';
    if (portal == 'Naver') {
        header = 'http://blog.naver.com/';
        tail = '';
    } else if (portal == 'Daum') {
        header = 'http://blog.daum.net/';
        tail = '';
    } else if (portal == 'Tistory') {
        header = 'http://';
        tail = '.tistory.com';
    }
    $("#txtBlogHeader").text(header);
    $("#txtBlogTail").text(tail);
}

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();
        if ($.trim(q_link) == '') {
            alert('URL을 입력해 주세요.');
            $("#q_blogurl").focus();
            return;
        } else {
            setBlogContents(q_link);
        }
     } else {
        getBlogContentsList();
     }
}
function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}
function setBlogContents(q_link) {
    if (typeof q_link == 'undefined' || q_link == '') {
        q_link = $("#listBlog [id^=q_link_]:checked").val();

        if (!q_link) {
            alert('블로그 컨텐츠를 선택하세요.');
            return;
        }
    }
    $("#btnSetBlogContents").html('<span class="fa fa-spinner fa-spin" style="width:' + $("#btnSetBlogContents").width() + 'px"></span>');
    if(fromOther == 'n') document.location.href = '/inbox/insert.html?ref_blog_url=' + encodeURIComponent(q_link);
    else if(fromOther == 'review') document.location.href = '/event/ins_review.html?seq=1821&review_seq=&ref_blog_url=' + encodeURIComponent(q_link);
}

        $("#top_blog_select_url").show();
    } else {
        $("#top_blog_select_url").hide();
        $("#top_blog_select_id").show();
    }

}

function doCommentSubmit()
{
    var txtVal = $("#cmtTextarea").val();
    if(!txtVal)
    {
        alert('의견을 입력해주세요.');
        return ;
    }
    if(txtVal.length < 4)
    {
        alert('의견은 4자 이상으로 작성해 주세요.');
        return ;
    }
    $.ajax({
        type: "POST",
        url: "/bbs/ajax.html",
        data : $("#insCmtFrm").serialize(),
        success: function(result) {
            if(result == 'SUCCESS')
            {
                $("#cmtTextarea").val('');
                getCommentList();
                var ccnt = parseInt($("#recipeCommentListCount").html(),10);
                $("#recipeCommentListCount").html(ccnt+1);
            }
            else if(result == 'NO_PARAM')
            {
                alert('파라미터 부족. 다시 확인해 주세요.')
            }
            else
            {
                alert('등록 실패! 잠시후 다시 시도해 주세요.')
            }
        },
        error: function () {
            alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
        }
    });
}

function doCommentDelete(commentSeq)
{
    if(confirm('정말 삭제할까요?'))
    {
        $.ajax({
            type: "GET",
            url: "/bbs/ajax.html",
            data: "q_mode=deleteComment&commentSeq="+commentSeq,
            success: function(result) {
                if(result == 'SUCCESS')
                {
                    getCommentList();
                    var ccnt = parseInt($("#recipeCommentListCount").html(),10);
                    $("#recipeCommentListCount").html(ccnt-1);
                }
            },
            error: function () {
                alert('의견 조회 실패! 잠시후 다시 시도해 주세요.')
            }
        });
    }
}

 <span>by.민달팽이블로그</span>
 <div class="recipe_info">

 <dt>[주재료]</dt>
 <dd>쪽파, 라면사리, 새우, 마늘, 청양고추, 치즈, 새송이버섯, 파슬리가루</dd>

 <dl class="recipe_cont">

 <li>물이 끓으면 라면사리를 넣고 꼬들꼬들하는 정도까지만 삶아 놓아야 한다. (나중에 후라이팬에서 볶으면서 간이 베어야 하기 때문)</li>
 <li>마늘 향이 나면 버섯을 넣고 볶아주고</li>
 <li>새우도 넣고 볶아준다.</li>
 <li>새우를 넣고 나면 맛술을 살짝 넣어 비린맛을 날려준다.</li>
 <li>고추가루와 후추를 넣고 볶아주고</li>
 <li>꼬들하게 삶아진 면과 우유소스를 부어 볶아주면서 간은 소금간으로 맞춘다.</li>
 <li>그릇에 담아 입맛에 따라 체다치즈도 올려주고 위에 파슬리가루도 뿌려준다. 집에서 만드는 아웃백 투움바파스타 따라하기 완성.</li>
 </ol>
 </dd>

var curVRPage = 1;
var totalVRcnt = 0;
function getViewRecipeList(page)
{
    if($("#rRecipContDivPage_"+page).length > 0)
    {
        $("[id^='rRecipContDivPage_']").hide();
        $("#rRecipContDivPage_"+page).show();
        curVRPage = page;
        return ;
    }
    var rvArr = storageUtil.get('stRecentViewRecipe');
    if(!rvArr || rvArr == 'FAIL')
    {
        $("#recentRecipeDiv").hide();
        return ;
    }
    totalVRcnt = rvArr.length;
    var scale = 6;
    var skip = (page-1) * scale;
    var lastpage = Math.ceil(totalVRcnt/scale);
    var hstr = '';
    hstr += '<div id="rRecipContDivPage_'+page+'" class="row">';
    if(page > 1) hstr += '<a href="javascript:void(0);" class="list_btn_pre" onClick="getViewRecipeList('+(page-1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_l.png" alt="이전"></a>';
    if(page < lastpage) hstr += '<a href="javascript:void(0);" class="list_btn_next" onClick="getViewRecipeList('+(page+1)+')"><img src="http://recipe1.ezmember.co.kr/img/btn_arrow1_r.png" alt="다음"></a>';
    for(var i=skip; i<skip+scale; i++)
    {
        var rvs = rvArr[i];
        if(!rvs) continue;
        hstr += '<div class="col-xs-2">';
        hstr += '<a class="thumbnail" href="/recipe/'+rvs.seq+'"><img src="'+rvs.thumb+'" style="width:180px; height:180px;">';
        hstr += '<div class="caption elipsis_rrtitle"><b>'+rvs.title+'</b></div></a>';
        hstr += '</div>';
    }
    hstr += '</div>';
    if(page != 1 && curVRPage > page) $("#recentRecipeContDiv").prepend(hstr);
    else  $("#recentRecipeContDiv").append(hstr);
    getViewRecipeList(page);
    $(".elipsis_rrtitle").ellipsis({row: 2});
}

    getViewRecipeList(1);
