function doChangePortal(portal) {

function doBlogSelect() {
     var select_type = $(':radio[name="q_blog_select_type"]:checked').val();
     if (select_type == 'url') {
        var q_link = $("#q_blogurl").val();

function getBlogContentsList(page,added_params) {
    if (page == 0) return;
    if (!page) {
        page = 1;
    }
    if ($("#q_blogid").val() == '') {
        alert("블로그 주소를 입력해 주세요.");
        return;
    }
    //var blog_url = $("#txtBlogHeader").text() + $("#q_blogid").val() + $("#txtBlogTail");

    if (_blogIsSubmit) {
        alert("처리중 입니다. 잠시만 기다리세요.");
        return;
    }

    if (page > parseInt($("#q_page").val(),10)) {
        $("#btnBlogNext").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogNext").width()+'px"></span>');
    } else if (page < parseInt($("#q_page").val(),10)) {
        $("#btnBlogPrev").html('<span class="fa fa-spinner fa-spin" style="width:'+$("#btnBlogPrev").width()+'px"></span>');
    }

    $("#q_page").val(page);
    _blogIsSubmit = true;
    var params = 'q_mode=get_blog_contents_list&q_portal='+$("[name=q_portal]:checked").val()+'&q_blogid='+$("#q_blogid").val()+'&q_scal='+$("#q_scal").val()+'&q_page='+page;
    if (added_params) {
        if (added_params.substring(0,1) != '&') params += '&';
        params += added_params;
    }

    $.ajax({
        type: "POST",
        cache: false,
        url: "/common/ajx_common.html",
        data: params,
        success: function(html) {
            $("#divBlogContentsList").html(html);
            $("[id^=spanBlogContentsTitle_]").ellipsis();
            $('#divModalBlogContentsList').modal('show');
            _blogIsSubmit = false;
            return html;
        },
        error: function () {
            alert('예기치 못한 오류로 인해 실패했습니다.');
            _blogIsSubmit = false;
        }
    });
}

 <h1><b>5.백주부님의 오징어볶음 </b>
 <span>by.완소꽃남매</span>
 <div class="recipe_info">
 <b>분량 :</b>4인분 <b>조리시간 :</b>30분 이내 <b>난이도 :</b>초급 </div>
 <img src="http://recipe.ezmember.co.kr/cache/recipe/2015/11/11/759daa126ce2e9ae1ee8d097c5f69f911.jpg">
 </h1>

 <dl class="recipe_cont">
 <dt>[주재료]</dt>
 <dd>오징어 2마리, 양파 1개, 대파 1대, 당근 조금, 양배추 1/4개, 청양고추 1개</dd>
 </dl>
 <dl class="recipe_cont">
 <dt>[양념 및 소스재료]</dt>
 <dd>다진마늘 1스푼, 설탕 1큰술, 고추장 1큰술, 간장 5스푼, 고춧가루 3스푼, 참기름 1스푼, 깨 조금</dd>
 </dl>

 <dl class="recipe_cont">
 <dt>[요리순서]</dt>
 <dd>
 <ol>
 <li>대파는 송송송 썬 것과 어슷썰기 해서 준비해주시고요. 송송송 썬 파는 후에 파기름을 만들 때에 향이 잘 퍼지라고 잘게 썰어야 한다더라고요. 그리고 양파, 당근, 청양고추, 양배추 등 준비하신 채소를 먹기 좋게 썰어주세요.</li>
 <li>오징어도 먹기 좋게 썰어주세요.</li>
 <li>달군 팬에 파를 먼저 넣어둔 다음 식용유 4스푼을 넣어주세요. 불 맛을 내기보다는 탄 맛이 아닌 기름에 향을 입히려고 하는 과정이라고 하네요. 은은하게 퍼지면서 파가 노릇노릇해질 때까지 볶아주세요.</li>
 <li>파가 노릇해지면 썰어둔 오징어를 넣어 볶아주세요. 데친 것이든 아닌 것이든 요때 넣어주어야 한다 해요. 파기름이 잘 섞이도록 쉐킷 쉐킷 해주시고요.</li>
 <li>단맛부터 넣어주어야 한다는 백주부님의 말씀. 설탕 1큰술하고도 조금 더 넣으시던데 저는 그냥 깔끔하게 1큰술 넣어 볶아주었어요.</li>
 <li>다진 마늘부터 1스푼 넣어주고</li>
 <li>고추장 1큰술 넣어주고</li>
 <li>간장 5스푼 넣어주고</li>
 <li>고춧가루 3스푼 순으로 넣어 골고루 볶아주세요. 양념은 짜겠다는 생각이 들 정도로요. 채소에서 단맛 수분이 나오니까 간은 좀 세게 합니다.</li>
 <li>볶아주시다 보면 이렇게 국물이 없이 뻑뻑해지는데요.</li>
 <li>이런 상태에서 채소를 넣으면 더 뭉치고 엉키기 때문에 물을 넣어주어야 한다 해요. 물 반컵 넣어주었어요. 불의 세기에 따라 물을 넣어주면 되는데 본의 아니게 좀 더 넣었다 하여도 느긋하게 졸여가면서 오징어의 감칠맛을 끌어낼 수 있으니 상관이 없다 하네요.</li>
 <li>그렇게 졸여가면서 이 정도면 되겠다 싶을 때 채소를 넣어 익혀주시면 돼요. 채소를 넣으면서 물기가 사라지는데요. 채소가 익으면서 숨이 살짝 죽기 시작하면서 물이 나올 때까지 익혀주세요.</li>
 <li>그렇게 볶아 익혀주고 약불로 줄여 마무리로 참기름을 넣어주는데요. 방송을 보니 엄청 많이 넣어주시더라고요. 근데 저희 집 참기름이 얼마 남지 않은 상태라 1스푼만 사용해주었어요. 마무리로 깨소금도 뿌려주면 완성입니다.</li>
 </ol>
 </dd>

소면도 먹고 밥도 비벼먹음 아주 맛있는 한끼식사로 손색없답니다.</li>
